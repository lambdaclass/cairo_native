16/// Compiles Cairo/Sierra to Native machine code.
17/// Outputs the generated MLIR, and the final shared library.
18///
19/// Supports different types of inputs:
20/// - Cairo project (default)
21/// - Standalone Cairo file (with the --single-file option).
22/// - Standalone Sierra JSON file (with the --sierra-json option).
23///
24/// There is no easy way of executing the compiled shared library, so this
25/// binary is mostly used for debugging compilation.
26///
27/// Exits with 1 if the compilation or run fails, otherwise 0.
28#[derive(Parser, Debug)]
29#[clap(version, verbatim_doc_comment)]
30struct Args {
31    /// The input path to compile.
32    /// By default, it is intrepreted as a Cairo project.
33    path: PathBuf,
34    /// Whether path is a single Cairo file.
35    #[arg(short, long, group = "input")]
36    single_file: bool,
37    /// Whether path is a single Sierra JSON file.
38    #[arg(long, group = "input")]
39    sierra_json: bool,
40    /// Optimization level (Valid: 0, 1, 2, 3).
41    /// Values higher than 3 are considered as 3.
42    #[arg(short = 'O', long, default_value_t = 0)]
43    opt_level: u8,
44    /// The output path for the generated MLIR.
45    #[arg(default_value = "out.mlir")]
46    output_mlir: PathBuf,
47    /// The output path for the shared library.
48    #[cfg_attr(target_os = "macos", arg(default_value = "out.dylib"))]
49    #[cfg_attr(not(target_os = "macos"), arg(default_value = "out.so"))]
50    output_library: PathBuf,
51    /// The compilation statistics path.
52    #[arg(long)]
53    stats_path: Option<PathBuf>,
54}
55
56fn main() -> anyhow::Result<()> {
57    // Configure logging and error handling.
58    tracing::subscriber::set_global_default(
59        FmtSubscriber::builder()
60            .with_env_filter(EnvFilter::from_default_env())
61            .finish(),
62    )?;
63
64    let args = Args::parse();
65
66    // First, we obtain the sierra program. If the --sierra-json file was
67    // given, we read the input as Sierra directly. Otherwise, we compile Cairo
68    // to Sierra.
69    let sierra_program = if args.sierra_json {
70        let sierra_file = File::open(&args.path)?;
71        let program =
72            serde_json::from_reader(sierra_file).context("Failed to read Sierra JSON file.")?;
73        Arc::new(program)
74    } else {
75        // Check if args.path is a file or a directory.
76        check_compiler_path(args.single_file, &args.path)?;
77        cairo_to_sierra(&args.path).context("Failed to compile Cairo to Sierra.")?
78    };
79
80    let mut stats_with_path = args.stats_path.map(|path| (Statistics::default(), path));
81    let stats = stats_with_path.as_mut().map(|v| &mut v.0);
82
83    let pre_compilation_instant = Instant::now();
84
85    // Compile the sierra program into a MLIR module.
86    let native_context = NativeContext::new();
87    let native_module = native_context
88        .compile(
89            &sierra_program,
90            false,
91            Some(Default::default()),
92            clone_option_mut!(stats),
93        )
94        .context("Failed to compile to MLIR.")?;
95    std::fs::write(
96        args.output_mlir,
97        native_module.module().as_operation().to_string(),
98    )
99    .context("Failed to write MLIR output.")?;
100
101    let object_data = module_to_object(
102        native_module.module(),
103        args.opt_level.into(),
104        clone_option_mut!(stats),
105    )
106    .context("Failed to convert MLIR to object.")?;
107    object_to_shared_lib(&object_data, &args.output_library, clone_option_mut!(stats))
108        .context("Failed to write shared library.")?;
109
110    let compilation_time = pre_compilation_instant.elapsed().as_millis();
111    if let Some(&mut ref mut stats) = stats {
112        stats.compilation_total_time_ms = Some(compilation_time);
113    }
114
115    if let Some((stats, stats_path)) = stats_with_path {
116        fs::write(stats_path, serde_json::to_string_pretty(&stats)?)?;
117    }
118
119    Ok(())
120}
