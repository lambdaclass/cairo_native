15/// Compiles Cairo/Sierra to Native machine code.
16/// Outputs the generated MLIR, and the final shared library.
17///
18/// Supports different types of inputs:
19/// - Cairo project (default)
20/// - Standalone Cairo file (with the --single-file option).
21/// - Standalone Sierra JSON file (with the --sierra-json option).
22///
23/// There is no easy way of executing the compiled shared library, so this
24/// binary is mostly used for debugging compilation.
25///
26/// Exits with 1 if the compilation or run fails, otherwise 0.
27#[derive(Parser, Debug)]
28#[clap(version, verbatim_doc_comment)]
29struct Args {
30    /// The input path to compile.
31    /// By default, it is intrepreted as a Cairo project.
32    path: PathBuf,
33    /// Whether path is a single Cairo file.
34    #[arg(short, long, group = "input")]
35    single_file: bool,
36    /// Whether path is a single Sierra JSON file.
37    #[arg(long, group = "input")]
38    sierra_json: bool,
39    /// Optimization level (Valid: 0, 1, 2, 3).
40    /// Values higher than 3 are considered as 3.
41    #[arg(short = 'O', long, default_value_t = 0)]
42    opt_level: u8,
43    /// The output path for the generated MLIR.
44    #[arg(default_value = "out.mlir")]
45    output_mlir: PathBuf,
46    /// The output path for the shared library.
47    #[cfg_attr(target_os = "macos", arg(default_value = "out.dylib"))]
48    #[cfg_attr(not(target_os = "macos"), arg(default_value = "out.so"))]
49    output_library: PathBuf,
50    /// The compilation statistics path.
51    #[arg(long)]
52    stats_path: Option<PathBuf>,
53}
54
55fn main() -> anyhow::Result<()> {
56    // Configure logging and error handling.
57    tracing::subscriber::set_global_default(
58        FmtSubscriber::builder()
59            .with_env_filter(EnvFilter::from_default_env())
60            .finish(),
61    )?;
62
63    let args = Args::parse();
64
65    // First, we obtain the sierra program. If the --sierra-json file was
66    // given, we read the input as Sierra directly. Otherwise, we compile Cairo
67    // to Sierra.
68    let sierra_program = if args.sierra_json {
69        let sierra_file = File::open(&args.path)?;
70        let program =
71            serde_json::from_reader(sierra_file).context("Failed to read Sierra JSON file.")?;
72        Arc::new(program)
73    } else {
74        // Check if args.path is a file or a directory.
75        check_compiler_path(args.single_file, &args.path)?;
76        cairo_to_sierra(&args.path).context("Failed to compile Cairo to Sierra.")?
77    };
78
79    let mut stats_with_path = args.stats_path.map(|path| (Statistics::default(), path));
80    let stats = stats_with_path.as_mut().map(|v| &mut v.0);
81
82    // Compile the sierra program into a MLIR module.
83    let native_context = NativeContext::new();
84    let native_module = native_context
85        .compile(
86            &sierra_program,
87            false,
88            Some(Default::default()),
89            clone_option_mut!(stats),
90        )
91        .context("Failed to compile to MLIR.")?;
92    std::fs::write(
93        args.output_mlir,
94        native_module.module().as_operation().to_string(),
95    )
96    .context("Failed to write MLIR output.")?;
97
98    let object_data = module_to_object(
99        native_module.module(),
100        args.opt_level.into(),
101        clone_option_mut!(stats),
102    )
103    .context("Failed to convert MLIR to object.")?;
104    object_to_shared_lib(&object_data, &args.output_library, stats)
105        .context("Failed to write shared library.")?;
106
107    if let Some((stats, stats_path)) = stats_with_path {
108        fs::write(stats_path, serde_json::to_string_pretty(&stats)?)?;
109    }
110
111    Ok(())
112}
