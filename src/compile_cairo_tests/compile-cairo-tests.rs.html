1use std::{error::Error, fs::File, path::PathBuf};
2
3use cairo_lang_test_plugin::TestsCompilationConfig;
4use cairo_lang_test_runner::TestCompiler;
5use clap::Parser;
6
7/// Compiles Cairo tests to Sierra
8#[derive(Parser, Debug)]
9#[command()]
10struct Args {
11    /// Path to input Cairo program
12    cairo_path: PathBuf,
13    /// Path to output Sierra test compilation
14    sierra_tests_path: PathBuf,
15    /// Whether to compile with the starknet plugin
16    #[arg(long, default_value_t = false)]
17    starknet: bool,
18}
19
20fn main() -> Result<(), Box<dyn Error>> {
21    let args = Args::parse();
22
23    let compiler = TestCompiler::try_new(
24        &args.cairo_path,
25        true,
26        true,
27        TestsCompilationConfig {
28            starknet: args.starknet,
29            contract_declarations: None,
30            contract_crate_ids: None,
31            executable_crate_ids: None,
32            add_statements_functions: false,
33            add_statements_code_locations: false,
34            add_functions_debug_info: false,
35        },
36    )?;
37    let compilation = compiler.build()?;
38
39    let sierra_tests_file = File::create(&args.sierra_tests_path)?;
40    serde_json::to_writer_pretty(sierra_tests_file, &compilation)?;
41
42    Ok(())
43}
