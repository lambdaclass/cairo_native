18/// Compiles a Cairo project and runs all the functions marked as `#[test]`.
19/// Exits with 1 if the compilation or run fails, otherwise 0.
20#[derive(Parser, Debug)]
21#[clap(version, verbatim_doc_comment)]
22struct Args {
23    /// The Cairo project path to compile and run its tests.
24    path: PathBuf,
25    /// Whether path is a single file.
26    #[arg(short, long)]
27    single_file: bool,
28    /// Allows the compilation to succeed with warnings.
29    #[arg(long)]
30    allow_warnings: bool,
31    /// The filter for the tests, running only tests containing the filter string.
32    #[arg(short, long, default_value_t = String::default())]
33    filter: String,
34    /// Should we run ignored tests as well.
35    #[arg(long, default_value_t = false)]
36    include_ignored: bool,
37    /// Should we run only the ignored tests.
38    #[arg(long, default_value_t = false)]
39    ignored: bool,
40    /// Should we add the starknet plugin to run the tests.
41    #[arg(long, default_value_t = false)]
42    starknet: bool,
43    /// Run with JIT or AOT (compiled).
44    #[arg(long, value_enum, default_value_t = RunMode::Jit)]
45    run_mode: RunMode,
46    /// Optimization level, Valid: 0, 1, 2, 3. Values higher than 3 are considered as 3.
47    #[arg(short = 'O', long, default_value_t = 0)]
48    opt_level: u8,
49    /// Compares test result with Cairo VM.
50    #[arg(long, default_value_t = false)]
51    compare_with_cairo_vm: bool,
52}
53
54fn main() -> anyhow::Result<()> {
55    // Parse command-line arguments.
56    let args = Args::parse();
57
58    // Configure logging and error handling.
59    tracing::subscriber::set_global_default(
60        FmtSubscriber::builder()
61            .with_env_filter(EnvFilter::from_default_env())
62            .finish(),
63    )?;
64
65    check_compiler_path(args.single_file, &args.path)?;
66
67    let db = &mut {
68        let mut b = RootDatabase::builder();
69        b.detect_corelib();
70        b.with_cfg(CfgSet::from_iter([Cfg::name("test")]));
71        b.with_default_plugin_suite(test_plugin_suite());
72        if args.starknet {
73            b.with_default_plugin_suite(starknet_plugin_suite());
74        }
75
76        b.build()?
77    };
78
79    let main_crate_inputs = setup_project(db, &args.path)?;
80
81    let db = db.snapshot();
82    let test_crate_ids = main_crate_inputs.clone();
83    let test_config = TestsCompilationConfig {
84        starknet: args.starknet,
85        add_statements_functions: false,
86        add_statements_code_locations: false,
87        contract_declarations: None,
88        contract_crate_ids: None,
89        executable_crate_ids: None,
90    };
91
92    let mut diag_reporter = DiagnosticsReporter::stderr().with_crates(&main_crate_inputs);
93    if args.allow_warnings {
94        diag_reporter = diag_reporter.allow_warnings();
95    }
96
97    let build_test_compilation =
98        compile_test_prepared_db(&db, test_config, test_crate_ids.clone(), diag_reporter)?;
99
100    let (compiled, filtered_out) = filter_test_cases(
101        build_test_compilation,
102        args.include_ignored,
103        args.ignored,
104        args.filter.clone(),
105    );
106
107    let summary = run_tests(
108        compiled.metadata.named_tests,
109        compiled.sierra_program.program,
110        compiled.metadata.function_set_costs,
111        compiled.metadata.contracts_info,
112        RunArgs {
113            run_mode: args.run_mode.clone(),
114            opt_level: args.opt_level,
115            compare_with_vm: args.compare_with_cairo_vm,
116        },
117    )?;
118
119    display_tests_summary(&summary, filtered_out);
120
121    if !summary.failed.is_empty() || !summary.mismatch.is_empty() {
122        bail!("test failed")
123    }
124
125    Ok(())
126}
