20/// Compiles a Cairo project and runs all the functions marked as `#[test]`.
21/// Exits with 1 if the compilation or run fails, otherwise 0.
22#[derive(Parser, Debug)]
23#[clap(version, verbatim_doc_comment)]
24struct Args {
25    /// The Cairo project path to compile and run its tests.
26    path: PathBuf,
27    /// Whether path is a single file.
28    #[arg(short, long)]
29    single_file: bool,
30    /// Allows the compilation to succeed with warnings.
31    #[arg(long)]
32    allow_warnings: bool,
33    /// The filter for the tests, running only tests containing the filter string.
34    #[arg(short, long, default_value_t = String::default())]
35    filter: String,
36    /// Should we run ignored tests as well.
37    #[arg(long, default_value_t = false)]
38    include_ignored: bool,
39    /// Should we run only the ignored tests.
40    #[arg(long, default_value_t = false)]
41    ignored: bool,
42    /// Should we add the starknet plugin to run the tests.
43    #[arg(long, default_value_t = false)]
44    starknet: bool,
45    /// Run with JIT or AOT (compiled).
46    #[arg(long, value_enum, default_value_t = RunMode::Jit)]
47    run_mode: RunMode,
48    /// Optimization level, Valid: 0, 1, 2, 3. Values higher than 3 are considered as 3.
49    #[arg(short = 'O', long, default_value_t = 0)]
50    opt_level: u8,
51    /// Compares test result with Cairo VM.
52    #[arg(long, default_value_t = false)]
53    compare_with_cairo_vm: bool,
54}
55
56fn main() -> anyhow::Result<()> {
57    // Parse command-line arguments.
58    let args = Args::parse();
59
60    // Configure logging and error handling.
61    tracing::subscriber::set_global_default(
62        FmtSubscriber::builder()
63            .with_env_filter(EnvFilter::from_default_env())
64            .finish(),
65    )?;
66
67    check_compiler_path(args.single_file, &args.path)?;
68
69    let db = &mut {
70        let mut b = RootDatabase::builder();
71        b.detect_corelib();
72        b.with_cfg(CfgSet::from_iter([Cfg::name("test")]));
73        b.with_default_plugin_suite(test_plugin_suite());
74        if args.starknet {
75            b.with_default_plugin_suite(starknet_plugin_suite());
76        }
77
78        b.build()?
79    };
80
81    let main_crate_ids = setup_project(db, &args.path)?;
82
83    let db = db.snapshot();
84    let test_crate_ids = main_crate_ids.clone();
85    let test_config = TestsCompilationConfig {
86        starknet: args.starknet,
87        add_statements_functions: false,
88        add_statements_code_locations: false,
89        contract_declarations: None,
90        contract_crate_ids: None,
91        executable_crate_ids: None,
92    };
93
94    let mut diag_reporter = DiagnosticsReporter::stderr().with_crates(&main_crate_ids);
95    if args.allow_warnings {
96        diag_reporter = diag_reporter.allow_warnings();
97    }
98
99    let build_test_compilation =
100        compile_test_prepared_db(&db, test_config, test_crate_ids.clone(), diag_reporter)?;
101
102    let (compiled, filtered_out) = filter_test_cases(
103        build_test_compilation,
104        args.include_ignored,
105        args.ignored,
106        args.filter.clone(),
107    );
108
109    let summary = run_tests(
110        compiled.metadata.named_tests,
111        compiled.sierra_program.program,
112        compiled.metadata.function_set_costs,
113        compiled.metadata.contracts_info,
114        RunArgs {
115            run_mode: args.run_mode.clone(),
116            opt_level: args.opt_level,
117            compare_with_vm: args.compare_with_cairo_vm,
118        },
119    )?;
120
121    display_tests_summary(&summary, filtered_out);
122
123    if !summary.failed.is_empty() || !summary.mismatch.is_empty() {
124        bail!("test failed")
125    }
126
127    Ok(())
128}
