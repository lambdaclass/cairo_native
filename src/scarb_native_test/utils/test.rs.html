25/// Summary data of the ran tests.
26pub struct TestsSummary {
27    pub passed: Vec<String>,
28    pub failed: Vec<String>,
29    pub ignored: Vec<String>,
30    pub failed_run_results: Vec<RunResultValue>,
31}

33/// The result of a ran test.
34struct TestResult {
35    /// The status of the run.
36    status: TestStatus,
37    /// The gas usage of the run if relevant.
38    gas_usage: Option<i64>,
39}

41/// The status of a ran test.
42enum TestStatus {
43    Success,
44    Fail(RunResultValue),
45}

57/// Filter compiled test cases with user provided arguments.
58///
59/// # Arguments
60/// * `compiled` - Compiled test cases with metadata.
61/// * `include_ignored` - Include ignored tests as well.
62/// * `ignored` - Run ignored tests only.
63/// * `filter` - Include only tests containing the filter string.
64/// # Returns
65/// * (`TestCompilation`, `usize`) - The filtered test cases and the number of filtered out cases.
66pub fn filter_test_cases(
67    compiled: TestCompilation,
68    include_ignored: bool,
69    ignored: bool,
70    filter: String,
71) -> (TestCompilation, usize) {
72    let total_tests_count = compiled.metadata.named_tests.len();
73    let named_tests = compiled
74        .metadata
75        .named_tests
76        .into_iter()
77        // Filtering unignored tests in `ignored` mode
78        .filter(|(_, test)| !ignored || test.ignored || include_ignored)
79        .map(|(func, mut test)| {
80            // Un-ignoring all the tests in `include-ignored` and `ignored` mode.
81            if include_ignored || ignored {
82                test.ignored = false;
83            }
84            (func, test)
85        })
86        .filter(|(name, _)| name.contains(&filter))
87        .collect_vec();
88    let filtered_out = total_tests_count - named_tests.len();
89    let tests = TestCompilation {
90        metadata: TestCompilationMetadata {
91            named_tests,
92            ..compiled.metadata
93        },
94        ..compiled
95    };
96    (tests, filtered_out)
97}

99/// Display the summary of the ran tests.
100pub fn display_tests_summary(summary: &TestsSummary, filtered_out: usize) {
101    if summary.failed.is_empty() {
102        println!(
103            "test result: {}. {} passed; {} failed; {} ignored; {filtered_out} filtered out;",
104            "ok".bright_green(),
105            summary.passed.len(),
106            summary.failed.len(),
107            summary.ignored.len()
108        );
109    } else {
110        println!("failures:");
111        for (failure, run_result) in summary
112            .failed
113            .iter()
114            .zip_eq(summary.failed_run_results.clone())
115        {
116            print!("   {failure} - ");
117            match run_result {
118                RunResultValue::Success(_) => {
119                    println!("expected panic but finished successfully.");
120                }
121                RunResultValue::Panic(values) => {
122                    println!("{}", format_for_panic(values.into_iter()));
123                }
124            }
125        }
126        println!();
127    }
128}

130/// Runs the tests and process the results for a summary.
131pub fn run_tests(
132    named_tests: Vec<(String, TestConfig)>,
133    sierra_program: Program,
134    function_set_costs: OrderedHashMap<FunctionId, OrderedHashMap<CostTokenType, i32>>,
135    args: RunArgs,
136) -> anyhow::Result<TestsSummary> {
137    let native_context = NativeContext::new();
138
139    // Compile the sierra program into a MLIR module.
140    let native_module = native_context
141        .compile(&sierra_program, false, Some(Default::default()))
142        .unwrap();
143
144    let native_executor: Box<dyn Fn(_, _, _, &mut StubSyscallHandler) -> _> = match args.run_mode {
145        RunMode::Aot => {
146            let executor =
147                AotNativeExecutor::from_native_module(native_module, args.opt_level.into())?;
148            Box::new(move |function_id, args, gas, syscall_handler| {
149                executor.invoke_dynamic_with_syscall_handler(
150                    function_id,
151                    args,
152                    gas,
153                    syscall_handler,
154                )
155            })
156        }
157        RunMode::Jit => {
158            let executor =
159                JitNativeExecutor::from_native_module(native_module, args.opt_level.into())?;
160            Box::new(move |function_id, args, gas, syscall_handler| {
161                executor.invoke_dynamic_with_syscall_handler(
162                    function_id,
163                    args,
164                    gas,
165                    syscall_handler,
166                )
167            })
168        }
169    };
170
171    let gas_metadata = GasMetadata::new(
172        &sierra_program,
173        Some(MetadataComputationConfig {
174            function_set_costs,
175            linear_ap_change_solver: true,
176            linear_gas_solver: true,
177            skip_non_linear_solver_comparisons: false,
178            compute_runtime_costs: false,
179        }),
180    )
181    .unwrap();
182
183    println!("running {} tests", named_tests.len());
184    let wrapped_summary = Mutex::new(Ok(TestsSummary {
185        passed: vec![],
186        failed: vec![],
187        ignored: vec![],
188        failed_run_results: vec![],
189    }));
190    named_tests
191        .into_iter()
192        .map(
193            |(name, test)| -> anyhow::Result<(String, Option<TestResult>)> {
194                if test.ignored {
195                    return Ok((name, None));
196                }
197                tracing::trace!("running test {name:?}");
198
199                let func = find_function(&sierra_program, name.as_str())?;
200
201                let initial_gas = test.available_gas.map(|x| x.try_into().unwrap());
202
203                let result = native_executor(
204                    &func.id,
205                    &[],
206                    initial_gas,
207                    &mut StubSyscallHandler::default(),
208                )
209                .with_context(|| format!("Failed to run the function `{}`.", name.as_str()))?;
210
211                let run_result = result_to_runresult(&result)?;
212                Ok((
213                    name,
214                    Some(TestResult {
215                        status: match &run_result {
216                            RunResultValue::Success(_) => match test.expectation {
217                                TestExpectation::Success => TestStatus::Success,
218                                TestExpectation::Panics(_) => TestStatus::Fail(run_result),
219                            },
220                            RunResultValue::Panic(value) => match test.expectation {
221                                TestExpectation::Success => TestStatus::Fail(run_result),
222                                TestExpectation::Panics(panic_expectation) => {
223                                    match panic_expectation {
224                                        PanicExpectation::Exact(expected) if value != &expected => {
225                                            TestStatus::Fail(run_result)
226                                        }
227                                        _ => TestStatus::Success,
228                                    }
229                                }
230                            },
231                        },
232                        gas_usage: test
233                            .available_gas
234                            .zip(result.remaining_gas)
235                            .map(|(before, after)| {
236                                before.into_or_panic::<i64>() - after.to_i64().unwrap()
237                            })
238                            .or_else(|| {
239                                gas_metadata
240                                    .initial_required_gas(&func.id)
241                                    .map(|gas| gas.try_into().unwrap())
242                            }),
243                    }),
244                ))
245            },
246        )
247        .for_each(|r| {
248            let mut wrapped_summary = wrapped_summary.lock().unwrap();
249            if wrapped_summary.is_err() {
250                return;
251            }
252            let (name, status) = match r {
253                Ok((name, status)) => (name, status),
254                Err(err) => {
255                    *wrapped_summary = Err(err);
256                    return;
257                }
258            };
259            let summary = wrapped_summary.as_mut().unwrap();
260            let (res_type, status_str, gas_usage) = match status {
261                Some(TestResult {
262                    status: TestStatus::Success,
263                    gas_usage,
264                }) => (&mut summary.passed, "ok".bright_green(), gas_usage),
265                Some(TestResult {
266                    status: TestStatus::Fail(run_result),
267                    gas_usage,
268                }) => {
269                    summary.failed_run_results.push(run_result);
270                    (&mut summary.failed, "fail".bright_red(), gas_usage)
271                }
272                None => (&mut summary.ignored, "ignored".bright_yellow(), None),
273            };
274            if let Some(gas_usage) = gas_usage {
275                println!("test {name} ... {status_str} (gas usage est.: {gas_usage})");
276            } else {
277                println!("test {name} ... {status_str}");
278            }
279            res_type.push(name);
280        });
281    wrapped_summary.into_inner().unwrap()
282}
