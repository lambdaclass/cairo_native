1use super::EvalAction;
2use crate::Value;
3use cairo_lang_sierra::{
4    extensions::{
5        core::{CoreLibfunc, CoreType, CoreTypeConcrete},
6        felt252_dict::Felt252DictConcreteLibfunc,
7        lib_func::SignatureOnlyConcreteLibfunc,
8    },
9    program_registry::ProgramRegistry,
10};
11use cairo_lang_sierra_gas::core_libfunc_cost::{
12    DICT_SQUASH_REPEATED_ACCESS_COST, DICT_SQUASH_UNIQUE_KEY_COST,
13};
14use smallvec::smallvec;
15use std::collections::HashMap;

17pub fn eval(
18    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
19    selector: &Felt252DictConcreteLibfunc,
20    args: Vec<Value>,
21) -> EvalAction {
22    match selector {
23        Felt252DictConcreteLibfunc::New(info) => eval_new(registry, info, args),
24        Felt252DictConcreteLibfunc::Squash(info) => eval_squash(registry, info, args),
25    }
26}
27
28pub fn eval_new(
29    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
30    info: &SignatureOnlyConcreteLibfunc,
31    args: Vec<Value>,
32) -> EvalAction {
33    let [segment_arena @ Value::Unit]: [Value; 1] = args.try_into().unwrap() else {
34        panic!()
35    };
36
37    let type_info = registry
38        .get_type(&info.signature.branch_signatures[0].vars[1].ty)
39        .unwrap();
40    let ty = match type_info {
41        CoreTypeConcrete::Felt252Dict(info) => &info.ty,
42        _ => unreachable!(),
43    };
44
45    EvalAction::NormalBranch(
46        0,
47        smallvec![
48            segment_arena,
49            Value::FeltDict {
50                ty: ty.clone(),
51                data: HashMap::new(),
52                count: 0
53            },
54        ],
55    )
56}
57
58pub fn eval_squash(
59    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
60    _info: &SignatureOnlyConcreteLibfunc,
61    args: Vec<Value>,
62) -> EvalAction {
63    let [range_check @ Value::Unit, Value::U64(gas_builtin), segment_arena @ Value::Unit, Value::FeltDict { ty, data, count }]: [Value; 4] =
64        args.try_into().unwrap()
65    else {
66        panic!();
67    };
68
69    const DICT_GAS_REFUND_PER_ACCESS: u64 =
70        (DICT_SQUASH_UNIQUE_KEY_COST.cost() - DICT_SQUASH_REPEATED_ACCESS_COST.cost()) as u64;
71
72    let refund = count.saturating_sub(data.len() as u64) * DICT_GAS_REFUND_PER_ACCESS;
73    let new_gas_builtin = gas_builtin + refund;
74
75    EvalAction::NormalBranch(
76        0,
77        smallvec![
78            range_check,
79            Value::U64(new_gas_builtin),
80            segment_arena,
81            Value::FeltDict { ty, data, count }
82        ],
83    )
84}
