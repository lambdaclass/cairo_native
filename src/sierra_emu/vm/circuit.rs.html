19fn u384_to_struct(num: BigUint) -> Value {
20    let output_big = num.to_bigint().unwrap();
21
22    let mask: BigInt = BigInt::from_bytes_be(Sign::Plus, &[255; 12]);
23
24    let l0: BigInt = &output_big & &mask;
25    let l1: BigInt = (&output_big >> 96) & &mask;
26    let l2: BigInt = (&output_big >> 192) & &mask;
27    let l3: BigInt = (output_big >> 288) & &mask;
28
29    let range = BigInt::ZERO..(BigInt::from(1) << 96);
30    Value::Struct(vec![
31        Value::BoundedInt {
32            range: range.clone(),
33            value: l0,
34        },
35        Value::BoundedInt {
36            range: range.clone(),
37            value: l1,
38        },
39        Value::BoundedInt {
40            range: range.clone(),
41            value: l2,
42        },
43        Value::BoundedInt { range, value: l3 },
44    ])
45}

47fn struct_to_u384(struct_members: Vec<Value>) -> BigUint {
48    let [Value::U128(l0), Value::U128(l1), Value::U128(l2), Value::U128(l3)]: [Value; 4] =
49        struct_members.try_into().unwrap()
50    else {
51        panic!()
52    };
53
54    let l0 = l0.to_le_bytes();
55    let l1 = l1.to_le_bytes();
56    let l2 = l2.to_le_bytes();
57    let l3 = l3.to_le_bytes();
58
59    BigUint::from_bytes_le(&[
60        l0[0], l0[1], l0[2], l0[3], l0[4], l0[5], l0[6], l0[7], l0[8], l0[9], l0[10],
61        l0[11], //
62        l1[0], l1[1], l1[2], l1[3], l1[4], l1[5], l1[6], l1[7], l1[8], l1[9], l1[10],
63        l1[11], //
64        l2[0], l2[1], l2[2], l2[3], l2[4], l2[5], l2[6], l2[7], l2[8], l2[9], l2[10],
65        l2[11], //
66        l3[0], l3[1], l3[2], l3[3], l3[4], l3[5], l3[6], l3[7], l3[8], l3[9], l3[10],
67        l3[11], //
68    ])
69}

71/// If the value is non-invertible a nullifier is returned instead. A nullifier
72/// is value which satisfies this following equation: num * nullifier â‰¡ 0(modulus).
73fn find_nullifier(num: &BigUint, modulus: &BigUint) -> BigUint {
74    let ExtendedGcd { gcd, .. } = num
75        .to_bigint()
76        .unwrap()
77        .extended_gcd(&modulus.to_bigint().unwrap());
78    let gcd = gcd.to_biguint().unwrap();
79
80    modulus / gcd
81}

83pub fn eval(
84    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
85    selector: &CircuitConcreteLibfunc,
86    args: Vec<Value>,
87) -> EvalAction {
88    match selector {
89        CircuitConcreteLibfunc::AddInput(info) => eval_add_input(registry, info, args),
90        CircuitConcreteLibfunc::Eval(info) => eval_eval(registry, info, args),
91        CircuitConcreteLibfunc::GetDescriptor(info) => eval_get_descriptor(registry, info, args),
92        CircuitConcreteLibfunc::InitCircuitData(info) => {
93            eval_init_circuit_data(registry, info, args)
94        }
95        CircuitConcreteLibfunc::GetOutput(info) => eval_get_output(registry, info, args),
96        CircuitConcreteLibfunc::TryIntoCircuitModulus(info) => {
97            eval_try_into_circuit_modulus(registry, info, args)
98        }
99        CircuitConcreteLibfunc::FailureGuaranteeVerify(info) => {
100            eval_failure_guarantee_verify(registry, info, args)
101        }
102        CircuitConcreteLibfunc::IntoU96Guarantee(info) => {
103            eval_into_u96_guarantee(registry, info, args)
104        }
105        CircuitConcreteLibfunc::U96GuaranteeVerify(info) => {
106            eval_u96_guarantee_verify(registry, info, args)
107        }
108        CircuitConcreteLibfunc::U96LimbsLessThanGuaranteeVerify(info) => {
109            eval_u96_limbs_less_than_guarantee_verify(registry, info, args)
110        }
111        CircuitConcreteLibfunc::U96SingleLimbLessThanGuaranteeVerify(info) => {
112            eval_u96_single_limb_less_than_guarantee_verify(registry, info, args)
113        }
114    }
115}

117fn eval_add_input(
118    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
119    info: &SignatureAndTypeConcreteLibfunc,
120    args: Vec<Value>,
121) -> EvalAction {
122    let [Value::Circuit(mut values), Value::Struct(members)]: [Value; 2] = args.try_into().unwrap()
123    else {
124        panic!()
125    };
126
127    let n_inputs = match registry.get_type(&info.ty).unwrap() {
128        CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(info)) => info.circuit_info.n_inputs,
129        _ => panic!(),
130    };
131
132    values.push(struct_to_u384(members));
133
134    EvalAction::NormalBranch(
135        (values.len() != n_inputs) as usize,
136        smallvec![Value::Circuit(values)],
137    )
138}

140fn eval_eval(
141    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
142    _info: &SignatureAndTypeConcreteLibfunc,
143    _args: Vec<Value>,
144) -> EvalAction {
145    let [add_mod @ Value::Unit, mul_mod @ Value::Unit, _descripctor @ Value::Unit, Value::Circuit(inputs), Value::CircuitModulus(modulus), _, _]: [Value; 7] = _args.try_into().unwrap()
146    else {
147        panic!()
148    };
149    let circ_info = match _registry.get_type(&_info.ty).unwrap() {
150        CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(info)) => &info.circuit_info,
151        _ => todo!(),
152    };
153    let mut outputs = vec![None; 1 + circ_info.n_inputs + circ_info.values.len()];
154    let mut add_gates = circ_info.add_offsets.iter().peekable();
155    let mut mul_gates = circ_info.mul_offsets.iter();
156
157    outputs[0] = Some(BigUint::from(1_u8));
158
159    for (i, input) in inputs.iter().enumerate() {
160        outputs[i + 1] = Some(input.to_owned());
161    }
162
163    let success = loop {
164        while let Some(add_gate) = add_gates.peek() {
165            let lhs = outputs[add_gate.lhs].to_owned();
166            let rhs = outputs[add_gate.rhs].to_owned();
167
168            match (lhs, rhs) {
169                (Some(l), Some(r)) => {
170                    outputs[add_gate.output] = Some((l + r) % &modulus);
171                }
172                (None, Some(r)) => {
173                    let res = match outputs[add_gate.output].to_owned() {
174                        Some(res) => res,
175                        None => break,
176                    };
177                    // if it is a sub_gate the output index is store in lhs
178                    outputs[add_gate.lhs] = Some((res + &modulus - r) % &modulus);
179                }
180                // there aren't enough gates computed for add_gate to compute
181                // the next gate so we need to compute a mul_gate
182                _ => break,
183            };
184
185            add_gates.next();
186        }
187
188        match mul_gates.next() {
189            Some(mul_gate) => {
190                let lhs = outputs[mul_gate.lhs].to_owned();
191                let rhs = outputs[mul_gate.rhs].to_owned();
192
193                match (lhs, rhs) {
194                    (Some(l), Some(r)) => {
195                        outputs[mul_gate.output] = Some((l * r) % &modulus);
196                    }
197                    (None, Some(r)) => {
198                        match r.modinv(&modulus) {
199                            // if it is a inv_gate the output index is store in lhs
200                            Some(r) => outputs[mul_gate.lhs] = Some(r),
201                            // Since we don't calculate CircuitPartialOutputs
202                            // perform an early break
203                            None => {
204                                outputs[mul_gate.lhs] = Some(find_nullifier(&r, &modulus));
205                                break false;
206                            }
207                        }
208                    }
209                    // this state should not be reached since it would mean that
210                    // not all the circuit's inputs where filled
211                    _ => unreachable!(),
212                }
213            }
214            None => break true,
215        }
216    };
217
218    if success {
219        let values = outputs
220            .into_iter()
221            .skip(1 + circ_info.n_inputs)
222            .collect::<Option<Vec<BigUint>>>()
223            .expect("The circuit cannot be calculated");
224
225        EvalAction::NormalBranch(
226            0,
227            smallvec![
228                add_mod,
229                mul_mod,
230                Value::CircuitOutputs {
231                    circuits: values,
232                    modulus
233                }
234            ],
235        )
236    } else {
237        EvalAction::NormalBranch(1, smallvec![add_mod, mul_mod, Value::Unit, Value::Unit])
238    }
239}

241fn eval_get_output(
242    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
243    _info: &ConcreteGetOutputLibFunc,
244    args: Vec<Value>,
245) -> EvalAction {
246    let [Value::CircuitOutputs {
247        circuits: outputs,
248        modulus,
249    }]: [Value; 1] = args.try_into().unwrap()
250    else {
251        panic!()
252    };
253    let circuit_info = match _registry.get_type(&_info.circuit_ty).unwrap() {
254        CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(info)) => &info.circuit_info,
255        _ => todo!(),
256    };
257    let gate_offset = *circuit_info.values.get(&_info.output_ty).unwrap();
258    let output_idx = gate_offset - 1 - circuit_info.n_inputs;
259    let output = outputs[output_idx].to_owned();
260
261    let output_struct = u384_to_struct(output);
262    let modulus = u384_to_struct(modulus);
263
264    EvalAction::NormalBranch(
265        0,
266        smallvec![
267            output_struct.clone(),
268            Value::Struct(vec![output_struct, modulus]),
269        ],
270    )
271}

273fn eval_u96_limbs_less_than_guarantee_verify(
274    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
275    info: &ConcreteU96LimbsLessThanGuaranteeVerifyLibfunc,
276    args: Vec<Value>,
277) -> EvalAction {
278    let [Value::Struct(guarantee)]: [Value; 1] = args.try_into().unwrap() else {
279        panic!()
280    };
281    let limb_count = info.limb_count;
282    let Value::Struct(gate) = &guarantee[0] else {
283        panic!();
284    };
285    let Value::Struct(modulus) = &guarantee[1] else {
286        panic!();
287    };
288    let Value::BoundedInt {
289        value: gate_last_limb,
290        ..
291    } = &gate[limb_count - 1]
292    else {
293        panic!();
294    };
295    let Value::BoundedInt {
296        value: modulus_last_limb,
297        ..
298    } = &modulus[limb_count - 1]
299    else {
300        panic!();
301    };
302    let diff = (modulus_last_limb - gate_last_limb).to_u128().unwrap();
303
304    if diff != 0 {
305        EvalAction::NormalBranch(1, smallvec![Value::U128(diff)])
306    } else {
307        // if there is no diff, build a new guarantee, skipping the last limb
308        let new_gate = Value::Struct(gate[0..limb_count].to_vec());
309        let new_modulus = Value::Struct(modulus[0..limb_count].to_vec());
310
311        EvalAction::NormalBranch(0, smallvec![Value::Struct(vec![new_gate, new_modulus])])
312    }
313}

337fn eval_failure_guarantee_verify(
338    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
339    info: &SignatureOnlyConcreteLibfunc,
340    _args: Vec<Value>,
341) -> EvalAction {
342    let [rc96 @ Value::Unit, mul_mod @ Value::Unit, _, _, _]: [Value; 5] =

344    else {
345        panic!()
346    };
347

354        }
355        _ => panic!(),
356    };
357
358    let zero_u96 = Value::BoundedInt {
359        range: BigInt::zero()..BigInt::one() << 96,
360        value: 0.into(),
361    };
362    // This should be changed with it correct value when we implement this libfunc in native
363    let limbs_struct = Value::Struct(vec![zero_u96; limbs_count]);
364
365    EvalAction::NormalBranch(
366        0,
367        smallvec![
368            rc96,
369            mul_mod,
370            Value::Struct(vec![limbs_struct.clone(), limbs_struct])
371        ],
372    )

406fn eval_try_into_circuit_modulus(
407    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
408    _info: &SignatureOnlyConcreteLibfunc,
409    args: Vec<Value>,
410) -> EvalAction {
411    let [Value::Struct(members)]: [Value; 1] = args.try_into().unwrap() else {
412        panic!()
413    };
414
415    let [Value::BoundedInt {
416        range: r0,
417        value: l0,
418    }, Value::BoundedInt {
419        range: r1,
420        value: l1,
421    }, Value::BoundedInt {
422        range: r2,
423        value: l2,
424    }, Value::BoundedInt {
425        range: r3,
426        value: l3,
427    }]: [Value; 4] = members.try_into().unwrap()
428    else {
429        panic!()
430    };
431    assert_eq!(r0, BigInt::ZERO..(BigInt::from(1) << 96));
432    assert_eq!(r1, BigInt::ZERO..(BigInt::from(1) << 96));
433    assert_eq!(r2, BigInt::ZERO..(BigInt::from(1) << 96));
434    assert_eq!(r3, BigInt::ZERO..(BigInt::from(1) << 96));
435
436    let l0 = l0.to_biguint().unwrap();
437    let l1 = l1.to_biguint().unwrap();
438    let l2 = l2.to_biguint().unwrap();
439    let l3 = l3.to_biguint().unwrap();
440
441    let value = l0 | (l1 << 96) | (l2 << 192) | (l3 << 288);
442
443    if value > BigUint::one() {
444        EvalAction::NormalBranch(0, smallvec![Value::CircuitModulus(value)])
445    } else {
446        EvalAction::NormalBranch(1, smallvec![])
447    }
448}
