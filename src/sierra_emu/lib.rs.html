18pub fn find_entry_point_by_idx(
19    program: &Program,
20    entry_point_idx: usize,
21) -> Option<&GenFunction<StatementIdx>> {
22    program
23        .funcs
24        .iter()
25        .find(|x| x.id.id == entry_point_idx as u64)
26}

28pub fn find_entry_point_by_name<'a>(
29    program: &'a Program,
30    name: &str,
31) -> Option<&'a GenFunction<StatementIdx>> {
32    program
33        .funcs
34        .iter()
35        .find(|x| x.id.debug_name.as_ref().map(|x| x.as_str()) == Some(name))
36}

38// If type is invisible to sierra (i.e. a single element container),
39// finds it's actual concrete type recursively.
40// If not, returns the current type
41pub fn find_real_type(
42    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
43    ty: &ConcreteTypeId,
44) -> ConcreteTypeId {
45    match registry.get_type(ty).unwrap() {
46        cairo_lang_sierra::extensions::core::CoreTypeConcrete::Box(info) => {
47            find_real_type(registry, &info.ty)
48        }
49        cairo_lang_sierra::extensions::core::CoreTypeConcrete::Uninitialized(info) => {
50            find_real_type(registry, &info.ty)
51        }
52        cairo_lang_sierra::extensions::core::CoreTypeConcrete::Span(info) => {
53            find_real_type(registry, &info.ty)
54        }
55        cairo_lang_sierra::extensions::core::CoreTypeConcrete::Snapshot(info) => {
56            find_real_type(registry, &info.ty)
57        }
58        _ => ty.clone(),
59    }
60}
