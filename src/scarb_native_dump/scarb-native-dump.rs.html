1use anyhow::Context;
2use cairo_lang_sierra::program::VersionedProgram;
3use cairo_native::context::NativeContext;
4use melior::ir::operation::OperationPrintingFlags;
5use scarb_metadata::{MetadataCommand, ScarbCommand};
6use std::{env, fs};
7
8mod utils;

10/// Compiles all packages from a Scarb project on the current directory.
11fn main() -> anyhow::Result<()> {
12    let metadata = MetadataCommand::new().inherit_stderr().exec()?;
13
14    // Build only the filtered packages.
15    ScarbCommand::new().arg("build").run()?;
16
17    // Get `target` directory.
18    let profile = env::var("SCARB_PROFILE").unwrap_or("dev".into());
19    let default_target_dir = metadata.runtime_manifest.join("target");
20    let target_dir = metadata
21        .target_dir
22        .clone()
23        .unwrap_or(default_target_dir)
24        .join(profile);
25
26    let native_context = NativeContext::new();
27    for package in metadata.packages.iter() {
28        for target in &package.targets {
29            let file_path = target_dir.join(format!("{}.sierra.json", target.name.clone()));
30
31            if file_path.exists() {
32                let compiled = serde_json::from_str::<VersionedProgram>(
33                    &fs::read_to_string(file_path.clone())
34                        .with_context(|| format!("failed to read file: {file_path}"))?,
35                )
36                .with_context(|| format!("failed to deserialize compiled file: {file_path}"))?;
37
38                // Compile the sierra program into a MLIR module.
39                let native_module = native_context
40                    .compile(
41                        &compiled.into_v1().unwrap().program,
42                        false,
43                        Some(Default::default()),
44                    )
45                    .unwrap();
46
47                // Write the output.
48                let output_str = native_module.module().as_operation().to_string_with_flags(
49                    OperationPrintingFlags::new().enable_debug_info(true, false),
50                )?;
51
52                fs::write(
53                    target_dir.join(format!("{}.mlir", target.name.clone())),
54                    &output_str,
55                )?;
56            }
57        }
58    }
59
60    Ok(())
61}
