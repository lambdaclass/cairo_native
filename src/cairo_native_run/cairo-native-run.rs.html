20#[derive(Clone, Debug, ValueEnum)]
21enum RunMode {
22    Aot,
23    Jit,
24}
25
26/// Command line args parser.
27/// Exits with 1 if the compilation or run fails, otherwise 0.
28#[derive(Parser, Debug)]
29#[clap(version, verbatim_doc_comment)]
30struct Args {
31    /// The Cairo project path to compile and run its tests.
32    path: PathBuf,
33    /// Whether path is a single file.
34    #[arg(short, long)]
35    single_file: bool,
36    /// Allows the compilation to succeed with warnings.
37    #[arg(long)]
38    allow_warnings: bool,
39    /// In cases where gas is available, the amount of provided gas.
40    #[arg(long)]
41    available_gas: Option<u64>,
42    /// Run with JIT or AOT (compiled).
43    #[arg(long, value_enum, default_value_t = RunMode::Jit)]
44    run_mode: RunMode,
45    /// Optimization level, Valid: 0, 1, 2, 3. Values higher than 3 are considered as 3.
46    #[arg(short = 'O', long, default_value_t = 0)]
47    opt_level: u8,
48}
49
50fn main() -> anyhow::Result<()> {
51    // Configure logging and error handling.
52    tracing::subscriber::set_global_default(
53        FmtSubscriber::builder()
54            .with_env_filter(EnvFilter::from_default_env())
55            .finish(),
56    )?;
57
58    let args = Args::parse();
59
60    let mut db = RootDatabase::builder().detect_corelib().build()?;
61    let main_crate_ids = setup_project(&mut db, &args.path)?;
62
63    let sierra_program = compile_prepared_db(
64        &db,
65        main_crate_ids,
66        CompilerConfig {
67            replace_ids: true,
68            ..Default::default()
69        },
70    )?
71    .program;
72
73    let native_context = NativeContext::new();
74
75    // Compile the sierra program into a MLIR module.
76    let native_module = native_context
77        .compile(&sierra_program, false, Some(Default::default()))
78        .unwrap();
79
80    let native_executor: Box<dyn Fn(_, _, _, &mut StubSyscallHandler) -> _> = match args.run_mode {
81        RunMode::Aot => {
82            let executor =
83                AotNativeExecutor::from_native_module(native_module, args.opt_level.into())?;
84            Box::new(move |function_id, args, gas, syscall_handler| {
85                executor.invoke_dynamic_with_syscall_handler(
86                    function_id,
87                    args,
88                    gas,
89                    syscall_handler,
90                )
91            })
92        }
93        RunMode::Jit => {
94            let executor =
95                JitNativeExecutor::from_native_module(native_module, args.opt_level.into())?;
96            Box::new(move |function_id, args, gas, syscall_handler| {
97                executor.invoke_dynamic_with_syscall_handler(
98                    function_id,
99                    args,
100                    gas,
101                    syscall_handler,
102                )
103            })
104        }
105    };
106
107    let gas_metadata =
108        GasMetadata::new(&sierra_program, Some(MetadataComputationConfig::default())).unwrap();
109
110    let func = find_function(&sierra_program, "::main")?;
111
112    let initial_gas = gas_metadata
113        .get_initial_available_gas(&func.id, args.available_gas)
114        .with_context(|| "not enough gas to run")?;
115
116    let mut syscall_handler = StubSyscallHandler::default();
117
118    let result = native_executor(&func.id, &[], Some(initial_gas), &mut syscall_handler)
119        .with_context(|| "Failed to run the function.")?;
120
121    let run_result = result_to_runresult(&result)?;
122
123    match run_result {
124        cairo_lang_runner::RunResultValue::Success(values) => {
125            println!("Run completed successfully, returning {values:?}")
126        }
127        cairo_lang_runner::RunResultValue::Panic(values) => {
128            print!("Run panicked with [");
129            for value in &values {
130                match as_cairo_short_string(value) {
131                    Some(as_string) => print!("{value} ('{as_string}'), "),
132                    None => print!("{value}, "),
133                }
134            }
135            println!("].")
136        }
137    }
138    if let Some(gas) = result.remaining_gas {
139        println!("Remaining gas: {gas}");
140    }
141
142    Ok(())
143}
