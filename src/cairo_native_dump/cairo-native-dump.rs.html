22fn main() -> Result<(), Box<dyn std::error::Error>> {
23    // Parse command-line arguments.
24    let args = CmdLine::parse();
25
26    // Configure logging and error handling.
27    tracing::subscriber::set_global_default(
28        FmtSubscriber::builder()
29            .with_env_filter(EnvFilter::from_default_env())
30            .finish(),
31    )?;
32
33    // Load the program.
34    let context = NativeContext::new();
35    let program = load_program(Path::new(&args.input), args.starknet)?;
36
37    // Compile the program.
38    let module = context.compile(&program, false, Some(Default::default()), None)?;
39
40    // Write the output.
41    let output_str = module
42        .module()
43        .as_operation()
44        .to_string_with_flags(OperationPrintingFlags::new().enable_debug_info(true, false))?;
45    match args.output {
46        CompilerOutput::Stdout => println!("{output_str}"),
47        CompilerOutput::Path(path) => fs::write(path, &output_str)?,
48    }
49
50    Ok(())
51}

53fn load_program(path: &Path, is_contract: bool) -> Result<Program, Box<dyn std::error::Error>> {
54    Ok(match path.extension().and_then(OsStr::to_str) {
55        Some("cairo") if !is_contract => {
56            let mut db = RootDatabase::builder().detect_corelib().build()?;
57            let main_crate_ids = setup_project(&mut db, path)?;
58
59            compile_prepared_db(
60                &db,
61                main_crate_ids,
62                CompilerConfig {
63                    replace_ids: true,
64                    ..Default::default()
65                },
66            )?
67            .program
68        }
69        Some("cairo") if is_contract => {
70            // mimics cairo_lang_starknet::contract_class::compile_path
71            let mut plugins = PluginSuite::default();
72            plugins
73                .add_plugin::<StarknetPlugin>()
74                .add_inline_macro_plugin_ex(SelectorMacro::NAME, Arc::new(SelectorMacro));
75            let mut db = RootDatabase::builder()
76                .detect_corelib()
77                .with_default_plugin_suite(plugins)
78                .build()?;
79
80            let main_crate_ids = setup_project(&mut db, Path::new(&path))?;
81
82            let contract = compile_contract_in_prepared_db(
83                &db,
84                None,
85                main_crate_ids,
86                CompilerConfig {
87                    replace_ids: true,
88                    ..Default::default()
89                },
90            )?;
91
92            contract.extract_sierra_program()?
93        }
94        Some("sierra") => {
95            let program_src = fs::read_to_string(path)?;
96
97            let program_parser = ProgramParser::new();
98
99            program_parser
100                .parse(&program_src)
101                .map_err(|e| e.map_token(|t| t.to_string()))?
102        }
103        _ => unreachable!(),
104    })
105}

107#[derive(Clone, Debug, Parser)]
108struct CmdLine {
109    #[clap(value_parser = parse_input)]
110    input: PathBuf,
111
112    #[clap(short = 'o', long = "output", value_parser = parse_output, default_value = "-")]
113    output: CompilerOutput,
114
115    /// Compile a starknet contract
116    #[clap(long)]
117    starknet: bool,
118}
119
120#[derive(Clone, Debug)]
121enum CompilerOutput {
122    Stdout,
123    Path(PathBuf),
124}
125
126fn parse_input(input: &str) -> Result<PathBuf, String> {
127    Ok(match Path::new(input).extension().and_then(OsStr::to_str) {
128        Some("cairo" | "sierra") => input.into(),
129        _ => {
130            return Err(
131                "Input path expected to have either `cairo` or `sierra` as its extension."
132                    .to_string(),
133            )
134        }
135    })
136}
137
138fn parse_output(input: &str) -> Result<CompilerOutput, String> {
139    Ok(if input == "-" {
140        CompilerOutput::Stdout
141    } else {
142        CompilerOutput::Path(match Path::new(input).extension().and_then(OsStr::to_str) {
143            Some("mlir") => input.into(),
144            _ => {
145                return Err(
146                    "Output path expected to be `-` for stdout or have `mlir` as its extension."
147                        .to_string(),
148                )
149            }
150        })
151    })
