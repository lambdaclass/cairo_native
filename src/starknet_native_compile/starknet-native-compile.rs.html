10/// Given a Sierra file (as saved in Starknet's contract tree), extracts the sierra_program from
11/// felts into readable Sierra code, compiles it to native, and saves the result to the given output
12/// path.
13#[derive(Parser, Debug)]
14#[clap(version, verbatim_doc_comment)]
15struct Args {
16    /// The path of the Sierra file to compile.
17    path: PathBuf,
18    /// Optimization level, Valid: 0, 1, 2, 3. Values higher than 3 are considered as 3.
19    #[arg(short = 'O', long, default_value_t = 0)]
20    opt_level: u8,
21    /// The output file path.
22    output: PathBuf,
23}
24
25fn main() -> anyhow::Result<()> {
26    let args = Args::parse();
27    let (contract_class, sierra_program, sierra_version) =
28        load_sierra_program_from_file(&args.path)?;
29
30    AotContractExecutor::new_into(
31        &sierra_program,
32        &contract_class.entry_points_by_type,
33        sierra_version,
34        args.output.clone(),
35        args.opt_level.into(),
36    )
37    .context("Error compiling Sierra program.")?
38    .with_context(|| format!("Failed to take lock on path {}", args.output.display()))?;
39    Ok(())
40}

42/// Extracts the first 3 felts from the Sierra program and parses them into a VersionId.
43fn get_sierra_version_from_program<F>(sierra_program: &[F]) -> anyhow::Result<VersionId>
44where
45    F: TryInto<usize> + std::fmt::Display + Clone,
46    <F as TryInto<usize>>::Error: std::fmt::Display,
47{
48    if sierra_program.len() < 3 {
49        bail!("Sierra program length must be at least 3 Felts.");
50    }
51
52    let version_components: Vec<usize> = sierra_program
53        .iter()
54        .take(3)
55        .enumerate()
56        .map(|(index, felt)| {
57            felt.clone().try_into().map_err(|err| {
58                anyhow!(
59                    "Failed to parse Sierra program to Sierra version. Index: {}, Felt: {}, \
60                        Error: {}",
61                    index,
62                    felt,
63                    err
64                )
65            })
66        })
67        .collect::<Result<_, _>>()?;
68
69    Ok(VersionId {
70        major: version_components[0],
71        minor: version_components[1],
72        patch: version_components[2],
73    })
74}

76/// Given a Sierra file path, loads the contract class from it, extracts the sierra version from the
77/// first 3 felts of the compressed sierra_program, and extracts the compressed sierra_program into
78/// readable sierra code.
79fn load_sierra_program_from_file(
80    path: &PathBuf,
81) -> anyhow::Result<(ContractClass, Program, VersionId)> {
82    let raw_contract_class = std::fs::read_to_string(path).context("Error reading Sierra file.")?;
83
84    let contract_class: ContractClass = serde_json::from_str(&raw_contract_class)
85        .context("Error deserializing Sierra file into contract class.")?;
86    let raw_sierra_program: Vec<_> = contract_class
87        .sierra_program
88        .iter()
89        .map(|big_uint_as_hex| big_uint_as_hex.value.clone())
90        .collect();
91
92    let sierra_version = get_sierra_version_from_program(&raw_sierra_program)?;
93    Ok((
94        contract_class.clone(),
95        contract_class
96            .extract_sierra_program()
97            .context("Error extracting Sierra program from contract class.")?,
98        sierra_version,
99    ))
100}
