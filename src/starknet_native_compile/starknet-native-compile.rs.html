12/// Given a Sierra file (as saved in Starknet's contract tree), extracts the sierra_program from
13/// felts into readable Sierra code, compiles it to native, and saves the result to the given output
14/// path.
15#[derive(Parser, Debug)]
16#[clap(version, verbatim_doc_comment)]
17struct Args {
18    /// The path of the Sierra file to compile.
19    path: PathBuf,
20    /// Optimization level, Valid: 0, 1, 2, 3. Values higher than 3 are considered as 3.
21    #[arg(short = 'O', long, default_value_t = 0)]
22    opt_level: u8,
23    /// The output file path.
24    output: PathBuf,
25
26    #[arg(long)]
27    /// Output path for compilation statistics
28    stats: Option<PathBuf>,
29}
30
31fn main() -> anyhow::Result<()> {
32    let args = Args::parse();
33    let (contract_class, sierra_program, sierra_version) =
34        load_sierra_program_from_file(&args.path)?;
35
36    let mut stats_with_path = args.stats.map(|path| (Statistics::default(), path));
37    let stats = stats_with_path.as_mut().map(|v| &mut v.0);
38
39    AotContractExecutor::new_into(
40        &sierra_program,
41        &contract_class.entry_points_by_type,
42        sierra_version,
43        args.output.clone(),
44        args.opt_level.into(),
45        stats,
46    )
47    .context("Error compiling Sierra program.")?
48    .with_context(|| format!("Failed to take lock on path {}", args.output.display()))?;
49
50    if let Some((stats, path)) = stats_with_path {
51        fs::write(path.with_extension("json"), serde_json::to_string(&stats)?)?;
52    }
53
54    Ok(())
55}

57/// Extracts the first 3 felts from the Sierra program and parses them into a VersionId.
58fn get_sierra_version_from_program<F>(sierra_program: &[F]) -> anyhow::Result<VersionId>
59where
60    F: TryInto<usize> + std::fmt::Display + Clone,
61    <F as TryInto<usize>>::Error: std::fmt::Display,
62{
63    if sierra_program.len() < 3 {
64        bail!("Sierra program length must be at least 3 Felts.");
65    }
66
67    let version_components: Vec<usize> = sierra_program
68        .iter()
69        .take(3)
70        .enumerate()
71        .map(|(index, felt)| {
72            felt.clone().try_into().map_err(|err| {
73                anyhow!(
74                    "Failed to parse Sierra program to Sierra version. Index: {}, Felt: {}, \
75                        Error: {}",
76                    index,
77                    felt,
78                    err
79                )
80            })
81        })
82        .collect::<Result<_, _>>()?;
83
84    Ok(VersionId {
85        major: version_components[0],
86        minor: version_components[1],
87        patch: version_components[2],
88    })
89}

91/// Given a Sierra file path, loads the contract class from it, extracts the sierra version from the
92/// first 3 felts of the compressed sierra_program, and extracts the compressed sierra_program into
93/// readable sierra code.
94fn load_sierra_program_from_file(
95    path: &PathBuf,
96) -> anyhow::Result<(ContractClass, Program, VersionId)> {
97    let raw_contract_class = std::fs::read_to_string(path).context("Error reading Sierra file.")?;
98
99    let contract_class: ContractClass = serde_json::from_str(&raw_contract_class)
100        .context("Error deserializing Sierra file into contract class.")?;
101    let raw_sierra_program: Vec<_> = contract_class
102        .sierra_program
103        .iter()
104        .map(|big_uint_as_hex| big_uint_as_hex.value.clone())
105        .collect();
106
107    let sierra_version = get_sierra_version_from_program(&raw_sierra_program)?;
108    Ok((
109        contract_class.clone(),
110        contract_class
111            .extract_sierra_program()
112            .context("Error extracting Sierra program from contract class.")?,
113        sierra_version,
114    ))
115}
