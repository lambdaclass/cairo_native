7/// A set of compilation statistics gathered during the compilation.
8/// It should be completely filled at the end of the compilation.
9#[derive(Default, Serialize)]
10pub struct Statistics {
11    /// Number of types defined in the Sierra code.
12    pub sierra_type_count: Option<usize>,
13    /// Number of libfuncs defined in the Sierra code.
14    pub sierra_libfunc_count: Option<usize>,
15    /// Number of statements contained in the Sierra code.

17    /// Number of user functions defined in the Sierra code.
18    pub sierra_func_count: Option<usize>,
19    /// Stats of the declared types in Sierra.
20    pub sierra_declared_types_stats: BTreeMap<u64, SierraDeclaredTypeStats>,
21    /// Stats about params and return types of each Sierra function.
22    pub sierra_func_stats: BTreeMap<u64, SierraFuncStats>,
23    /// Number of statements for each distinct libfunc.
24    pub sierra_libfunc_frequency: BTreeMap<String, u128>,
25    /// Number of times each circuit gate is used.
26    pub sierra_circuit_gates_count: CircuitGatesStats,
27    /// Number of MLIR operations generated.
28    pub mlir_operation_count: Option<u128>,
29    /// Number of MLIR operations generated for each distinct libfunc.
30    pub mlir_operations_by_libfunc: BTreeMap<String, u128>,
31    /// Number of LLVMIR instructions generated.
32    pub llvmir_instruction_count: Option<u128>,
33    /// Number of LLVMIR virtual registers defined.
34    pub llvmir_virtual_register_count: Option<u128>,
35    /// Number of LLVMIR instructions for each distinct opcode.
36    pub llvmir_opcode_frequency: BTreeMap<String, u128>,
37    /// Total compilation time.
38    pub compilation_total_time_ms: Option<u128>,
39    /// Time spent at Sierra to MLIR.
40    pub compilation_sierra_to_mlir_time_ms: Option<u128>,
41    /// Time spent at MLIR passes.
42    pub compilation_mlir_passes_time_ms: Option<u128>,
43    /// Time spent at MLIR to LLVMIR translation.
44    pub compilation_mlir_to_llvm_time_ms: Option<u128>,
45    /// Time spent at LLVM passes.
46    pub compilation_llvm_passes_time_ms: Option<u128>,
47    /// Time spent at LLVM to object compilation.
48    pub compilation_llvm_to_object_time_ms: Option<u128>,
49    /// Time spent at linking the shared library.
50    pub compilation_linking_time_ms: Option<u128>,
51    /// Size of the compiled object.
52    pub object_size_bytes: Option<usize>,
53}

104    /// Counts the gates in a circuit. It uses the same algorithm used
105    /// to evaluate the gates on a circuit when evaluating it.
106    pub fn add_circuit_gates(&mut self, info: &CircuitInfo) -> Result<()> {
107        let mut known_gates = vec![false; 1 + info.n_inputs + info.values.len()];
108        known_gates[0] = true;
109        for i in 0..info.n_inputs {
110            known_gates[i + 1] = true;
111        }
112
113        let mut add_offsets = info.add_offsets.iter().peekable();
114        let mut mul_offsets = info.mul_offsets.iter();
115
116        loop {
117            while let Some(&add_gate_offset) = add_offsets.peek() {
118                let lhs = known_gates[add_gate_offset.lhs].to_owned();
119                let rhs = known_gates[add_gate_offset.rhs].to_owned();
120                let output = known_gates[add_gate_offset.output].to_owned();
121
122                match (lhs, rhs, output) {
123                    (true, true, false) => {
124                        // ADD
125                        self.sierra_circuit_gates_count.add_gate_count += 1;
126                        known_gates[add_gate_offset.output] = true;
127                    }
128                    (false, true, true) => {
129                        // SUB
130                        self.sierra_circuit_gates_count.sub_gate_count += 1;
131                        known_gates[add_gate_offset.lhs] = true;
132                    }
133                    _ => break,
134                }
135                add_offsets.next();
136            }
137
138            if let Some(&GateOffsets { lhs, rhs, output }) = mul_offsets.next() {
139                let lhs_value = known_gates[lhs];
140                let rhs_value = known_gates[rhs];
141                let output_value = known_gates[output];
142
143                match (lhs_value, rhs_value, output_value) {
144                    (true, true, false) => {
145                        // MUL
146                        self.sierra_circuit_gates_count.mul_gate_count += 1;
147                        known_gates[output] = true;
148                    }
149                    (false, true, true) => {
150                        self.sierra_circuit_gates_count.inverse_gate_count += 1;
151                        known_gates[lhs] = true;
152                    }
153                    _ => native_panic!("Imposible circuit"), // It should never reach this point, since it would have failed in the compilation before
154                }
155            } else {
156                break;
157            }
158        }
159        Ok(())
160    }
161}

163/// Clones a variable of type `Option<&mut T>` without consuming self
164///
165/// # Example
166///
167/// The following example would fail to compile otherwise.
168///
169/// ```
170/// # use cairo_native::clone_option_mut;
171/// fn consume(v: Option<&mut Vec<u8>>) {}
172///
173/// let mut vec = Vec::new();
174/// let option = Some(&mut vec);
175/// consume(clone_option_mut!(option));
176/// consume(option);
177/// ```
178#[macro_export]
179macro_rules! clone_option_mut {
180    ( $var:ident ) => {
181        match $var {
182            None => None,
183            Some(&mut ref mut s) => Some(s),
184        }
185    };
