14type OperationWalkCallback =
15    unsafe extern "C" fn(MlirOperation, *mut ::std::os::raw::c_void) -> MlirWalkResult;

17/// Traverses the given operation tree in preorder.
18///
19/// Calls `f` on each operation encountered. The second argument to `f` should
20/// be interpreted as a pointer to a value of type `T`.
21///
22/// TODO: Can we receive a closure instead?
23/// We may need to save a pointer to the closure
24/// inside of the callback data.
25pub fn walk_mlir_operations<T: Sized>(
26    top_op: OperationRef,
27    f: OperationWalkCallback,
28    initial: T,
29) -> T {
30    let mut data = Box::new(initial);
31    unsafe {
32        mlir_sys::mlirOperationWalk(
33            top_op.to_raw(),
34            Some(f),
35            data.as_mut() as *mut _ as *mut c_void,
36            mlir_sys::MlirWalkOrder_MlirWalkPreOrder,
37        );
38    };
39    *data
40}

42/// Traverses from start block to end block (including) in preorder.
43///
44/// Calls `f` on each operation encountered. The second argument to `f` should
45/// be interpreted as a pointer to a value of type `T`.
46///
47/// TODO: Can we receive a closure instead?
48/// We may need to save a pointer to the closure
49/// inside of the callback data.
50pub fn walk_mlir_block<T: Sized>(
51    start_block: BlockRef,
52    end_block: BlockRef,
53    f: OperationWalkCallback,
54    initial: T,
55) -> T {
56    let mut data = Box::new(initial);
57
58    let mut current_block = start_block;
59    loop {
60        let mut next_operation = current_block.first_operation();
61
62        while let Some(operation) = next_operation {
63            unsafe {
64                mlir_sys::mlirOperationWalk(
65                    operation.to_raw(),
66                    Some(f),
67                    data.as_mut() as *mut _ as *mut c_void,
68                    mlir_sys::MlirWalkOrder_MlirWalkPreOrder,
69                );
70            };
71
72            // we have to convert it to raw, and back to ref to bypass borrow checker.
73            next_operation = unsafe {
74                operation
75                    .next_in_block()
76                    .map(OperationRef::to_raw)
77                    .map(|op| OperationRef::from_raw(op))
78            }
79        }
80
81        if current_block == end_block {
82            break;
83        }
84
85        current_block = current_block
86            .next_in_region()
87            .expect("should always reach `end_block`");
88    }
89
90    *data
91}

93/// Traverses the whole LLVM Module, calling `f` on each instruction.
94///
95/// As this function receives a closure rather than a function, there is no need
96/// to receive initial data, and can instead modify the captured environment.
97pub unsafe fn walk_llvm_instructions(llvm_module: LLVMModuleRef, mut f: impl FnMut(LLVMValueRef)) {
98    let new_value = |function_ptr: *mut LLVMValue| {
99        if function_ptr.is_null() {
100            None
101        } else {
102            Some(function_ptr)
103        }
104    };
105    let new_block = |function_ptr: *mut LLVMBasicBlock| {
106        if function_ptr.is_null() {
107            None
108        } else {
109            Some(function_ptr)
110        }
111    };
112
113    let mut current_function = new_value(LLVMGetFirstFunction(llvm_module));
114    while let Some(function) = current_function {
115        let mut current_block = new_block(LLVMGetFirstBasicBlock(function));
116        while let Some(block) = current_block {
117            let mut current_instruction = new_value(LLVMGetFirstInstruction(block));
118            while let Some(instruction) = current_instruction {
119                f(instruction);
120
121                current_instruction = new_value(LLVMGetNextInstruction(instruction));
122            }
123            current_block = new_block(LLVMGetNextBasicBlock(block));
124        }
125        current_function = new_value(LLVMGetNextFunction(function));
126    }
127}
