11/// Traverses the given operation tree in preorder.
12///
13/// Calls `f` on each operation encountered.
14pub fn walk_mlir_operations(top_op: OperationRef, f: &mut impl FnMut(OperationRef)) {
15    f(top_op);
16
17    for region in top_op.regions() {
18        let mut next_block = region.first_block();
19
20        while let Some(block) = next_block {
21            if let Some(operation) = block.first_operation() {
22                walk_mlir_block_operations(operation, f);
23            }
24
25            next_block = block.next_in_region();
26        }
27    }
28}

30/// Traverses all following operations in the current block
31///
32/// Calls `f` on each operation encountered.
33///
34/// NOTE: The lifetime of each operation is bound to the previous operation,
35/// so the only way I found to comply with the borrow checker was to make the
36/// function recursive. This convinces the compiler that the full operation
37/// chain is in scope. This has been fixed in the latest melior release, but
38/// updating the dependency requires us to update to LLVM 20.
39pub fn walk_mlir_block_operations(operation: OperationRef, f: &mut impl FnMut(OperationRef)) {
40    walk_mlir_operations(operation, f);
41
42    if let Some(next_operation) = operation.next_in_block() {
43        walk_mlir_block_operations(next_operation, f);
44    }
45}

47/// Traverses from start block to end block (including) in preorder.
48///
49/// Calls `f` on each operation encountered.
50pub fn walk_mlir_block(
51    start_block: BlockRef,
52    end_block: BlockRef,
53    f: &mut impl FnMut(OperationRef),
54) {
55    let mut next_block = Some(start_block);
56
57    while let Some(block) = next_block {
58        if let Some(operation) = block.first_operation() {
59            walk_mlir_block_operations(operation, f);
60        }
61
62        if block == end_block {
63            return;
64        }
65
66        next_block = block.next_in_region();
67    }
68}

70/// Traverses the whole LLVM Module, calling `f` on each instruction.
71///
72/// As this function receives a closure rather than a function, there is no need
73/// to receive initial data, and can instead modify the captured environment.
74pub unsafe fn walk_llvm_instructions(llvm_module: LLVMModuleRef, mut f: impl FnMut(LLVMValueRef)) {
75    let new_value = |function_ptr: *mut LLVMValue| {
76        if function_ptr.is_null() {
77            None
78        } else {
79            Some(function_ptr)
80        }
81    };
82    let new_block = |function_ptr: *mut LLVMBasicBlock| {
83        if function_ptr.is_null() {
84            None
85        } else {
86            Some(function_ptr)
87        }
88    };
89
90    let mut current_function = new_value(LLVMGetFirstFunction(llvm_module));
91    while let Some(function) = current_function {
92        let mut current_block = new_block(LLVMGetFirstBasicBlock(function));
93        while let Some(block) = current_block {
94            let mut current_instruction = new_value(LLVMGetFirstInstruction(block));
95            while let Some(instruction) = current_instruction {
96                f(instruction);
97
98                current_instruction = new_value(LLVMGetNextInstruction(instruction));
99            }
100            current_block = new_block(LLVMGetNextBasicBlock(block));
101        }
102        current_function = new_value(LLVMGetNextFunction(function));
103    }
104}
