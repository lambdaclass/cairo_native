2//! The trace dump feature is used to generate the execution trace of a sierra program.
3//!
4//! Take, for example, the following sierra code:
5//!
6//! ```sierra
7//! const_as_immediate<Const<felt252, 10>>() -> ([0]);
8//! const_as_immediate<Const<felt252, 20>>() -> ([1]);
9//! store_temp<felt252>([0]) -> ([0]);
10//! felt252_add([0], [1]) -> ([2]);
11//! store_temp<felt252>([2]) -> ([2]);
12//! return([2]);
13//! ```
14//!
15//! The compiler will call `build_state_snapshot` right before each statement.
16//! Iterating every variable on the current scope and saving its value on a global
17//! static variable.
18//!
19//! At the end of the execution, the full trace dump can be retrieved, which
20//! looks something like this:
21//!
22//! ```json
23//! {
24//!   "states": [
25//!     {
26//!       "statementIdx": 0,
27//!       "preStateDump": {}
28//!     },
29//!     {
30//!       "statementIdx": 1,
31//!       "preStateDump": {
32//!         "0": { "Felt": "0xa" }
33//!       }
34//!     },
35//!     {
36//!       "statementIdx": 2,
37//!       "preStateDump": {
38//!         "0": { "Felt": "0xa" },
39//!         "1": { "Felt": "0x14" }
40//!       }
41//!     },
42//!     ...
43//!   ]
44//! }
45//! ```
46//!
47//! To support this feature even on the context of starknet contracts, then we
48//! must support building a trace dump for multiple programs at the same time, as
49//! starknet contracts can themselves call another contracts. To achieve this, we
50//! need two important elements.
51//!
52//! 1. The global static variable must me able to store multiple trace dumps at the
53//!    same time. We have a global static hashmap from trace id to trace dump content.
54//!    See `TRACE_DUMP`.
55//!
56//! 2. We must store somewhere the ID of the current trace dump, and update it
57//!    acordingly when switching between contract executors
58//!
59//! Both these elements must be properly setup before running the executor. See
60//! `cairo-native-run` for an example on how to do it. You can also check on
61//! this file's integration tests.
62//!
63//! When executing starknet contracts, the trace id must be set right
64//! before each execution, restoring the previous trace id after the execution.

87#[allow(clippy::too_many_arguments)]
88pub fn build_state_snapshot(
89    context: &Context,
90    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
91    module: &Module,
92    block: &BlockRef,
93    location: Location,
94    metadata: &mut MetadataStorage,
95    statement_idx: StatementIdx,
96    state: &OrderedHashMap<VarId, Value>,
97    var_types: &HashMap<VarId, ConcreteTypeId>,
98) {
99    let trace_dump = metadata.get_or_insert_with(TraceDumpMeta::default);
100
101    for (var_id, value) in state.iter() {
102        let value_type_id = var_types.get(var_id).unwrap();
103        let value_type = registry.get_type(value_type_id).unwrap();
104
105        let layout = value_type.layout(registry).unwrap();
106
107        let value_ptr = block
108            .alloca1(context, location, value.r#type(), layout.align())
109            .unwrap();
110        block.store(context, location, value_ptr, *value).unwrap();
111
112        trace_dump
113            .build_state(
114                context,
115                module,
116                block,
117                var_id,
118                value_type_id,
119                value_ptr,
120                location,
121            )
122            .unwrap();
123    }
124
125    trace_dump
126        .build_push(context, module, block, statement_idx, location)
127        .unwrap();
128}

152        let (_, program) = load_cairo! {
153            use core::felt252;
154
155            fn main() -> felt252 {
156                let n = 10;
157                let result = fib(1, 1, n);
158                result
159            }
160
161            fn fib(a: felt252, b: felt252, n: felt252) -> felt252 {
162                match n {
163                    0 => a,
164                    _ => fib(b, a + b, n - 1),
165                }
166            }

171    #[rstest]
172    fn test_program(program: Program) {
173        let entrypoint_function = &program
174            .funcs
175            .iter()
176            .find(|x| {
177                x.id.debug_name
178                    .as_ref()
179                    .map(|x| x.contains("main"))
180                    .unwrap_or_default()
181            })
182            .unwrap()
183            .clone();
184
185        let native_context = NativeContext::new();
186        let module = native_context
187            .compile(&program, false, Some(Default::default()))
188            .expect("failed to compile context");
189        let executor = AotNativeExecutor::from_native_module(module, OptLevel::default()).unwrap();
190
191        if let Some(trace_id) = executor.find_symbol_ptr(TraceBinding::TraceId.symbol()) {
192            let trace_id = trace_id.cast::<u64>();
193            unsafe { *trace_id = 0 };
194        }
195
196        TRACE_DUMP
197            .lock()
198            .unwrap()
199            .insert(0, TraceDump::new(ProgramRegistry::new(&program).unwrap()));
200
201        executor
202            .invoke_dynamic(&entrypoint_function.id, &[], Some(u64::MAX))
203            .unwrap();
204
205        let native_trace = TRACE_DUMP
206            .lock()
207            .unwrap()
208            .values()
209            .next()
210            .unwrap()
211            .trace
212            .clone();
213
214        let mut vm = VirtualMachine::new(Arc::new(program));
215
216        let initial_gas = u64::MAX;
217        let args = [];
218        vm.call_program(entrypoint_function, initial_gas, args.into_iter());
219
220        let syscall_handler = &mut StubSyscallHandler::default();
221        let emu_trace = vm.run_with_trace(syscall_handler);
222
223        assert_eq_sorted!(emu_trace, native_trace);
224    }
