1//! # Safe runner
2//!
3//! The safe runner provides a way to run contracts without any risk of crashing due to invalid
4//! memory accesses, including stack overflows. The same mechanism can also be used to abort the
5//! current Cairo program execution, although it will probably leak some memory.
6//!
7//! It works by setting a signal handler for the `SIGSEGV` signal, which is generated by the
8//! operating system on invalid memory accesses. This signal handler is global, therefore it must be
9//! set by the application, rather than the library, by invoking `setup_safe_runner`.
10//!
11//! Since returning from a `SIGSEGV` signal handler would crash the program, the safe runner will
12//! perform a non-local goto using the `setjmp` and `longjmp` functions. The first one configures
13//! the jump target while the second performs the goto.
14//!
15//! Note: The protection is not 100% guaranteed. Since all memory addresses are within the same
16//!   memory space, some combinations of base addresses plus offsets may end up in a different page
17//!   allocation.
18
19use libc::{
20    c_int, sigaction, sigaltstack, siginfo_t, sigset_t, stack_t, ucontext_t, SA_ONSTACK,
21    SA_SIGINFO, SIGSEGV, SIGSTKSZ,
22};
23use std::{
24    cell::UnsafeCell,
25    mem::MaybeUninit,
26    ptr::{self, null_mut},
27};
28use thiserror::Error;
29
30extern "C" {
31    fn setjmp(env: *mut ()) -> c_int;
32    fn longjmp(env: *mut (), val: c_int) -> !;
33}
34
35thread_local! {
36    static STATE: UnsafeCell<SafeRunnerState> = const { UnsafeCell::new(SafeRunnerState::Inactive) };
37    static STACK: UnsafeCell<SignalStack> = const { UnsafeCell::new(SignalStack(MaybeUninit::uninit())) };
38}
39
40type JmpBuf = MaybeUninit<[u8; 1024]>;
41
42#[repr(align(16))]
43#[allow(dead_code)]
44struct SignalStack(MaybeUninit<[u8; SIGSTKSZ]>);
45
46enum SafeRunnerState {
47    Inactive,
48    Active(Box<JmpBuf>),
49}
50
51#[derive(Debug, Error)]
52pub enum SafeRunnerError {
53    #[error("program execution aborted")]
54    Aborted,
55    #[error("program execution segfaulted")]
56    Segfault,
57}
58
59/// Configure the current **process** for the [`SafeRunner`].
60///
61/// Note: It will override the previous signal handler for SIGSEGV.
62pub fn setup_safe_runner() {
63    unsafe {
64        assert_eq!(
65            sigaction(
66                SIGSEGV,
67                &sigaction {
68                    sa_sigaction: segfault_handler
69                        as *const extern "C" fn(c_int, &siginfo_t, &mut ucontext_t)
70                        as usize,
71                    sa_mask: MaybeUninit::<sigset_t>::zeroed().assume_init(),
72                    sa_flags: SA_ONSTACK | SA_SIGINFO,
73                    #[cfg(target_os = "linux")]
74                    sa_restorer: None,
75                },
76                null_mut(),
77            ),
78            0,
79        );
80        assert_eq!(
81            sigaltstack(
82                &stack_t {
83                    ss_sp: STACK.with(|x| x.get()).cast(),
84                    ss_flags: 0,
85                    ss_size: SIGSTKSZ,
86                },
87                null_mut(),
88            ),
89            0,
90        );
91    }
92}

94/// Manually trigger the segfault handler, thus aborting the current program.
95pub fn abort_safe_runner() -> ! {
96    unsafe {
97        match STATE.with(|x| &mut *x.get()) {
98            SafeRunnerState::Inactive => {
99                panic!("manual abort triggered from outside a safe runner");
100            }
101            SafeRunnerState::Active(jmp_buf) => longjmp(jmp_buf.as_mut_ptr().cast(), 2),
102        }
103    }
104}

106/// Run a closure within a safe runner.
107pub fn run_safely<T>(f: impl FnOnce() -> T) -> Result<T, SafeRunnerError> {
108    let (jmp_buf, prev_state) = STATE.with(|x| unsafe {
109        let jmp_buf;
110        let prev_state = ptr::replace(
111            x.get(),
112            SafeRunnerState::Active({
113                let mut tmp = Box::new(JmpBuf::uninit());
114                jmp_buf = tmp.as_mut_ptr();
115                tmp
116            }),
117        );
118
119        (jmp_buf, prev_state)
120    });
121
122    let jmp_ret = unsafe { setjmp(jmp_buf.cast()) };
123    let result = match jmp_ret {
124        0 => Ok(f()),
125        1 => Err(SafeRunnerError::Segfault),
126        2 => Err(SafeRunnerError::Aborted),
127        _ => unreachable!(),
128    };
129
130    STATE.with(|x| unsafe { ptr::write(x.get(), prev_state) });
131    result
132}
133
134unsafe extern "C" fn segfault_handler(_sig: c_int, _info: &siginfo_t, _context: &mut ucontext_t) {
135    match STATE.with(|x| &mut *x.get()) {
136        SafeRunnerState::Inactive => libc::abort(),
137        SafeRunnerState::Active(jmp_buf) => longjmp(jmp_buf.as_mut_ptr().cast(), 1),
138    }
