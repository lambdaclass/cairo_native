65/// Generation of MLIR operations from their Sierra counterparts.
66///
67/// All possible Sierra libfuncs must implement it. It is already implemented for all the core
68/// libfuncs, contained in [CoreConcreteLibfunc].
69pub trait LibfuncBuilder {
70    /// Error type returned by this trait's methods.
71    type Error: Error;
72
73    /// Generate the MLIR operations.
74    fn build<'ctx, 'this>(
75        &self,
76        context: &'ctx Context,
77        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
78        entry: &'this Block<'ctx>,
79        location: Location<'ctx>,
80        helper: &LibfuncHelper<'ctx, 'this>,
81        metadata: &mut MetadataStorage,
82    ) -> Result<()>;
83
84    /// Return the target function if the statement is a function call.
85    ///
86    /// This is used by the compiler to check whether a statement is a function call and apply the
87    /// tail recursion logic.
88    fn is_function_call(&self) -> Option<&FunctionId>;
89}

260/// Helper struct which contains logic generation for extra MLIR blocks and branch operations to the
261/// next statements.
262///
263/// Each branch index should be present in exactly one call a branching method (either
264/// [`br`](#method.br) or [`cond_br`](#method.cond_br)).
265///
266/// This helper is necessary because the statement following the current one may not have the same
267/// arguments as the results returned by the current statement. Because of that, a direct jump
268/// cannot be made and some processing is required.
269pub struct LibfuncHelper<'ctx, 'this>
270where
271    'this: 'ctx,
272{
273    pub module: &'this Module<'ctx>,
274    pub init_block: &'this BlockRef<'ctx, 'this>,
275
276    pub region: &'this Region<'ctx>,
277    pub blocks_arena: &'this Bump,
278    pub last_block: Cell<&'this BlockRef<'ctx, 'this>>,
279
280    pub branches: Vec<(&'this Block<'ctx>, Vec<BranchArg<'ctx, 'this>>)>,
281    pub results: Vec<Vec<Cell<Option<Value<'ctx, 'this>>>>>,
282
283    #[cfg(feature = "with-libfunc-profiling")]
284    // Since function calls don't get profiled, this field is optional
285    pub profiler: Option<(
286        crate::metadata::profiler::ProfilerMeta,
287        cairo_lang_sierra::program::StatementIdx,
288        (Value<'ctx, 'this>, Value<'ctx, 'this>),
289    )>,
290}
291
292impl<'ctx, 'this> LibfuncHelper<'ctx, 'this>
293where
294    'this: 'ctx,
295{
296    #[doc(hidden)]
297    pub(crate) fn results(self) -> Result<Vec<Vec<Value<'ctx, 'this>>>> {
298        self.results
299            .into_iter()
300            .enumerate()
301            .map(|(branch_idx, x)| {
302                x.into_iter()
303                    .enumerate()
304                    .map(|(arg_idx, x)| {
305                        x.into_inner().to_native_assert_error(&format!(
306                            "Argument #{arg_idx} of branch {branch_idx} doesn't have a value."
307                        ))
308                    })
309                    .collect()
310            })
311            .collect()
312    }
313
314    /// Return the initialization block.
315    ///
316    /// The init block is used for `llvm.alloca` instructions. It is guaranteed to not be executed
317    /// multiple times on tail-recursive functions. This property allows generating tail-recursive
318    /// functions that do not grow the stack.
319    pub fn init_block(&self) -> &'this Block<'ctx> {
320        self.init_block
321    }
322
323    /// Inserts a new block after all the current libfunc's blocks.
324    pub fn append_block(&self, block: Block<'ctx>) -> &'this Block<'ctx> {
325        let block = self
326            .region
327            .insert_block_after(*self.last_block.get(), block);
328
329        let block_ref: &'this mut BlockRef<'ctx, 'this> = self.blocks_arena.alloc(block);
330        self.last_block.set(block_ref);
331
332        block_ref
333    }
334
335    /// Creates an unconditional branching operation out of the libfunc and into the next statement.
336    ///
337    /// This method will also store the returned values so that they can be moved into the state and
338    /// used later on when required.
339    fn br(
340        &self,
341        block: &'this Block<'ctx>,
342        branch: usize,
343        results: &[Value<'ctx, 'this>],
344        location: Location<'ctx>,
345    ) -> Result<()> {
346        let (successor, operands) = &self.branches[branch];
347
348        for (dst, src) in self.results[branch].iter().zip(results) {
349            dst.replace(Some(*src));
350        }
351
352        let destination_operands = operands
353            .iter()
354            .copied()
355            .map(|op| match op {
356                BranchArg::External(x) => x,
357                BranchArg::Returned(i) => results[i],
358            })
359            .collect::<Vec<_>>();
360
361        #[cfg(feature = "with-libfunc-profiling")]
362        self.push_profiler_frame(
363            unsafe { self.context().to_ref() },
364            self.module,
365            block,
366            location,
367        )?;
368
369        block.append_operation(cf::br(successor, &destination_operands, location));
370        Ok(())
371    }
372
373    /// Creates a conditional binary branching operation, potentially jumping out of the libfunc and
374    /// into the next statement.
375    ///
376    /// While generating a `cond_br` that doesn't jump out of the libfunc is possible, it should be
377    /// avoided whenever possible. In those cases just use [melior::dialect::cf::cond_br].
378    ///
379    /// This method will also store the returned values so that they can be moved into the state and
380    /// used later on when required.
381    // TODO: Allow one block to be libfunc-internal.
382    fn cond_br(
383        &self,
384        context: &'ctx Context,
385        block: &'this Block<'ctx>,
386        condition: Value<'ctx, 'this>,
387        branches: [usize; 2],
388        results: [&[Value<'ctx, 'this>]; 2],
389        location: Location<'ctx>,
390    ) -> Result<()> {
391        let (block_true, args_true) = {
392            let (successor, operands) = &self.branches[branches[0]];
393
394            for (dst, src) in self.results[branches[0]].iter().zip(results[0]) {
395                dst.replace(Some(*src));
396            }
397
398            let destination_operands = operands
399                .iter()
400                .copied()
401                .map(|op| match op {
402                    BranchArg::External(x) => x,
403                    BranchArg::Returned(i) => results[0][i],
404                })
405                .collect::<Vec<_>>();
406
407            (*successor, destination_operands)
408        };
409
410        let (block_false, args_false) = {
411            let (successor, operands) = &self.branches[branches[1]];
412
413            for (dst, src) in self.results[branches[1]].iter().zip(results[1]) {
414                dst.replace(Some(*src));
415            }
416
417            let destination_operands = operands
418                .iter()
419                .copied()
420                .map(|op| match op {
421                    BranchArg::External(x) => x,
422                    BranchArg::Returned(i) => results[1][i],
423                })
424                .collect::<Vec<_>>();
425
426            (*successor, destination_operands)
427        };
428
429        #[cfg(feature = "with-libfunc-profiling")]
430        self.push_profiler_frame(context, self.module, block, location)?;
431
432        block.append_operation(cf::cond_br(
433            context,
434            condition,
435            block_true,
436            block_false,
437            &args_true,
438            &args_false,
439            location,
440        ));
441        Ok(())
442    }
443
444    #[cfg(feature = "with-libfunc-profiling")]
445    fn push_profiler_frame(
446        &self,
447        context: &'ctx Context,
448        module: &'this Module,
449        block: &'this Block<'ctx>,
450        location: Location<'ctx>,
451    ) -> Result<()> {
452        if let Some((profiler_meta, statement_idx, t0)) = self.profiler.as_ref() {
453            let t0 = *t0;
454            let t1 = profiler_meta.measure_timestamp(context, block, location)?;
455
456            profiler_meta.push_frame(context, module, block, statement_idx.0, t0, t1, location)?;
457        }
458
459        Ok(())
460    }
461}

477fn increment_builtin_counter<'ctx: 'a, 'a>(
478    context: &'ctx Context,
479    block: &'ctx Block<'ctx>,
480    location: Location<'ctx>,
481    value: Value<'ctx, '_>,
482) -> crate::error::Result<Value<'ctx, 'a>> {
483    increment_builtin_counter_by(context, block, location, value, 1)
484}
485
486fn increment_builtin_counter_by<'ctx: 'a, 'a>(
487    context: &'ctx Context,
488    block: &'ctx Block<'ctx>,
489    location: Location<'ctx>,
490    value: Value<'ctx, '_>,
491    amount: impl Into<BigInt>,
492) -> crate::error::Result<Value<'ctx, 'a>> {
493    block.append_op_result(arith::addi(
494        value,
495        block.const_int(context, location, amount, 64)?,
496        location,
497    ))
498}
499
500fn increment_builtin_counter_by_if<'ctx: 'a, 'a>(
501    context: &'ctx Context,
502    block: &'ctx Block<'ctx>,
503    location: Location<'ctx>,
504    value_to_inc: Value<'ctx, '_>,
505    true_amount: impl Into<BigInt>,
506    false_amount: impl Into<BigInt>,
507    condition: Value<'ctx, '_>,
508) -> crate::error::Result<Value<'ctx, 'a>> {
509    let true_amount_value = block.const_int(context, location, true_amount, 64)?;
510    let false_amount_value = block.const_int(context, location, false_amount, 64)?;
511
512    let true_incremented =
513        block.append_op_result(arith::addi(value_to_inc, true_amount_value, location))?;
514    let false_incremented =
515        block.append_op_result(arith::addi(value_to_inc, false_amount_value, location))?;
516
517    block.append_op_result(arith::select(
518        condition,
519        true_incremented,
520        false_incremented,
521        location,
522    ))
523}

525fn build_noop<'ctx, 'this, const N: usize, const PROCESS_BUILTINS: bool>(
526    context: &'ctx Context,
527    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
528    entry: &'this Block<'ctx>,
529    location: Location<'ctx>,
530    helper: &LibfuncHelper<'ctx, 'this>,
531    _metadata: &mut MetadataStorage,
532    param_signatures: &[ParamSignature],
533) -> Result<()> {
534    let mut params = Vec::with_capacity(N);
535
536    #[allow(clippy::needless_range_loop)]
537    for i in 0..N {
538        let param_ty = registry.get_type(&param_signatures[i].ty)?;
539        let mut param_val = entry.argument(i)?.into();
540
541        if PROCESS_BUILTINS
542            && param_ty.is_builtin()
543            && !matches!(
544                param_ty,
545                CoreTypeConcrete::BuiltinCosts(_)
546                    | CoreTypeConcrete::Coupon(_)
547                    | CoreTypeConcrete::GasBuiltin(_)
548                    | CoreTypeConcrete::Starknet(StarknetTypeConcrete::System(_))
549            )
550        {
551            param_val = increment_builtin_counter(context, entry, location, param_val)?;
552        }
553
554        params.push(param_val);
555    }
556
557    helper.br(entry, 0, &params, location)
558}
