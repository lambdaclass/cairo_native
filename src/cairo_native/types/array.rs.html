1//! # Array type
2//!
3//! An array type is a dynamically allocated list of items.
4//!
5//! ## Layout
6//!
7//! Being dynamically allocated, we just need to keep the pointer to the data, its length and
8//! its capacity:
9//!
10//! | Index | Type           | Description              |
11//! | ----- | -------------- | ------------------------ |
12//! |   0   | `!llvm.ptr<T>` | Pointer to the data[^1]. |
13//! |   1   | `i32`          | Array start offset[^2].  |
14//! |   1   | `i32`          | Array end offset[^2].    |
15//! |   2   | `i32`          | Allocated capacity[^2].  |
16//!
17//! The pointer to the allocation (which is **not the data**) contains:
18//!   1. Reference counter.
19//!   2. Padding.
20//!   3. Array data. Its address is the pointer to the data stored in the type.
21//!
22//! [^1]: When capacity is zero, this field is not guaranteed to be valid.
23//! [^2]: Those numbers are number of items, **not bytes**.
24

56/// Build the MLIR type.
57///
58/// Check out [the module](self) for more info.
59pub fn build<'ctx>(
60    context: &'ctx Context,
61    module: &Module<'ctx>,
62    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
63    metadata: &mut MetadataStorage,
64    info: WithSelf<InfoAndTypeConcreteType>,
65) -> Result<Type<'ctx>> {
66    DupOverridesMeta::register_with(
67        context,
68        module,
69        registry,
70        metadata,
71        info.self_ty(),
72        |metadata| {
73            // There's no need to build the type here because it'll always be built within
74            // `build_dup`.
75
76            Ok(Some(build_dup(context, module, registry, metadata, &info)?))
77        },
78    )?;
79    DropOverridesMeta::register_with(
80        context,
81        module,
82        registry,
83        metadata,
84        info.self_ty(),
85        |metadata| {
86            // There's no need to build the type here because it'll always be built within
87            // `build_drop`.
88
89            Ok(Some(build_drop(
90                context, module, registry, metadata, &info,
91            )?))
92        },
93    )?;
94
95    let ptr_ty = llvm::r#type::pointer(context, 0);
96    let len_ty = IntegerType::new(context, 32).into();
97
98    Ok(llvm::r#type::r#struct(
99        context,
100        &[ptr_ty, len_ty, len_ty, len_ty],
101        false,
102    ))
103}

105/// This function clones the array shallowly. That is, it'll increment the reference counter but not
106/// actually clone anything. The deep clone implementation is provided in `src/libfuncs/array.rs` as
107/// part of some libfuncs's implementations.
108fn build_dup<'ctx>(
109    context: &'ctx Context,
110    module: &Module<'ctx>,
111    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
112    metadata: &mut MetadataStorage,
113    info: &WithSelf<InfoAndTypeConcreteType>,
114) -> Result<Region<'ctx>> {
115    let location = Location::unknown(context);
116    let value_ty = registry.build_type(context, module, metadata, info.self_ty())?;
117
118    let elem_layout = registry.get_type(&info.ty)?.layout(registry)?;
119    let refcount_offset = calc_data_prefix_offset(elem_layout);
120
121    let region = Region::new();
122    let entry = region.append_block(Block::new(&[(value_ty, location)]));
123
124    let array_cap = entry.extract_value(
125        context,
126        location,
127        entry.argument(0)?.into(),
128        IntegerType::new(context, 32).into(),
129        3,
130    )?;
131    let k0 = entry.const_int(context, location, 0, 32)?;
132    let is_empty = entry.append_op_result(arith::cmpi(
133        context,
134        CmpiPredicate::Eq,
135        array_cap,
136        k0,
137        location,
138    ))?;
139
140    entry.append_operation(scf::r#if(
141        is_empty,
142        &[],
143        {
144            let region = Region::new();
145            let block = region.append_block(Block::new(&[]));
146
147            block.append_operation(scf::r#yield(&[], location));
148            region
149        },
150        {
151            let region = Region::new();
152            let block = region.append_block(Block::new(&[]));
153
154            let array_ptr_ptr = block.extract_value(
155                context,
156                location,
157                entry.argument(0)?.into(),
158                llvm::r#type::pointer(context, 0),
159                0,
160            )?;
161            let array_ptr = block.load(
162                context,
163                location,
164                array_ptr_ptr,
165                llvm::r#type::pointer(context, 0),
166            )?;
167
168            let refcount_ptr = block.gep(
169                context,
170                location,
171                array_ptr,
172                &[GepIndex::Const(-(refcount_offset as i32))],
173                IntegerType::new(context, 8).into(),
174            )?;
175            let ref_count = block.load(
176                context,
177                location,
178                refcount_ptr,
179                IntegerType::new(context, 32).into(),
180            )?;
181
182            let k1 = block.const_int(context, location, 1, 32)?;
183            let ref_count = block.append_op_result(arith::addi(ref_count, k1, location))?;
184            block.store(context, location, refcount_ptr, ref_count)?;
185
186            block.append_operation(scf::r#yield(&[], location));
187            region
188        },
189        location,
190    ));
191
192    entry.append_operation(func::r#return(
193        &[entry.argument(0)?.into(), entry.argument(0)?.into()],
194        location,
195    ));
196    Ok(region)
197}

199/// This function decreases the reference counter of the array by one.
200/// If the reference counter reaches zero, then all the resources are freed.
201fn build_drop<'ctx>(
202    context: &'ctx Context,
203    module: &Module<'ctx>,
204    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
205    metadata: &mut MetadataStorage,
206    info: &WithSelf<InfoAndTypeConcreteType>,
207) -> Result<Region<'ctx>> {
208    let location = Location::unknown(context);
209    if metadata.get::<ReallocBindingsMeta>().is_none() {
210        metadata.insert(ReallocBindingsMeta::new(context, module));
211    }
212
213    let value_ty = registry.build_type(context, module, metadata, info.self_ty())?;
214
215    let elem_ty = registry.get_type(&info.ty)?;
216    let elem_stride = elem_ty.layout(registry)?.pad_to_align().size();
217    let elem_ty = elem_ty.build(context, module, registry, metadata, &info.ty)?;
218    let elem_layout = registry.get_type(&info.ty)?.layout(registry)?;
219    let refcount_offset = calc_data_prefix_offset(elem_layout);
220
221    let region = Region::new();
222    let entry = region.append_block(Block::new(&[(value_ty, location)]));
223
224    let array_ptr_ptr = entry.extract_value(
225        context,
226        location,
227        entry.argument(0)?.into(),
228        llvm::r#type::pointer(context, 0),
229        0,
230    )?;
231
232    let array_cap = entry.extract_value(
233        context,
234        location,
235        entry.argument(0)?.into(),
236        IntegerType::new(context, 32).into(),
237        3,
238    )?;
239    let k0 = entry.const_int(context, location, 0, 32)?;
240    let zero_capacity = entry.append_op_result(arith::cmpi(
241        context,
242        CmpiPredicate::Eq,
243        array_cap,
244        k0,
245        location,
246    ))?;
247
248    entry.append_operation(scf::r#if(
249        zero_capacity,
250        &[],
251        {
252            // if the array has no capacity, do nothing, as there is no allocation
253
254            let region = Region::new();
255            let block = region.append_block(Block::new(&[]));
256
257            block.append_operation(scf::r#yield(&[], location));
258            region
259        },
260        {
261            // if the array has capacity, decrease the reference counter
262            // and, in case it reaches zero, free all the resources.
263
264            let region = Region::new();
265            let block = region.append_block(Block::new(&[]));
266
267            // obtain the reference counter
268            let array_ptr = block.load(
269                context,
270                location,
271                array_ptr_ptr,
272                llvm::r#type::pointer(context, 0),
273            )?;
274            let refcount_ptr = block.gep(
275                context,
276                location,
277                array_ptr,
278                &[GepIndex::Const(-(refcount_offset as i32))],
279                IntegerType::new(context, 8).into(),
280            )?;
281            let ref_count = block.load(
282                context,
283                location,
284                refcount_ptr,
285                IntegerType::new(context, 32).into(),
286            )?;
287
288            // if the reference counter is greater than 1, then it's shared
289            let k1 = block.const_int(context, location, 1, 32)?;
290            let is_shared = block.append_op_result(arith::cmpi(
291                context,
292                CmpiPredicate::Ne,
293                ref_count,
294                k1,
295                location,
296            ))?;
297
298            block.append_operation(scf::r#if(
299                is_shared,
300                &[],
301                {
302                    // if the array is shared, decrease the reference counter by one
303                    let region = Region::new();
304                    let block = region.append_block(Block::new(&[]));
305
306                    let ref_count = block.append_op_result(arith::subi(ref_count, k1, location))?;
307                    block.store(context, location, refcount_ptr, ref_count)?;
308
309                    block.append_operation(scf::r#yield(&[], location));
310                    region
311                },
312                {
313                    // if the array is not shared, drop all elements and free the memory
314                    let region = Region::new();
315                    let block = region.append_block(Block::new(&[]));
316
317                    match metadata.get::<DropOverridesMeta>() {
318                        Some(drop_overrides_meta) if drop_overrides_meta.is_overriden(&info.ty) => {
319                            let k0 = block.const_int(context, location, 0, 64)?;
320                            let elem_stride =
321                                block.const_int(context, location, elem_stride, 64)?;
322
323                            let max_len_ptr = block.gep(
324                                context,
325                                location,
326                                array_ptr,
327                                &[GepIndex::Const(
328                                    -((refcount_offset - size_of::<u32>()) as i32),
329                                )],
330                                IntegerType::new(context, 8).into(),
331                            )?;
332                            let max_len = block.load(
333                                context,
334                                location,
335                                max_len_ptr,
336                                IntegerType::new(context, 32).into(),
337                            )?;
338                            let max_len = block.extui(
339                                max_len,
340                                IntegerType::new(context, 64).into(),
341                                location,
342                            )?;
343                            let offset_end = block.muli(max_len, elem_stride, location)?;
344
345                            // Drop each element in the array.
346                            block.append_operation(scf::r#for(
347                                k0,
348                                offset_end,
349                                elem_stride,
350                                {
351                                    let region = Region::new();
352                                    let block = region.append_block(Block::new(&[(
353                                        IntegerType::new(context, 64).into(),
354                                        location,
355                                    )]));
356
357                                    let elem_offset = block.argument(0)?.into();
358                                    let elem_ptr = block.gep(
359                                        context,
360                                        location,
361                                        array_ptr,
362                                        &[GepIndex::Value(elem_offset)],
363                                        IntegerType::new(context, 8).into(),
364                                    )?;
365                                    let elem_val =
366                                        block.load(context, location, elem_ptr, elem_ty)?;
367
368                                    drop_overrides_meta.invoke_override(
369                                        context, &block, location, &info.ty, elem_val,
370                                    )?;
371
372                                    block.append_operation(scf::r#yield(&[], location));
373                                    region
374                                },
375                                location,
376                            ));
377                        }
378                        _ => {}
379                    }
380
381                    // finally, free the array allocation
382                    block.append_operation(ReallocBindingsMeta::free(
383                        context,
384                        refcount_ptr,
385                        location,
386                    )?);
387                    block.append_operation(ReallocBindingsMeta::free(
388                        context,
389                        array_ptr_ptr,
390                        location,
391                    )?);
392
393                    block.append_operation(scf::r#yield(&[], location));
394                    region
395                },
396                location,
397            ));
398
399            block.append_operation(scf::r#yield(&[], location));
400            region
401        },
402        location,
403    ));
404
405    entry.append_operation(func::r#return(&[], location));
406    Ok(region)
407}

409pub fn calc_data_prefix_offset(layout: Layout) -> usize {
410    get_integer_layout(32)
411        .extend(get_integer_layout(32))
412        .expect("creating a layout of two i32 should never fail")
413        .0
414        .align_to(layout.align())
415        .expect("layout size rounded up to the next multiple of layout alignment should never be greater than ISIZE::MAX")
416        .pad_to_align()
417        .size()
418}
