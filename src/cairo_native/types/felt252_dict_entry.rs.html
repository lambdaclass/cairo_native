1//! # `Felt` dictionary entry type
2//!
3//! The entry type returning when getting a value from a dictionary.
4//!
5//! It is represented as the following struct:
6//!
7//! | Index | Type           | Description                                  |
8//! | ----- | -------------- | -------------------------------------------- |
9//! |   0   | `!llvm.ptr`    | Pointer to the dictionary (Rust).            |
10//! |   1   | `!llvm.ptr`    | Pointer to the entry's value pointer (Rust). |
11//!
12
13use super::WithSelf;
14use crate::{error::Result, metadata::MetadataStorage};
15use cairo_lang_sierra::{
16    extensions::{
17        core::{CoreLibfunc, CoreType},
18        types::InfoAndTypeConcreteType,
19    },
20    program_registry::ProgramRegistry,
21};
22use melior::{
23    dialect::llvm,
24    ir::{Module, Type},
25    Context,
26};
27
28/// Build the MLIR type.
29///
30/// Check out [the module](self) for more info.
31pub fn build<'ctx>(
32    context: &'ctx Context,
33    _module: &Module<'ctx>,
34    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
35    _metadata: &mut MetadataStorage,
36    _info: WithSelf<InfoAndTypeConcreteType>,
37) -> Result<Type<'ctx>> {
38    // Note: This is neither droppable nor cloneable.
39    Ok(llvm::r#type::r#struct(
40        context,
41        &[
42            llvm::r#type::pointer(context, 0), // dict ptr
43            llvm::r#type::pointer(context, 0), // value ptr
44        ],
45        false,
46    ))
47}
