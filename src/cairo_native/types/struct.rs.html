1//! # Struct type
2//!
3//! A struct is just a fixed collection of values that may have different types, which are known at
4//! compile-time. Its fields are properly aligned and respect the declaration's field ordering.
5//!
6//! For example, the following struct would have a layout as described in the table below:
7//!
8//! ```cairo
9//! struct MyStruct {
10//!     U8: u8,
11//!     U16: u16,
12//!     U32: u32,
13//!     U64: u64,
14//!     Felt: Felt,
15//! }
16//! ```
17//!
18//! | Index | Type   | ABI (in Rust types) | Alignment | Size |
19//! | ----- | ------ | ------------------- | --------- | ---- |
20//! |   0   | `i8`   | `u8`                |         1 |    1 |
21//! |  N/A  | N/A    | `[u8; 1]`           |         1 |    1 |
22//! |   1   | `i16`  | `u16`               |         2 |    2 |
23//! |  N/A  | N/A    | `[u8; 2]`           |         1 |    2 |
24//! |   2   | `i32`  | `u32`               |         4 |    4 |
25//! |  N/A  | N/A    | `[u8; 4]`           |         1 |    4 |
26//! |   3   | `i64`  | `u64`               |         8 |    8 |
27//! |   4   | `i252` | `[u64; 4]`          |         8 |    8 |
28//!
29//! As inferred in the table above, the struct will have 8-byte alignment and a size of 30 bytes.
30//! Since this way of generating structs is equivalent to the one used in C and C++, the same
31//! effects apply. For example, if we invert the order of the fields the ABI will change but we
32//! won't waste a single byte in padding; unless we're creating an array, in which case we'd waste
33//! only a single byte per element.

57/// Build the MLIR type.
58///
59/// Check out [the module](self) for more info.
60pub fn build<'ctx>(
61    context: &'ctx Context,
62    module: &Module<'ctx>,
63    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
64    metadata: &mut MetadataStorage,
65    info: WithSelf<StructConcreteType>,
66) -> Result<Type<'ctx>> {
67    DupOverridesMeta::register_with(
68        context,
69        module,
70        registry,
71        metadata,
72        info.self_ty(),
73        |metadata| {
74            // The following unwrap is unreachable because `register_with` will always insert it
75            // before calling this closure.
76            let mut needs_override = false;
77            for member in &info.members {
78                registry.build_type(context, module, metadata, member)?;
79                if metadata
80                    .get::<DupOverridesMeta>()
81                    .ok_or(Error::MissingMetadata)?
82                    .is_overriden(member)
83                {
84                    needs_override = true;
85                    break;
86                }
87            }
88
89            needs_override
90                .then(|| build_dup(context, module, registry, metadata, &info))
91                .transpose()
92        },
93    )?;
94    DropOverridesMeta::register_with(
95        context,
96        module,
97        registry,
98        metadata,
99        info.self_ty(),
100        |metadata| {
101            // The following unwrap is unreachable because `register_with` will always insert it
102            // before calling this closure.
103            let mut needs_override = false;
104            for member in &info.members {
105                registry.build_type(context, module, metadata, member)?;
106                if metadata
107                    .get::<DropOverridesMeta>()
108                    .ok_or(Error::MissingMetadata)?
109                    .is_overriden(member)
110                {
111                    needs_override = true;
112                    break;
113                }
114            }
115
116            needs_override
117                .then(|| build_drop(context, module, registry, metadata, &info))
118                .transpose()
119        },
120    )?;
121
122    let members = info
123        .members
124        .iter()
125        .map(|member| registry.build_type(context, module, metadata, member))
126        .collect::<Result<Vec<_>>>()?;
127    Ok(llvm::r#type::r#struct(context, &members, false))
128}

130fn build_dup<'ctx>(
131    context: &'ctx Context,
132    module: &Module<'ctx>,
133    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
134    metadata: &mut MetadataStorage,
135    info: &WithSelf<StructConcreteType>,
136) -> Result<Region<'ctx>> {
137    let location = Location::unknown(context);
138
139    let self_ty = registry.build_type(context, module, metadata, info.self_ty())?;
140
141    let region = Region::new();
142    let entry = region.append_block(Block::new(&[(self_ty, location)]));
143
144    let mut src_value = entry.arg(0)?;
145    let mut dst_value = entry.append_op_result(llvm::undef(self_ty, location))?;
146
147    for (idx, member_id) in info.members.iter().enumerate() {
148        let member_ty = registry.build_type(context, module, metadata, member_id)?;
149        let member_val = entry.extract_value(context, location, src_value, member_ty, idx)?;
150
151        // The following unwrap is unreachable because the registration logic will always insert it.
152        let values = metadata
153            .get::<DupOverridesMeta>()
154            .ok_or(Error::MissingMetadata)?
155            .invoke_override(context, &entry, location, member_id, member_val)?;
156
157        src_value = entry.insert_value(context, location, src_value, values.0, idx)?;
158        dst_value = entry.insert_value(context, location, dst_value, values.1, idx)?;
159    }
160
161    entry.append_operation(func::r#return(&[src_value, dst_value], location));
162    Ok(region)
163}

165fn build_drop<'ctx>(
166    context: &'ctx Context,
167    module: &Module<'ctx>,
168    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
169    metadata: &mut MetadataStorage,
170    info: &WithSelf<StructConcreteType>,
171) -> Result<Region<'ctx>> {
172    let location = Location::unknown(context);
173
174    let self_ty = registry.build_type(context, module, metadata, info.self_ty())?;
175
176    let region = Region::new();
177    let entry = region.append_block(Block::new(&[(self_ty, location)]));
178
179    let value = entry.arg(0)?;
180    for (idx, member_id) in info.members.iter().enumerate() {
181        let member_ty = registry.build_type(context, module, metadata, member_id)?;
182        let member_val = entry.extract_value(context, location, value, member_ty, idx)?;
183
184        // The following unwrap is unreachable because the registration logic will always insert it.
185        metadata
186            .get::<DropOverridesMeta>()
187            .ok_or(Error::MissingMetadata)?
188            .invoke_override(context, &entry, location, member_id, member_val)?;
189    }
190
191    entry.append_operation(func::r#return(&[], location));
192    Ok(region)
193}
