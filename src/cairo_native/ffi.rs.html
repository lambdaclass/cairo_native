43/// For any `!llvm.struct<...>` type, return the MLIR type of the field at the requested index.
44pub fn get_struct_field_type_at<'c>(r#type: &Type<'c>, index: usize) -> Type<'c> {
45    assert!(r#type.is_llvm_struct_type());
46    unsafe {
47        Type::from_raw(mlirLLVMStructTypeGetElementType(
48            r#type.to_raw(),
49            index as isize,
50        ))
51    }
52}

54/// Optimization levels.
55#[derive(Clone, Copy, Debug, Default, Eq, Hash, Ord, PartialEq, PartialOrd)]
56pub enum OptLevel {
57    None,
58    Less,
59    #[default]
60    Default,
61    Aggressive,
62}
63
64impl From<usize> for OptLevel {
65    fn from(value: usize) -> Self {
66        match value {
67            0 => OptLevel::None,
68            1 => OptLevel::Less,
69            2 => OptLevel::Default,
70            _ => OptLevel::Aggressive,
71        }
72    }
73}
74
75impl From<OptLevel> for usize {
76    fn from(val: OptLevel) -> Self {
77        match val {
78            OptLevel::None => 0,
79            OptLevel::Less => 1,
80            OptLevel::Default => 2,
81            OptLevel::Aggressive => 3,
82        }
83    }
84}
85
86impl From<u8> for OptLevel {
87    fn from(value: u8) -> Self {
88        match value {
89            0 => OptLevel::None,
90            1 => OptLevel::Less,
91            2 => OptLevel::Default,
92            _ => OptLevel::Aggressive,
93        }
94    }
95}

97/// Converts a MLIR module to a compile object, that can be linked with a linker.
98pub fn module_to_object(module: &Module<'_>, opt_level: OptLevel) -> Result<Vec<u8>> {
99    static INITIALIZED: OnceLock<()> = OnceLock::new();
100
101    INITIALIZED.get_or_init(|| unsafe {
102        LLVM_InitializeAllTargets();
103        LLVM_InitializeAllTargetInfos();
104        LLVM_InitializeAllTargetMCs();
105        LLVM_InitializeAllAsmPrinters();
106        LLVM_InitializeAllAsmParsers();
107    });
108
109    unsafe {
110        let llvm_context = LLVMContextCreate();
111
112        let op = module.as_operation().to_raw();
113
114        trace!("starting mlir to llvm compilation");
115        let pre_mlir_instant = Instant::now();
116        let llvm_module = mlirTranslateModuleToLLVMIR(op, llvm_context as *mut _) as *mut _;
117        let mlir_time = pre_mlir_instant.elapsed().as_millis();
118        trace!(time = mlir_time, "mlir to llvm finished");
119
120        let mut null = null_mut();
121        let mut error_buffer = addr_of_mut!(null);
122
123        let target_triple = LLVMGetDefaultTargetTriple();
124        let target_cpu = LLVMGetHostCPUName();
125        let target_cpu_features = LLVMGetHostCPUFeatures();
126
127        let mut target: MaybeUninit<LLVMTargetRef> = MaybeUninit::uninit();
128
129        if LLVMGetTargetFromTriple(target_triple, target.as_mut_ptr(), error_buffer) != 0 {
130            let error = CStr::from_ptr(*error_buffer);
131            let err = error.to_string_lossy().to_string();
132            LLVMDisposeMessage(*error_buffer);
133            Err(Error::LLVMCompileError(err))?;
134        } else if !(*error_buffer).is_null() {
135            LLVMDisposeMessage(*error_buffer);
136            error_buffer = addr_of_mut!(null);
137        }
138
139        let target = target.assume_init();
140
141        let machine = LLVMCreateTargetMachine(
142            target,
143            target_triple.cast(),
144            target_cpu.cast(),
145            target_cpu_features.cast(),
146            match opt_level {
147                OptLevel::None => LLVMCodeGenOptLevel::LLVMCodeGenLevelNone,
148                OptLevel::Less => LLVMCodeGenOptLevel::LLVMCodeGenLevelLess,
149                OptLevel::Default => LLVMCodeGenOptLevel::LLVMCodeGenLevelDefault,
150                OptLevel::Aggressive => LLVMCodeGenOptLevel::LLVMCodeGenLevelAggressive,
151            },
152            LLVMRelocMode::LLVMRelocPIC,
153            LLVMCodeModel::LLVMCodeModelDefault,
154        );
155
156        let opts = LLVMCreatePassBuilderOptions();
157
158        let opt = match opt_level {
159            OptLevel::None => 0,
160            OptLevel::Less => 1,
161            // slp-vectorizer pass did cause some issues, but after the change
162            // on function attributes it seems to not trigger them anymore.
163            // https://github.com/llvm/llvm-project/issues/107198
164            OptLevel::Default => 2,
165            OptLevel::Aggressive => 3,
166        };
167        let passes = CString::new(format!("default<O{opt}>"))
168            .to_native_assert_error("only fails if the hardcoded string contains a null byte")?;
169
170        trace!("starting llvm passes");
171        let pre_passes_instant = Instant::now();
172        let error = LLVMRunPasses(llvm_module, passes.as_ptr(), machine, opts);
173        let passes_time = pre_passes_instant.elapsed().as_millis();
174        trace!(time = passes_time, "llvm passes finished");
175
176        if !error.is_null() {
177            let msg = LLVMGetErrorMessage(error);
178            let msg = CStr::from_ptr(msg);
179            Err(Error::LLVMCompileError(msg.to_string_lossy().into_owned()))?;
180        }
181
182        LLVMDisposePassBuilderOptions(opts);
183
184        let mut out_buf: MaybeUninit<LLVMMemoryBufferRef> = MaybeUninit::uninit();
185
186        trace!("starting llvm to object compilation");
187        let pre_llvm_compilation_instant = Instant::now();
188        let ok = LLVMTargetMachineEmitToMemoryBuffer(
189            machine,
190            llvm_module,
191            LLVMCodeGenFileType::LLVMObjectFile,
192            error_buffer,
193            out_buf.as_mut_ptr(),
194        );
195        let llvm_compilation_time = pre_llvm_compilation_instant.elapsed().as_millis();
196        trace!(
197            time = llvm_compilation_time,
198            "llvm to object compilation finished"
199        );
200
201        if ok != 0 {
202            let error = CStr::from_ptr(*error_buffer);
203            let err = error.to_string_lossy().to_string();
204            LLVMDisposeMessage(*error_buffer);
205            Err(Error::LLVMCompileError(err))?;
206        } else if !(*error_buffer).is_null() {
207            LLVMDisposeMessage(*error_buffer);
208        }
209
210        let out_buf = out_buf.assume_init();
211
212        let out_buf_start: *const u8 = LLVMGetBufferStart(out_buf).cast();
213        let out_buf_size = LLVMGetBufferSize(out_buf);
214
215        // keep it in rust side
216        let data = std::slice::from_raw_parts(out_buf_start, out_buf_size).to_vec();
217
218        LLVMDisposeMemoryBuffer(out_buf);
219        LLVMDisposeTargetMachine(machine);
220        LLVMDisposeModule(llvm_module);
221        LLVMContextDispose(llvm_context);
222
223        Ok(data)
224    }
225}

227/// Links the passed object into a shared library, stored on the given path.
228pub fn object_to_shared_lib(object: &[u8], output_filename: &Path) -> Result<()> {
229    // linker seems to need a file and doesn't accept stdin
230    let mut file = NamedTempFile::new()?;
231    file.write_all(object)?;
232    let file = file.into_temp_path();
233
234    let file_path = file.display().to_string();
235    let output_path = output_filename.display().to_string();
236    if let Ok(x) = std::env::var("NATIVE_DEBUG_DUMP") {
237        if x == "1" || x == "true" {
238            // forget so the temp file is not deleted and the debugger can load it.
239            // its still in a temp file directory so eventually the OS will delete it, but just not instantly.
240            // todo: maybe remove it when exiting, for example using atexit.
241            std::mem::forget(file);
242        }
243    }
244
245    let args: Vec<Cow<'static, str>> = {
246        #[cfg(target_os = "macos")]
247        {
248            let mut args: Vec<Cow<'static, str>> = vec![
249                "-demangle".into(),
250                "-no_deduplicate".into(),
251                "-dynamic".into(),
252                "-dylib".into(),
253                "-L/usr/local/lib".into(),
254                "-L/Library/Developer/CommandLineTools/SDKs/MacOSX.sdk/usr/lib".into(),
255            ];
256
257            args.extend([
258                Cow::from(file_path),
259                "-o".into(),
260                Cow::from(output_path),
261                "-lSystem".into(),
262            ]);
263
264            args
265        }
266        #[cfg(target_os = "linux")]
267        {
268            let mut args: Vec<Cow<'static, str>> = vec![
269                "--hash-style=gnu".into(),
270                "-shared".into(),
271                "-L/lib/../lib64".into(),
272                "-L/usr/lib/../lib64".into(),
273            ];
274
275            args.extend([
276                "-o".into(),
277                Cow::from(output_path),
278                "-lc".into(),
279                Cow::from(file_path),
280            ]);
281
282            args
283        }
284        #[cfg(target_os = "windows")]
285        {
286            unimplemented!()
287        }
288    };
289
290    let mut linker = std::process::Command::new("ld");
291
292    trace!("starting linking");
293    let pre_linking_instant = Instant::now();
294    let proc = linker.args(args.iter().map(|x| x.as_ref())).output()?;
295    let linking_time = pre_linking_instant.elapsed().as_millis();
296    trace!(time = linking_time, "linking finished");
297
298    if proc.status.success() {
299        Ok(())
300    } else {
301        let msg = String::from_utf8_lossy(&proc.stderr);
302        Err(Error::LinkError(msg.to_string()))
303    }
304}

306/// Gets the target triple, which identifies the platform and ABI.
307pub fn get_target_triple() -> String {
308    let target_triple = unsafe {
309        let value = LLVMGetDefaultTargetTriple();
310        CStr::from_ptr(value).to_string_lossy().into_owned()
311    };
312    target_triple
313}

315/// Gets the data layout reprrsentation as a string, to be given to the MLIR module.
316/// LLVM uses this to know the proper alignments for the given sizes, etc.
317/// This function gets the data layout of the host target triple.
318pub fn get_data_layout_rep() -> Result<String> {
319    unsafe {
320        let mut null = null_mut();
321        let error_buffer = addr_of_mut!(null);
322
323        let target_triple = LLVMGetDefaultTargetTriple();
324
325        let target_cpu = LLVMGetHostCPUName();
326
327        let target_cpu_features = LLVMGetHostCPUFeatures();
328
329        let mut target: MaybeUninit<LLVMTargetRef> = MaybeUninit::uninit();
330
331        if LLVMGetTargetFromTriple(target_triple, target.as_mut_ptr(), error_buffer) != 0 {
332            let error = CStr::from_ptr(*error_buffer);
333            let err = error.to_string_lossy().to_string();
334            tracing::error!("error getting target triple: {}", err);
335            LLVMDisposeMessage(*error_buffer);
336            Err(Error::LLVMCompileError(err))?;
337        }
338        if !(*error_buffer).is_null() {
339            LLVMDisposeMessage(*error_buffer);
340        }
341
342        let target = target.assume_init();
343
344        let machine = LLVMCreateTargetMachine(
345            target,
346            target_triple.cast(),
347            target_cpu.cast(),
348            target_cpu_features.cast(),
349            LLVMCodeGenOptLevel::LLVMCodeGenLevelNone,
350            LLVMRelocMode::LLVMRelocDynamicNoPic,
351            LLVMCodeModel::LLVMCodeModelDefault,
352        );
353
354        let data_layout = llvm_sys::target_machine::LLVMCreateTargetDataLayout(machine);
355        let data_layout_str =
356            CStr::from_ptr(llvm_sys::target::LLVMCopyStringRepOfTargetData(data_layout));
357        Ok(data_layout_str.to_string_lossy().into_owned())
358    }
359}

365    #[test]
366    fn test_opt_level_default() {
367        // Asserts that the default implementation of `OptLevel` returns `OptLevel::Default`.
368        assert_eq!(OptLevel::default(), OptLevel::Default);
369
370        // Asserts that converting from usize value 2 returns `OptLevel::Default`.
371        assert_eq!(OptLevel::from(2usize), OptLevel::Default);
372
373        // Asserts that converting from u8 value 2 returns `OptLevel::Default`.
374        assert_eq!(OptLevel::from(2u8), OptLevel::Default);
375    }
376
377    #[test]
378    fn test_opt_level_conversion() {
379        // Test conversion from usize to OptLevel
380        assert_eq!(OptLevel::from(0usize), OptLevel::None);
381        assert_eq!(OptLevel::from(1usize), OptLevel::Less);
382        assert_eq!(OptLevel::from(2usize), OptLevel::Default);
383        assert_eq!(OptLevel::from(3usize), OptLevel::Aggressive);
384        assert_eq!(OptLevel::from(30usize), OptLevel::Aggressive);
385
386        // Test conversion from OptLevel to usize
387        assert_eq!(usize::from(OptLevel::None), 0usize);
388        assert_eq!(usize::from(OptLevel::Less), 1usize);
389        assert_eq!(usize::from(OptLevel::Default), 2usize);
390        assert_eq!(usize::from(OptLevel::Aggressive), 3usize);
391
392        // Test conversion from u8 to OptLevel

394        assert_eq!(OptLevel::from(1u8), OptLevel::Less);
395        assert_eq!(OptLevel::from(2u8), OptLevel::Default);
396        assert_eq!(OptLevel::from(3u8), OptLevel::Aggressive);
397        assert_eq!(OptLevel::from(30u8), OptLevel::Aggressive);
