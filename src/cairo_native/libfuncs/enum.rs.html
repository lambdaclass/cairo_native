34/// Select and call the correct libfunc builder function from the selector.
35pub fn build<'ctx, 'this>(
36    context: &'ctx Context,
37    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
38    entry: &'this Block<'ctx>,
39    location: Location<'ctx>,
40    helper: &LibfuncHelper<'ctx, 'this>,
41    metadata: &mut MetadataStorage,
42    selector: &EnumConcreteLibfunc,
43) -> Result<()> {
44    match selector {
45        EnumConcreteLibfunc::Init(info) => {
46            build_init(context, registry, entry, location, helper, metadata, info)
47        }
48        EnumConcreteLibfunc::Match(info) => {
49            build_match(context, registry, entry, location, helper, metadata, info)
50        }
51        EnumConcreteLibfunc::SnapshotMatch(info) => {
52            build_snapshot_match(context, registry, entry, location, helper, metadata, info)
53        }
54        EnumConcreteLibfunc::FromBoundedInt(info) => {
55            build_from_bounded_int(context, registry, entry, location, helper, metadata, info)
56        }
57    }
58}

60/// Generate MLIR operations for the `enum_init` libfunc.
61pub fn build_init<'ctx, 'this>(
62    context: &'ctx Context,
63    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
64    entry: &'this Block<'ctx>,
65    location: Location<'ctx>,
66    helper: &LibfuncHelper<'ctx, 'this>,
67    metadata: &mut MetadataStorage,
68    info: &EnumInitConcreteLibfunc,
69) -> Result<()> {
70    #[cfg(feature = "with-debug-utils")]
71    if let Some(auto_breakpoint) =
72        metadata.get::<crate::metadata::auto_breakpoint::AutoBreakpoint>()
73    {
74        auto_breakpoint.maybe_breakpoint(
75            entry,
76            location,
77            metadata,
78            &crate::metadata::auto_breakpoint::BreakpointEvent::EnumInit {
79                type_id: info.signature.branch_signatures[0].vars[0].ty.clone(),
80                variant_idx: info.index,
81            },
82        )?;
83    }
84
85    let val = build_enum_value(
86        context,
87        registry,
88        entry,
89        location,
90        helper,
91        metadata,
92        entry.arg(0)?,
93        &info.branch_signatures()[0].vars[0].ty,
94        &info.signature.param_signatures[0].ty,
95        info.index,
96    )?;
97
98    helper.br(entry, 0, &[val], location)
99}

101#[allow(clippy::too_many_arguments)]
102pub fn build_enum_value<'ctx, 'this>(
103    context: &'ctx Context,
104    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
105    entry: &'this Block<'ctx>,
106    location: Location<'ctx>,
107    helper: &LibfuncHelper<'ctx, 'this>,
108    metadata: &mut MetadataStorage,
109    payload_value: Value<'ctx, 'this>,
110    enum_type: &ConcreteTypeId,
111    variant_type: &ConcreteTypeId,
112    variant_index: usize,
113) -> Result<Value<'ctx, 'this>> {
114    let type_info = registry.get_type(enum_type)?;
115    let payload_type_info = registry.get_type(variant_type)?;
116
117    let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
118        context,
119        helper,
120        registry,
121        metadata,
122        type_info
123            .variants()
124            .to_native_assert_error("found non-enum type where an enum is required")?,
125    )?;
126
127    Ok(match variant_tys.len() {
128        0 => native_panic!("attempt to initialize a zero-variant enum"),
129        1 => payload_value,
130        _ => {
131            let enum_ty = llvm::r#type::r#struct(
132                context,
133                &[
134                    tag_ty,
135                    if payload_type_info.is_zst(registry)? {
136                        llvm::r#type::array(IntegerType::new(context, 8).into(), 0)
137                    } else {
138                        variant_tys[variant_index].0
139                    },
140                ],
141                false,
142            );
143
144            let tag_val = entry
145                .append_operation(arith::constant(
146                    context,
147                    IntegerAttribute::new(tag_ty, variant_index.try_into()?).into(),
148                    location,
149                ))
150                .result(0)?
151                .into();
152
153            let val = entry.append_op_result(llvm::undef(enum_ty, location))?;
154            let val = entry.insert_value(context, location, val, tag_val, 0)?;
155
156            let mut val = if payload_type_info.is_zst(registry)? {
157                val
158            } else {
159                entry.insert_value(context, location, val, payload_value, 1)?
160            };
161
162            if type_info.is_memory_allocated(registry)? {
163                let stack_ptr = helper.init_block().alloca1(
164                    context,
165                    location,
166                    type_info.build(context, helper, registry, metadata, enum_type)?,
167                    layout.align(),
168                )?;
169
170                // Convert the enum from the concrete variant to the internal representation.
171                entry.store(context, location, stack_ptr, val)?;
172                val = entry.load(
173                    context,
174                    location,
175                    stack_ptr,
176                    type_info.build(context, helper, registry, metadata, enum_type)?,
177                )?;
178            };
179
180            val
181        }
182    })
183}

185/// Generate MLIR operations for the `enum_from_bounded_int` libfunc.
186pub fn build_from_bounded_int<'ctx, 'this>(
187    context: &'ctx Context,
188    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
189    entry: &'this Block<'ctx>,
190    location: Location<'ctx>,
191    helper: &LibfuncHelper<'ctx, 'this>,
192    metadata: &mut MetadataStorage,
193    info: &EnumFromBoundedIntConcreteLibfunc,
194) -> Result<()> {
195    let inp_ty = registry.get_type(&info.param_signatures()[0].ty)?;
196    let varaint_selector_type: IntegerType = inp_ty
197        .build(
198            context,
199            helper,
200            registry,
201            metadata,
202            &info.param_signatures()[0].ty,
203        )?
204        .try_into()?;
205    let enum_type = registry.get_type(&info.branch_signatures()[0].vars[0].ty)?;
206    // we assume its never memory allocated since its always an enum with only a tag
207    native_assert!(
208        !enum_type.is_memory_allocated(registry)?,
209        "an enum with only a tag should not be allocated in memory"
210    );
211
212    let enum_ty = enum_type.build(
213        context,
214        helper,
215        registry,
216        metadata,
217        &info.branch_signatures()[0].vars[0].ty,
218    )?;
219
220    let tag_bits = info.n_variants.next_power_of_two().trailing_zeros();
221    let tag_type = IntegerType::new(context, tag_bits);
222
223    let mut tag_value: Value = entry.arg(0)?;
224
225    match tag_type.width().cmp(&varaint_selector_type.width()) {
226        std::cmp::Ordering::Less => {
227            tag_value = entry.append_op_result(
228                ods::llvm::trunc(context, tag_type.into(), tag_value, location).into(),
229            )?;
230        }
231        std::cmp::Ordering::Equal => {}
232        std::cmp::Ordering::Greater => {
233            tag_value = entry.append_op_result(
234                ods::llvm::zext(context, tag_type.into(), tag_value, location).into(),
235            )?;
236        }
237    };
238
239    let value = entry.append_op_result(llvm::undef(enum_ty, location))?;
240    let value = entry.insert_value(context, location, value, tag_value, 0)?;
241
242    helper.br(entry, 0, &[value], location)
243}

245/// Generate MLIR operations for the `enum_match` libfunc.
246pub fn build_match<'ctx, 'this>(
247    context: &'ctx Context,
248    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
249    entry: &'this Block<'ctx>,
250    location: Location<'ctx>,
251    helper: &LibfuncHelper<'ctx, 'this>,
252    metadata: &mut MetadataStorage,
253    info: &SignatureOnlyConcreteLibfunc,
254) -> Result<()> {
255    let type_info = registry.get_type(&info.param_signatures()[0].ty)?;
256
257    let variant_ids = type_info
258        .variants()
259        .to_native_assert_error("found non-enum type where an enum is required")?;
260    match variant_ids.len() {
261        0 => {
262            // The Cairo compiler will generate an enum match for enums without variants, so this
263            // case cannot be a compile-time error. We're assuming that even though it's been
264            // generated, it's just dead code and can be made into an assertion that always fails.
265
266            let k0 = entry.const_int(context, location, 0, 1)?;
267            entry.append_operation(cf::assert(
268                context,
269                k0,
270                "attempt to match a zero-variant enum",
271                location,
272            ));
273
274            entry.append_operation(llvm::unreachable(location));
275        }
276        1 => {
277            helper.br(entry, 0, &[entry.arg(0)?], location)?;
278        }
279        _ => {
280            let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
281                context,
282                helper,
283                registry,
284                metadata,
285                variant_ids,
286            )?;
287
288            let (stack_ptr, tag_val) = if type_info.is_memory_allocated(registry)? {
289                let stack_ptr = helper.init_block().alloca1(
290                    context,
291                    location,
292                    type_info.build(
293                        context,
294                        helper,
295                        registry,
296                        metadata,
297                        &info.param_signatures()[0].ty,
298                    )?,
299                    layout.align(),
300                )?;
301                entry.store(context, location, stack_ptr, entry.arg(0)?)?;
302                let tag_val = entry.load(context, location, stack_ptr, tag_ty)?;
303
304                (Some(stack_ptr), tag_val)
305            } else {
306                let tag_val = entry
307                    .append_operation(llvm::extract_value(
308                        context,
309                        entry.arg(0)?,
310                        DenseI64ArrayAttribute::new(context, &[0]),
311                        tag_ty,
312                        location,
313                    ))
314                    .result(0)?
315                    .into();
316
317                (None, tag_val)
318            };
319
320            let default_block = helper.append_block(Block::new(&[]));
321            let variant_blocks = variant_tys
322                .iter()
323                .map(|_| helper.append_block(Block::new(&[])))
324                .collect::<Vec<_>>();
325
326            let case_values = (0..variant_tys.len())
327                .map(i64::try_from)
328                .collect::<std::result::Result<Vec<_>, TryFromIntError>>()?;
329
330            entry.append_operation(cf::switch(
331                context,
332                &case_values,
333                tag_val,
334                tag_ty,
335                (default_block, &[]),
336                &variant_blocks
337                    .iter()
338                    .copied()
339                    .map(|block| (block, [].as_slice()))
340                    .collect::<Vec<_>>(),
341                location,
342            )?);
343
344            // Default block.
345            {
346                let val = default_block
347                    .append_operation(arith::constant(
348                        context,
349                        IntegerAttribute::new(IntegerType::new(context, 1).into(), 0).into(),
350                        location,
351                    ))
352                    .result(0)?
353                    .into();
354
355                default_block.append_operation(cf::assert(
356                    context,
357                    val,
358                    "Invalid enum tag.",
359                    location,
360                ));
361                default_block.append_operation(llvm::unreachable(location));
362            }
363
364            // Enum variants.
365            for (i, (block, (payload_ty, _))) in
366                variant_blocks.into_iter().zip(variant_tys).enumerate()
367            {
368                let enum_ty = llvm::r#type::r#struct(context, &[tag_ty, payload_ty], false);
369
370                let payload_val = match stack_ptr {
371                    Some(stack_ptr) => {
372                        let val = block.load(context, location, stack_ptr, enum_ty)?;
373                        block.extract_value(context, location, val, payload_ty, 1)?
374                    }
375                    None => {
376                        // If the enum is not memory-allocated it means that:
377                        //   - Either it's a C-style enum and all payloads have the same type.
378                        //   - Or the enum only has a single non-memory-allocated variant.
379                        if variant_ids.len() == 1 {
380                            entry.arg(0)?
381                        } else {
382                            native_assert!(
383                                registry.get_type(&variant_ids[i])?.is_zst(registry)?,
384                                "should be zero sized"
385                            );
386                            block
387                                .append_operation(llvm::undef(payload_ty, location))
388                                .result(0)?
389                                .into()
390                        }
391                    }
392                };
393
394                helper.br(block, i, &[payload_val], location)?;
395            }
396        }
397    }
398
399    Ok(())
400}

402/// Generate MLIR operations for the `enum_snapshot_match` libfunc.
403pub fn build_snapshot_match<'ctx, 'this>(
404    context: &'ctx Context,
405    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
406    entry: &'this Block<'ctx>,
407    location: Location<'ctx>,
408    helper: &LibfuncHelper<'ctx, 'this>,
409    metadata: &mut MetadataStorage,
410    info: &SignatureOnlyConcreteLibfunc,
411) -> Result<()> {
412    let type_info = registry.get_type(&info.param_signatures()[0].ty)?;
413
414    // This libfunc's implementation is identical to `enum_match` aside from fetching the snapshotted enum's variants from the metadata:
415    let variant_ids = metadata
416        .get::<EnumSnapshotVariantsMeta>()
417        .ok_or(Error::MissingMetadata)?
418        .get_variants(&info.param_signatures()[0].ty)
419        .to_native_assert_error("enum should always have variants")?
420        .clone();
421    match variant_ids.len() {
422        0 => {
423            // The Cairo compiler will generate an enum match for enums without variants, so this
424            // case cannot be a compile-time error. We're assuming that even though it's been
425            // generated, it's just dead code and can be made into an assertion that always fails.
426
427            let k0 = entry.const_int(context, location, 0, 1)?;
428            entry.append_operation(cf::assert(
429                context,
430                k0,
431                "attempt to match a zero-variant enum",
432                location,
433            ));
434
435            entry.append_operation(llvm::unreachable(location));
436        }
437        1 => {
438            helper.br(entry, 0, &[entry.arg(0)?], location)?;
439        }
440        _ => {
441            let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
442                context,
443                helper,
444                registry,
445                metadata,
446                &variant_ids,
447            )?;
448
449            let (stack_ptr, tag_val) = if type_info.is_memory_allocated(registry)? {
450                let stack_ptr = helper.init_block().alloca1(
451                    context,
452                    location,
453                    type_info.build(
454                        context,
455                        helper,
456                        registry,
457                        metadata,
458                        &info.param_signatures()[0].ty,
459                    )?,
460                    layout.align(),
461                )?;
462                entry.store(context, location, stack_ptr, entry.arg(0)?)?;
463                let tag_val = entry.load(context, location, stack_ptr, tag_ty)?;
464
465                (Some(stack_ptr), tag_val)
466            } else {
467                let tag_val = entry.extract_value(context, location, entry.arg(0)?, tag_ty, 0)?;
468
469                (None, tag_val)
470            };
471
472            let default_block = helper.append_block(Block::new(&[]));
473            let variant_blocks = variant_tys
474                .iter()
475                .map(|_| helper.append_block(Block::new(&[])))
476                .collect::<Vec<_>>();
477
478            let case_values = (0..variant_tys.len())
479                .map(i64::try_from)
480                .collect::<std::result::Result<Vec<_>, TryFromIntError>>()?;
481
482            entry.append_operation(cf::switch(
483                context,
484                &case_values,
485                tag_val,
486                tag_ty,
487                (default_block, &[]),
488                &variant_blocks
489                    .iter()
490                    .copied()
491                    .map(|block| (block, [].as_slice()))
492                    .collect::<Vec<_>>(),
493                location,
494            )?);
495
496            // Default block.
497            {
498                let val = default_block.const_int(context, location, 0, 1)?;
499
500                default_block.append_operation(cf::assert(
501                    context,
502                    val,
503                    "Invalid enum tag.",
504                    location,
505                ));
506                default_block.append_operation(llvm::unreachable(location));
507            }
508
509            // Enum variants.
510            for (i, (block, (payload_ty, _))) in
511                variant_blocks.into_iter().zip(variant_tys).enumerate()
512            {
513                let enum_ty = llvm::r#type::r#struct(context, &[tag_ty, payload_ty], false);
514
515                let payload_val = match stack_ptr {
516                    Some(stack_ptr) => {
517                        let val = block.load(context, location, stack_ptr, enum_ty)?;
518                        block.extract_value(context, location, val, payload_ty, 1)?
519                    }
520                    None => {
521                        // If the enum is not memory-allocated it means that:
522                        //   - Either it's a C-style enum and all payloads have the same type.
523                        //   - Or the enum only has a single non-memory-allocated variant.
524                        if variant_ids.len() == 1 {
525                            entry.arg(0)?
526                        } else {
527                            native_assert!(
528                                registry.get_type(&variant_ids[i])?.is_zst(registry)?,
529                                "should be zero sized"
530                            );
531                            block.append_op_result(llvm::undef(payload_ty, location))?
532                        }
533                    }
534                };
535
536                helper.br(block, i, &[payload_val], location)?;
537            }
538        }
539    }
540
541    Ok(())
542}
