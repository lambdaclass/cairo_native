42/// Select and call the correct libfunc builder function from the selector.
43pub fn build<'ctx, 'this>(
44    context: &'ctx Context,
45    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
46    entry: &'this Block<'ctx>,
47    location: Location<'ctx>,
48    helper: &LibfuncHelper<'ctx, 'this>,
49    metadata: &mut MetadataStorage,
50    selector: &EnumConcreteLibfunc,
51) -> Result<()> {
52    match selector {
53        EnumConcreteLibfunc::Init(info) => {
54            build_init(context, registry, entry, location, helper, metadata, info)
55        }
56        EnumConcreteLibfunc::Match(info) => {
57            build_match(context, registry, entry, location, helper, metadata, info)
58        }
59        EnumConcreteLibfunc::SnapshotMatch(info) => {
60            build_snapshot_match(context, registry, entry, location, helper, metadata, info)
61        }
62        EnumConcreteLibfunc::FromBoundedInt(info) => {
63            build_from_bounded_int(context, registry, entry, location, helper, metadata, info)
64        }
65        EnumConcreteLibfunc::BoxedMatch(info) => {
66            build_boxed_match(context, registry, entry, location, helper, metadata, info)
67        }
68    }
69}

71/// Generate MLIR operations for the `enum_init` libfunc.
72pub fn build_init<'ctx, 'this>(
73    context: &'ctx Context,
74    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
75    entry: &'this Block<'ctx>,
76    location: Location<'ctx>,
77    helper: &LibfuncHelper<'ctx, 'this>,
78    metadata: &mut MetadataStorage,
79    info: &EnumInitConcreteLibfunc,
80) -> Result<()> {
81    #[cfg(feature = "with-debug-utils")]
82    if let Some(auto_breakpoint) =
83        metadata.get::<crate::metadata::auto_breakpoint::AutoBreakpoint>()
84    {
85        auto_breakpoint.maybe_breakpoint(
86            entry,
87            location,
88            metadata,
89            &crate::metadata::auto_breakpoint::BreakpointEvent::EnumInit {
90                type_id: info.signature.branch_signatures[0].vars[0].ty.clone(),
91                variant_idx: info.index,
92            },
93        )?;
94    }
95
96    let val = build_enum_value(
97        context,
98        registry,
99        entry,
100        location,
101        helper,
102        metadata,
103        entry.arg(0)?,
104        &info.branch_signatures()[0].vars[0].ty,
105        &info.signature.param_signatures[0].ty,
106        info.index,
107    )?;
108
109    helper.br(entry, 0, &[val], location)
110}

112#[allow(clippy::too_many_arguments)]
113pub fn build_enum_value<'ctx, 'this>(
114    context: &'ctx Context,
115    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
116    entry: &'this Block<'ctx>,
117    location: Location<'ctx>,
118    helper: &LibfuncHelper<'ctx, 'this>,
119    metadata: &mut MetadataStorage,
120    payload_value: Value<'ctx, 'this>,
121    enum_type: &ConcreteTypeId,
122    variant_type: &ConcreteTypeId,
123    variant_index: usize,
124) -> Result<Value<'ctx, 'this>> {
125    let type_info = registry.get_type(enum_type)?;
126    let payload_type_info = registry.get_type(variant_type)?;
127
128    let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
129        context,
130        helper,
131        registry,
132        metadata,
133        type_info
134            .variants()
135            .to_native_assert_error("found non-enum type where an enum is required")?,
136    )?;
137
138    Ok(match variant_tys.len() {
139        0 => native_panic!("attempt to initialize a zero-variant enum"),
140        1 => payload_value,
141        _ => {
142            let enum_ty = llvm::r#type::r#struct(
143                context,
144                &[
145                    tag_ty,
146                    if payload_type_info.is_zst(registry)? {
147                        llvm::r#type::array(IntegerType::new(context, 8).into(), 0)
148                    } else {
149                        variant_tys[variant_index].0
150                    },
151                ],
152                false,
153            );
154
155            let tag_val = entry
156                .append_operation(arith::constant(
157                    context,
158                    IntegerAttribute::new(tag_ty, variant_index.try_into()?).into(),
159                    location,
160                ))
161                .result(0)?
162                .into();
163
164            let val = entry.append_op_result(llvm::undef(enum_ty, location))?;
165            let val = entry.insert_value(context, location, val, tag_val, 0)?;
166
167            let mut val = if payload_type_info.is_zst(registry)? {
168                val
169            } else {
170                entry.insert_value(context, location, val, payload_value, 1)?
171            };
172
173            if type_info.is_memory_allocated(registry)? {
174                let stack_ptr = helper.init_block().alloca1(
175                    context,
176                    location,
177                    type_info.build(context, helper, registry, metadata, enum_type)?,
178                    layout.align(),
179                )?;
180
181                // Convert the enum from the concrete variant to the internal representation.
182                entry.store(context, location, stack_ptr, val)?;
183                val = entry.load(
184                    context,
185                    location,
186                    stack_ptr,
187                    type_info.build(context, helper, registry, metadata, enum_type)?,
188                )?;
189            };
190
191            val
192        }
193    })
194}

196/// Generate MLIR operations for the `enum_from_bounded_int` libfunc.
197///
198/// # Constraints
199///
200/// - The target `Enum` must contain the same number of empty variants as the number
201///   of possible values in the `BoundedInt` range.
202/// - The range of the `BoundedInt` must start from **0**.
203///
204/// # Signature
205///
206/// ```cairo
207/// fn enum_from_bounded_int<T, U>(index: U) -> T nopanic
208/// ```
209pub fn build_from_bounded_int<'ctx, 'this>(
210    context: &'ctx Context,
211    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
212    entry: &'this Block<'ctx>,
213    location: Location<'ctx>,
214    helper: &LibfuncHelper<'ctx, 'this>,
215    metadata: &mut MetadataStorage,
216    info: &EnumFromBoundedIntConcreteLibfunc,
217) -> Result<()> {
218    let inp_ty = registry.get_type(&info.param_signatures()[0].ty)?;
219    let varaint_selector_type: IntegerType = inp_ty
220        .build(
221            context,
222            helper,
223            registry,
224            metadata,
225            &info.param_signatures()[0].ty,
226        )?
227        .try_into()?;
228    let enum_type = registry.get_type(&info.branch_signatures()[0].vars[0].ty)?;
229
230    let enum_ty = enum_type.build(
231        context,
232        helper,
233        registry,
234        metadata,
235        &info.branch_signatures()[0].vars[0].ty,
236    )?;
237
238    let tag_bits = info.n_variants.next_power_of_two().trailing_zeros();
239    let tag_type = IntegerType::new(context, tag_bits);
240
241    let mut tag_value: Value = entry.arg(0)?;
242
243    match tag_type.width().cmp(&varaint_selector_type.width()) {
244        std::cmp::Ordering::Less => {
245            tag_value = entry.append_op_result(
246                ods::llvm::trunc(context, tag_type.into(), tag_value, location).into(),
247            )?;
248        }
249        std::cmp::Ordering::Equal => {}
250        std::cmp::Ordering::Greater => {
251            tag_value = entry.append_op_result(
252                ods::llvm::zext(context, tag_type.into(), tag_value, location).into(),
253            )?;
254        }
255    };
256
257    let mut value = entry.append_op_result(llvm::undef(enum_ty, location))?;
258    if info.n_variants > 1 {
259        value = entry.insert_value(context, location, value, tag_value, 0)?;
260    }
261
262    helper.br(entry, 0, &[value], location)
263}

265/// Generate MLIR operations for the `enum_match` libfunc.
266pub fn build_match<'ctx, 'this>(
267    context: &'ctx Context,
268    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
269    entry: &'this Block<'ctx>,
270    location: Location<'ctx>,
271    helper: &LibfuncHelper<'ctx, 'this>,
272    metadata: &mut MetadataStorage,
273    info: &SignatureOnlyConcreteLibfunc,
274) -> Result<()> {
275    let type_info = registry.get_type(&info.param_signatures()[0].ty)?;
276
277    let variant_ids = type_info
278        .variants()
279        .to_native_assert_error("found non-enum type where an enum is required")?;
280    match variant_ids.len() {
281        0 => {
282            // The Cairo compiler will generate an enum match for enums without variants, so this
283            // case cannot be a compile-time error. We're assuming that even though it's been
284            // generated, it's just dead code and can be made into an assertion that always fails.
285
286            let k0 = entry.const_int(context, location, 0, 1)?;
287            entry.append_operation(cf::assert(
288                context,
289                k0,
290                "attempt to match a zero-variant enum",
291                location,
292            ));
293
294            entry.append_operation(llvm::unreachable(location));
295        }
296        1 => {
297            helper.br(entry, 0, &[entry.arg(0)?], location)?;
298        }
299        _ => {
300            let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
301                context,
302                helper,
303                registry,
304                metadata,
305                variant_ids,
306            )?;
307
308            let (stack_ptr, tag_val) = if type_info.is_memory_allocated(registry)? {
309                let stack_ptr = helper.init_block().alloca1(
310                    context,
311                    location,
312                    type_info.build(
313                        context,
314                        helper,
315                        registry,
316                        metadata,
317                        &info.param_signatures()[0].ty,
318                    )?,
319                    layout.align(),
320                )?;
321                entry.store(context, location, stack_ptr, entry.arg(0)?)?;
322                let tag_val = entry.load(context, location, stack_ptr, tag_ty)?;
323
324                (Some(stack_ptr), tag_val)
325            } else {
326                let tag_val = entry
327                    .append_operation(llvm::extract_value(
328                        context,
329                        entry.arg(0)?,
330                        DenseI64ArrayAttribute::new(context, &[0]),
331                        tag_ty,
332                        location,
333                    ))
334                    .result(0)?
335                    .into();
336
337                (None, tag_val)
338            };
339
340            let default_block = helper.append_block(Block::new(&[]));
341            let variant_blocks = variant_tys
342                .iter()
343                .map(|_| helper.append_block(Block::new(&[])))
344                .collect::<Vec<_>>();
345
346            let case_values = (0..variant_tys.len())
347                .map(i64::try_from)
348                .collect::<std::result::Result<Vec<_>, TryFromIntError>>()?;
349
350            entry.append_operation(cf::switch(
351                context,
352                &case_values,
353                tag_val,
354                tag_ty,
355                (default_block, &[]),
356                &variant_blocks
357                    .iter()
358                    .copied()
359                    .map(|block| (block, [].as_slice()))
360                    .collect::<Vec<_>>(),
361                location,
362            )?);
363
364            // Default block.
365            {
366                let val = default_block
367                    .append_operation(arith::constant(
368                        context,
369                        IntegerAttribute::new(IntegerType::new(context, 1).into(), 0).into(),
370                        location,
371                    ))
372                    .result(0)?
373                    .into();
374
375                default_block.append_operation(cf::assert(
376                    context,
377                    val,
378                    "Invalid enum tag.",
379                    location,
380                ));
381                default_block.append_operation(llvm::unreachable(location));
382            }
383
384            // Enum variants.
385            for (i, (block, (payload_ty, _))) in
386                variant_blocks.into_iter().zip(variant_tys).enumerate()
387            {
388                let enum_ty = llvm::r#type::r#struct(context, &[tag_ty, payload_ty], false);
389
390                let payload_val = match stack_ptr {
391                    Some(stack_ptr) => {
392                        let val = block.load(context, location, stack_ptr, enum_ty)?;
393                        block.extract_value(context, location, val, payload_ty, 1)?
394                    }
395                    None => {
396                        // If the enum is not memory-allocated it means that:
397                        //   - Either it's a C-style enum and all payloads have the same type.
398                        //   - Or the enum only has a single non-memory-allocated variant.
399                        if variant_ids.len() == 1 {
400                            entry.arg(0)?
401                        } else {
402                            native_assert!(
403                                registry.get_type(&variant_ids[i])?.is_zst(registry)?,
404                                "should be zero sized"
405                            );
406                            block
407                                .append_operation(llvm::undef(payload_ty, location))
408                                .result(0)?
409                                .into()
410                        }
411                    }
412                };
413
414                helper.br(block, i, &[payload_val], location)?;
415            }
416        }
417    }
418
419    Ok(())
420}

422/// Generate MLIR operations for the `enum_snapshot_match` libfunc.
423pub fn build_snapshot_match<'ctx, 'this>(
424    context: &'ctx Context,
425    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
426    entry: &'this Block<'ctx>,
427    location: Location<'ctx>,
428    helper: &LibfuncHelper<'ctx, 'this>,
429    metadata: &mut MetadataStorage,
430    info: &SignatureOnlyConcreteLibfunc,
431) -> Result<()> {
432    let type_info = registry.get_type(&info.param_signatures()[0].ty)?;
433
434    // This libfunc's implementation is identical to `enum_match` aside from fetching the snapshotted enum's variants from the metadata:
435    let variant_ids = metadata
436        .get::<EnumSnapshotVariantsMeta>()
437        .ok_or(Error::MissingMetadata)?
438        .get_variants(&info.param_signatures()[0].ty)
439        .to_native_assert_error("enum should always have variants")?
440        .clone();
441    match variant_ids.len() {
442        0 => {
443            // The Cairo compiler will generate an enum match for enums without variants, so this
444            // case cannot be a compile-time error. We're assuming that even though it's been
445            // generated, it's just dead code and can be made into an assertion that always fails.
446
447            let k0 = entry.const_int(context, location, 0, 1)?;
448            entry.append_operation(cf::assert(
449                context,
450                k0,
451                "attempt to match a zero-variant enum",
452                location,
453            ));
454
455            entry.append_operation(llvm::unreachable(location));
456        }
457        1 => {
458            helper.br(entry, 0, &[entry.arg(0)?], location)?;
459        }
460        _ => {
461            let (layout, (tag_ty, _), variant_tys) = crate::types::r#enum::get_type_for_variants(
462                context,
463                helper,
464                registry,
465                metadata,
466                &variant_ids,
467            )?;
468
469            let (stack_ptr, tag_val) = if type_info.is_memory_allocated(registry)? {
470                let stack_ptr = helper.init_block().alloca1(
471                    context,
472                    location,
473                    type_info.build(
474                        context,
475                        helper,
476                        registry,
477                        metadata,
478                        &info.param_signatures()[0].ty,
479                    )?,
480                    layout.align(),
481                )?;
482                entry.store(context, location, stack_ptr, entry.arg(0)?)?;
483                let tag_val = entry.load(context, location, stack_ptr, tag_ty)?;
484
485                (Some(stack_ptr), tag_val)
486            } else {
487                let tag_val = entry.extract_value(context, location, entry.arg(0)?, tag_ty, 0)?;
488
489                (None, tag_val)
490            };
491
492            let default_block = helper.append_block(Block::new(&[]));
493            let variant_blocks = variant_tys
494                .iter()
495                .map(|_| helper.append_block(Block::new(&[])))
496                .collect::<Vec<_>>();
497
498            let case_values = (0..variant_tys.len())
499                .map(i64::try_from)
500                .collect::<std::result::Result<Vec<_>, TryFromIntError>>()?;
501
502            entry.append_operation(cf::switch(
503                context,
504                &case_values,
505                tag_val,
506                tag_ty,
507                (default_block, &[]),
508                &variant_blocks
509                    .iter()
510                    .copied()
511                    .map(|block| (block, [].as_slice()))
512                    .collect::<Vec<_>>(),
513                location,
514            )?);
515
516            // Default block.
517            {
518                let val = default_block.const_int(context, location, 0, 1)?;
519
520                default_block.append_operation(cf::assert(
521                    context,
522                    val,
523                    "Invalid enum tag.",
524                    location,
525                ));
526                default_block.append_operation(llvm::unreachable(location));
527            }
528
529            // Enum variants.
530            for (i, (block, (payload_ty, _))) in
531                variant_blocks.into_iter().zip(variant_tys).enumerate()
532            {
533                let enum_ty = llvm::r#type::r#struct(context, &[tag_ty, payload_ty], false);
534
535                let payload_val = match stack_ptr {
536                    Some(stack_ptr) => {
537                        let val = block.load(context, location, stack_ptr, enum_ty)?;
538                        block.extract_value(context, location, val, payload_ty, 1)?
539                    }
540                    None => {
541                        // If the enum is not memory-allocated it means that:
542                        //   - Either it's a C-style enum and all payloads have the same type.
543                        //   - Or the enum only has a single non-memory-allocated variant.
544                        if variant_ids.len() == 1 {
545                            entry.arg(0)?
546                        } else {
547                            native_assert!(
548                                registry.get_type(&variant_ids[i])?.is_zst(registry)?,
549                                "should be zero sized"
550                            );
551                            block.append_op_result(llvm::undef(payload_ty, location))?
552                        }
553                    }
554                };
555
556                helper.br(block, i, &[payload_val], location)?;
557            }
558        }
559    }
560
561    Ok(())
562}

564/// Generate MLIR operations for the `enum_boxed_match` libfunc.
565///
566/// Receives an `Enum` inside a `Box` and branches based on the variant,
567/// returning a `Box` containing the variant's payload.
568///
569/// # Signature
570///
571/// ```cairo
572/// enum MyEnum {
573///     A: felt252,
574///     B: u128,
575/// }
576///
577/// enum BoxedMyEnum {
578///     A: Box<felt252>,
579///     B: Box<u128>,
580/// }
581///
582/// extern fn enum_boxed_match<MyEnum>(
583///     value: Box<MyEnum>
584/// ) -> BoxedMyEnum nopanic;
585/// ```
586pub fn build_boxed_match<'ctx, 'this>(
587    context: &'ctx Context,
588    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
589    entry: &'this Block<'ctx>,
590    location: Location<'ctx>,
591    helper: &LibfuncHelper<'ctx, 'this>,
592    metadata: &mut MetadataStorage,
593    info: &EnumBoxedMatchConcreteLibfunc,
594) -> Result<()> {
595    metadata.get_or_insert_with(|| ReallocBindingsMeta::new(context, helper));
596
597    // Get the variant type IDs from the concrete libfunc info
598    let variant_ids = &info.variants;
599
600    match variant_ids.len() {
601        0 => {
602            // The Cairo compiler will generate an enum match for enums without variants, so this
603            // case cannot be a compile-time error. We're assuming that even though it's been
604            // generated, it's just dead code and can be made into an assertion that always fails.
605            let k0 = entry.const_int(context, location, 0, 1)?;
606            entry.append_operation(cf::assert(
607                context,
608                k0,
609                "attempt to match a zero-variant enum",
610                location,
611            ));
612            entry.append_operation(llvm::unreachable(location));
613        }
614        1 => {
615            // For single-variant enums, the enum type IS the payload type (no tag),
616            // so Box<Enum> is already identical to Box<Payload> — just forward the pointer.
617            helper.br(entry, 0, &[entry.arg(0)?], location)?;
618        }
619        _ => {
620            let (_layout, (tag_ty, tag_layout), variant_tys) =
621                crate::types::r#enum::get_type_for_variants(
622                    context,
623                    helper,
624                    registry,
625                    metadata,
626                    variant_ids,
627                )?;
628
629            // Tag is at offset 0 in the box, so load it directly from the pointer
630            let tag_val = entry.load(context, location, entry.arg(0)?, tag_ty)?;
631
632            let default_block = helper.append_block(Block::new(&[]));
633            let variant_blocks = variant_tys
634                .iter()
635                .map(|_| helper.append_block(Block::new(&[])))
636                .collect::<Vec<_>>();
637
638            let case_values = (0..variant_tys.len())
639                .map(i64::try_from)
640                .collect::<std::result::Result<Vec<_>, TryFromIntError>>()?;
641
642            entry.append_operation(cf::switch(
643                context,
644                &case_values,
645                tag_val,
646                tag_ty,
647                (default_block, &[]),
648                &variant_blocks
649                    .iter()
650                    .copied()
651                    .map(|block| (block, [].as_slice()))
652                    .collect::<Vec<_>>(),
653                location,
654            )?);
655
656            // Default block (invalid tag).
657            {
658                let val = default_block.const_int(context, location, 0, 1)?;
659
660                default_block.append_operation(cf::assert(
661                    context,
662                    val,
663                    "Invalid enum tag.",
664                    location,
665                ));
666                default_block.append_operation(llvm::unreachable(location));
667            }
668
669            // Enum variants.
670            for (i, (block, (payload_ty, payload_layout))) in
671                variant_blocks.into_iter().zip(variant_tys).enumerate()
672            {
673                // Extract the payload from the enum
674                let payload_val = {
675                    let payload_offset = tag_layout.extend(payload_layout)?.1;
676                    let ptr = block.gep(
677                        context,
678                        location,
679                        entry.arg(0)?,
680                        &[GepIndex::Const(payload_offset.try_into()?)],
681                        IntegerType::new(context, 8).into(),
682                    )?;
683                    block.load(context, location, ptr, payload_ty)?
684                };
685
686                // Free the input box
687                block.append_operation(ReallocBindingsMeta::free(
688                    context,
689                    entry.arg(0)?,
690                    location,
691                )?);
692
693                // Get the output variant type layout for boxing
694                let output_variant_type_id = &info.branch_signatures()[i].vars[0].ty;
695                let CoreTypeConcrete::Box(output_box_info) =
696                    registry.get_type(output_variant_type_id)?
697                else {
698                    native_panic!("Output should be a Box type");
699                };
700                let (_, output_layout) = registry.build_type_with_layout(
701                    context,
702                    helper,
703                    metadata,
704                    &output_box_info.ty,
705                )?;
706
707                // Box the payload
708                let boxed_payload = into_box(context, block, location, payload_val, output_layout)?;
709
710                helper.br(block, i, &[boxed_payload], location)?;
711            }
712        }
713    }
714
715    Ok(())
716}

934    #[test]
935    fn enum_boxed_match_different_padding() {
936        // Variants with different sizes/alignments exercise different payload
937        // offsets in the GEP-based extraction: u8 (1 byte), u128 (16 bytes),
938        // and (felt252, u128) (a 48-byte struct with 16-byte alignment).
939        let program = load_cairo! {
940            use core::box::BoxTrait;
941
942            enum Mixed {
943                Small: u8,
944                Medium: u128,
945                Large: (felt252, u128),
946            }
947
948            enum BoxedMixed {
949                Small: Box<u8>,
950                Medium: Box<u128>,
951                Large: Box<(felt252, u128)>,
952            }
953
954            extern fn enum_boxed_match<T>(e: Box<T>) -> BoxedMixed nopanic;
955
956            fn test_small() -> u8 {
957                let boxed = identity(BoxTrait::new(Mixed::Small(7_u8)));
958                match enum_boxed_match(boxed) {
959                    BoxedMixed::Small(v) => BoxTrait::unbox(v),
960                    BoxedMixed::Medium(_) => 0_u8,
961                    BoxedMixed::Large(_) => 0_u8,
962                }
963            }
964
965            fn test_medium() -> u128 {
966                let boxed = identity(BoxTrait::new(Mixed::Medium(999_u128)));
967                match enum_boxed_match(boxed) {
968                    BoxedMixed::Small(_) => 0_u128,
969                    BoxedMixed::Medium(v) => BoxTrait::unbox(v),
970                    BoxedMixed::Large(_) => 0_u128,
971                }
972            }
973
974            fn test_large() -> (felt252, u128) {
975                let boxed = identity(BoxTrait::new(Mixed::Large((42, 123_u128))));
976                match enum_boxed_match(boxed) {
977                    BoxedMixed::Small(_) => (0, 0_u128),
978                    BoxedMixed::Medium(_) => (0, 0_u128),
979                    BoxedMixed::Large(v) => BoxTrait::unbox(v),
980                }
981            }
982
983            #[inline(never)]

986                x
987            }
988        };
