27/// Select and call the correct libfunc builder function from the selector.
28pub fn build<'ctx, 'this>(
29    context: &'ctx Context,
30    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
31    entry: &'this Block<'ctx>,
32    location: Location<'ctx>,
33    helper: &LibfuncHelper<'ctx, 'this>,
34    metadata: &mut MetadataStorage,
35    selector: &IntRangeConcreteLibfunc,
36) -> Result<()> {
37    match selector {
38        IntRangeConcreteLibfunc::TryNew(info) => {
39            build_int_range_try_new(context, registry, entry, location, helper, metadata, info)
40        }
41        IntRangeConcreteLibfunc::PopFront(info) => {
42            build_int_range_pop_front(context, registry, entry, location, helper, metadata, info)
43        }
44    }
45}

47/// Generate MLIR operations for the `int_range_try_new` libfunc.
48pub fn build_int_range_try_new<'ctx, 'this>(
49    context: &'ctx Context,
50    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
51    entry: &'this Block<'ctx>,
52    location: Location<'ctx>,
53    helper: &LibfuncHelper<'ctx, 'this>,
54    metadata: &mut MetadataStorage,
55    info: &SignatureOnlyConcreteLibfunc,
56) -> Result<()> {
57    // The sierra-to-casm compiler uses the range check builtin a total of 1 time.
58    // https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/range.rs?plain=1#L24
59    let range_check = super::increment_builtin_counter(context, entry, location, entry.arg(0)?)?;
60    let x = entry.arg(1)?;
61    let y = entry.arg(2)?;
62    let range_ty = registry.build_type(
63        context,
64        helper,
65        metadata,
66        &info.branch_signatures()[0].vars[1].ty,
67    )?;
68    let inner = registry.get_type(&info.param_signatures()[1].ty)?;
69    // to know if it is signed
70    let inner_range = inner.integer_range(registry)?;
71
72    let is_valid = if inner_range.lower < BigInt::ZERO {
73        entry.cmpi(context, CmpiPredicate::Sle, x, y, location)?
74    } else {
75        entry.cmpi(context, CmpiPredicate::Ule, x, y, location)?
76    };
77
78    let range =
79        entry.append_op_result(ods::llvm::mlir_undef(context, range_ty, location).into())?;
80
81    // if the range is not valid, return the empty range [y, y)
82    let x_val = entry.append_op_result(arith::select(is_valid, x, y, location))?;
83    let range = entry.insert_values(context, location, range, &[x_val, y])?;
84
85    helper.cond_br(
86        context,
87        entry,
88        is_valid,
89        [0, 1],
90        [&[range_check, range], &[range_check, range]],
91        location,
92    )
93}

95/// Generate MLIR operations for the `int_range_pop_front` libfunc.
96pub fn build_int_range_pop_front<'ctx, 'this>(
97    context: &'ctx Context,
98    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
99    entry: &'this Block<'ctx>,
100    location: Location<'ctx>,
101    helper: &LibfuncHelper<'ctx, 'this>,
102    metadata: &mut MetadataStorage,
103    info: &SignatureOnlyConcreteLibfunc,
104) -> Result<()> {
105    let range = entry.arg(0)?;
106
107    let inner_ty = registry.build_type(
108        context,
109        helper,
110        metadata,
111        &info.branch_signatures()[1].vars[1].ty,
112    )?;
113
114    let inner = registry.get_type(&info.branch_signatures()[1].vars[1].ty)?;
115
116    let x = entry.extract_value(context, location, range, inner_ty, 0)?;
117    let k1 = entry.const_int_from_type(context, location, 1, inner_ty)?;
118    let x_p_1 = entry.addi(x, k1, location)?;
119    let y = entry.extract_value(context, location, range, inner_ty, 1)?;
120
121    // to know if it is signed
122    let inner_range = inner.integer_range(registry)?;
123
124    let is_valid = if inner_range.lower < BigInt::ZERO {
125        entry.cmpi(context, CmpiPredicate::Slt, x, y, location)?
126    } else {
127        entry.cmpi(context, CmpiPredicate::Ult, x, y, location)?
128    };
129    let range = entry.insert_value(context, location, range, x_p_1, 0)?;
130
131    helper.cond_br(
132        context,
133        entry,
134        is_valid,
135        [1, 0], // failure, success
136        [&[range, x], &[]],
137        location,
138    )
139}
