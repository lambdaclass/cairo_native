30/// Select and call the correct libfunc builder function from the selector.
31pub fn build<'ctx, 'this>(
32    context: &'ctx Context,
33    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
34    entry: &'this Block<'ctx>,
35    location: Location<'ctx>,
36    helper: &LibfuncHelper<'ctx, 'this>,
37    metadata: &mut MetadataStorage,
38    selector: &CastConcreteLibfunc,
39) -> Result<()> {
40    match selector {
41        CastConcreteLibfunc::Downcast(info) => {
42            build_downcast(context, registry, entry, location, helper, metadata, info)
43        }
44        CastConcreteLibfunc::Upcast(info) => {
45            build_upcast(context, registry, entry, location, helper, metadata, info)
46        }
47    }
48}

50/// Generate MLIR operations for the `downcast` libfunc which converts from a
51/// source type `T` to a target type `U`, where `U` might not fully include `T`.
52/// This means that the operation can fail.
53///
54/// ## Signature
55/// ```cairo
56/// pub extern const fn downcast<FromType, ToType>(
57///     x: FromType,
58/// ) -> Option<ToType> implicits(RangeCheck) nopanic;
59/// ```
60pub fn build_downcast<'ctx, 'this>(
61    context: &'ctx Context,
62    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
63    entry: &'this Block<'ctx>,
64    location: Location<'ctx>,
65    helper: &LibfuncHelper<'ctx, 'this>,
66    _metadata: &mut MetadataStorage,
67    info: &DowncastConcreteLibfunc,
68) -> Result<()> {
69    let range_check = entry.arg(0)?;
70    let src_value: Value = entry.arg(1)?;
71
72    let src_ty = registry.get_type(&info.signature.param_signatures[1].ty)?;
73    let dst_ty = registry.get_type(&info.signature.branch_signatures[0].vars[1].ty)?;
74
75    let dst_range = dst_ty.integer_range(registry)?;
76    let src_range = if src_ty.is_felt252(registry)? && dst_range.lower.sign() == Sign::Minus {
77        if dst_range.upper.sign() != Sign::Plus {
78            Range {
79                lower: BigInt::from_biguint(Sign::Minus, PRIME.clone()) + 1,
80                upper: BigInt::one(),
81            }
82        } else {
83            Range {
84                lower: BigInt::from_biguint(Sign::Minus, HALF_PRIME.clone()),
85                upper: BigInt::from_biguint(Sign::Plus, HALF_PRIME.clone()) + BigInt::one(),
86            }
87        }
88    } else {
89        src_ty.integer_range(registry)?
90    };
91
92    // When the source type is the same as the target type, we just return the
93    // value as it cannot fail. However, for backwards compatibility, we need to
94    // increment the range check as if we were checking the upper bound. See:
95    // - https://github.com/starkware-libs/cairo/tree/v2.12.3/crates/cairo-lang-sierra/src/extensions/modules/casts.rs#L67.
96    // - https://github.com/starkware-libs/cairo/tree/v2.12.3/crates/cairo-lang-sierra-to-casm/src/invocations/casts.rs#L56.
97    if info.signature.param_signatures[1].ty == info.signature.branch_signatures[0].vars[1].ty {
98        let range_check = if src_range.lower == 0.into() {
99            increment_builtin_counter(context, entry, location, range_check)?
100        } else {
101            range_check
102        };
103        let k1 = entry.const_int(context, location, 1, 1)?;
104        return helper.cond_br(
105            context,
106            entry,
107            k1,
108            [0, 1],
109            [&[range_check, src_value], &[range_check]],
110            location,
111        );
112    }
113
114    let src_width = if src_ty.is_bounded_int(registry)? {
115        src_range.offset_bit_width()
116    } else {
117        src_ty.integer_range(registry)?.zero_based_bit_width()
118    };
119    let dst_width = if dst_ty.is_bounded_int(registry)? {
120        dst_range.offset_bit_width()
121    } else {
122        dst_range.zero_based_bit_width()
123    };
124
125    let compute_width = src_range
126        .zero_based_bit_width()
127        .max(dst_range.zero_based_bit_width());
128
129    let is_signed = src_range.lower.sign() == Sign::Minus;
130
131    // If the target type is wider than the source type, extend the value representation width.
132    let src_value = if compute_width > src_width {
133        if is_signed && !src_ty.is_bounded_int(registry)? && !src_ty.is_felt252(registry)? {
134            entry.extsi(
135                src_value,
136                IntegerType::new(context, compute_width).into(),
137                location,
138            )?
139        } else {
140            entry.extui(
141                src_value,
142                IntegerType::new(context, compute_width).into(),
143                location,
144            )?
145        }
146    } else {
147        src_value
148    };
149
150    // Correct the value representation accordingly.
151    // 1. if it is a felt, then we need to convert the value from [0,P) to
152    //    [-P/2, P/2].
153    // 2. if it is a bounded_int, we need to offset the value to get the
154    //    actual value.
155    let src_value = if is_signed && src_ty.is_felt252(registry)? {
156        if src_range.upper.is_one() {
157            let adj_offset =
158                entry.const_int_from_type(context, location, PRIME.clone(), src_value.r#type())?;
159            entry.append_op_result(arith::subi(src_value, adj_offset, location))?
160        } else {
161            let adj_offset = entry.const_int_from_type(
162                context,
163                location,
164                HALF_PRIME.clone(),
165                src_value.r#type(),
166            )?;
167            let is_negative =
168                entry.cmpi(context, CmpiPredicate::Ugt, src_value, adj_offset, location)?;
169
170            let k_prime =
171                entry.const_int_from_type(context, location, PRIME.clone(), src_value.r#type())?;
172            let adj_value = entry.append_op_result(arith::subi(src_value, k_prime, location))?;
173
174            entry.append_op_result(arith::select(is_negative, adj_value, src_value, location))?
175        }
176    } else if src_ty.is_bounded_int(registry)? && src_range.lower != BigInt::ZERO {
177        let dst_offset = entry.const_int_from_type(
178            context,
179            location,
180            src_range.lower.clone(),
181            src_value.r#type(),
182        )?;
183        entry.addi(src_value, dst_offset, location)?
184    } else {
185        src_value
186    };
187
188    // Check if the source type is included in the target type. If it is not
189    // then check if the value is in bounds. If the value is also not in
190    // bounds then return an error.
191    if dst_range.lower <= src_range.lower && dst_range.upper >= src_range.upper {
192        let dst_value = if dst_ty.is_bounded_int(registry)? && dst_range.lower != BigInt::ZERO {
193            let dst_offset = entry.const_int_from_type(
194                context,
195                location,
196                dst_range.lower,
197                src_value.r#type(),
198            )?;
199            entry.append_op_result(arith::subi(src_value, dst_offset, location))?
200        } else {
201            src_value
202        };
203
204        let dst_value = if dst_width < compute_width {
205            entry.trunci(
206                dst_value,
207                IntegerType::new(context, dst_width).into(),
208                location,
209            )?
210        } else {
211            dst_value
212        };
213
214        let is_in_bounds = entry.const_int(context, location, 1, 1)?;
215
216        helper.cond_br(
217            context,
218            entry,
219            is_in_bounds,
220            [0, 1],
221            [&[range_check, dst_value], &[range_check]],
222            location,
223        )?;
224    } else {
225        // Check if the value is in bounds with respect to the lower bound.
226        let lower_check = if dst_range.lower > src_range.lower {
227            let dst_lower = entry.const_int_from_type(
228                context,
229                location,
230                dst_range.lower.clone(),
231                src_value.r#type(),
232            )?;
233            Some(entry.cmpi(
234                context,
235                if !is_signed {
236                    CmpiPredicate::Uge
237                } else {
238                    CmpiPredicate::Sge
239                },
240                src_value,
241                dst_lower,
242                location,
243            )?)
244        } else {
245            None
246        };
247        // Check if the value is in bounds with respect to the upper bound.
248        let upper_check = if dst_range.upper < src_range.upper {
249            let dst_upper = entry.const_int_from_type(
250                context,
251                location,
252                dst_range.upper.clone(),
253                src_value.r#type(),
254            )?;
255            Some(entry.cmpi(
256                context,
257                if !is_signed {
258                    CmpiPredicate::Ult
259                } else {
260                    CmpiPredicate::Slt
261                },
262                src_value,
263                dst_upper,
264                location,
265            )?)
266        } else {
267            None
268        };
269
270        let is_in_bounds = match (lower_check, upper_check) {
271            (Some(lower_check), Some(upper_check)) => {
272                entry.append_op_result(arith::andi(lower_check, upper_check, location))?
273            }
274            (Some(lower_check), None) => lower_check,
275            (None, Some(upper_check)) => upper_check,
276            // its always in bounds since dst is larger than src (i.e no bounds checks needed)
277            (None, None) => {
278                native_panic!("matched an unreachable: no bounds checks are being performed")
279            }
280        };
281
282        // Incrementing the range check depends on whether the source range can hold a felt252 or not
283        let range_check = if info.from_range.is_full_felt252_range() {
284            let rc_size = BigInt::from(1) << 128;
285            // If the range can contain a felt252, how the range check is increased depends on whether the value is in bounds or not:
286            // * If it is in bounds, we check whether the destination range size is less than range check size. If it is, increment
287            //   the range check builtin by 2. Otherwise, increment it by 1.
288            //   https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/range_reduction.rs#L87
289            // * If it is not in bounds, increment the range check builtin by 3.
290            //   https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/range_reduction.rs#L79
291            super::increment_builtin_counter_conditionally_by(
292                context,
293                entry,
294                location,
295                range_check,
296                if dst_range.size() < rc_size { 2 } else { 1 },
297                3,
298                is_in_bounds,
299            )?
300        } else {
301            match (lower_check, upper_check) {
302                (Some(_), None) | (None, Some(_)) => {
303                    // If either the lower or the upper bound was checked, increment the range check builtin by 1.
304                    // * In case the lower bound was checked: https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/casts.rs#L135
305                    // * In case the upper bound was checked: https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/casts.rs#L111
306                    super::increment_builtin_counter_by(context, entry, location, range_check, 1)?
307                }
308                (Some(lower_check), Some(upper_check)) => {
309                    let is_in_range =
310                        entry.append_op_result(arith::andi(lower_check, upper_check, location))?;
311
312                    // If the result is in range, increment the range check builtin by 2. Otherwise, increment it by 1.
313                    // https://github.com/starkware-libs/cairo/blob/v2.12.0-dev.1/crates/cairo-lang-sierra-to-casm/src/invocations/casts.rs#L160
314                    super::increment_builtin_counter_conditionally_by(
315                        context,
316                        entry,
317                        location,
318                        range_check,
319                        2,
320                        1,
321                        is_in_range,
322                    )?
323                }
324                (None, None) => range_check,
325            }
326        };
327
328        let dst_value = if dst_ty.is_bounded_int(registry)? && dst_range.lower != BigInt::ZERO {
329            let dst_offset = entry.const_int_from_type(
330                context,
331                location,
332                dst_range.lower,
333                src_value.r#type(),
334            )?;
335            entry.append_op_result(arith::subi(src_value, dst_offset, location))?
336        } else {
337            src_value
338        };
339
340        let dst_value = if dst_width < compute_width {
341            entry.trunci(
342                dst_value,
343                IntegerType::new(context, dst_width).into(),
344                location,
345            )?
346        } else {
347            dst_value
348        };
349
350        helper.cond_br(
351            context,
352            entry,
353            is_in_bounds,
354            [0, 1],
355            [&[range_check, dst_value], &[range_check]],
356            location,
357        )?;
358    }
359
360    Ok(())
361}

363/// Builds the `upcast` libfunc, which converts from a source type `T` to a
364/// target type `U`, where `U` fully includes `T`. This means that the operation
365/// cannot fail.
366///
367/// ## Signature
368///
369/// ```cairo
370/// extern const fn upcast<FromType, ToType>(x: FromType) -> ToType nopanic;
371/// ```
372pub fn build_upcast<'ctx, 'this>(
373    context: &'ctx Context,
374    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
375    entry: &'this Block<'ctx>,
376    location: Location<'ctx>,
377    helper: &LibfuncHelper<'ctx, 'this>,
378    _metadata: &mut MetadataStorage,
379    info: &SignatureOnlyConcreteLibfunc,
380) -> Result<()> {
381    let src_value = entry.arg(0)?;
382
383    if info.signature.param_signatures[0].ty == info.signature.branch_signatures[0].vars[0].ty {
384        return helper.br(entry, 0, &[src_value], location);
385    }
386
387    let src_ty = registry.get_type(&info.signature.param_signatures[0].ty)?;
388    let dst_ty = registry.get_type(&info.signature.branch_signatures[0].vars[0].ty)?;
389
390    let src_range = src_ty.integer_range(registry)?;
391    let dst_range = dst_ty.integer_range(registry)?;
392
393    // An upcast is infallible, so the target type should always contain the source type.
394    {
395        let dst_contains_src =
396            dst_range.lower <= src_range.lower && dst_range.upper >= src_range.upper;
397
398        // If the target type is a felt, then both [0; P) and [-P/2, P/2] ranges are valid.
399        let dst_contains_src = if dst_ty.is_felt252(registry)? {
400            let signed_dst_range = Range {
401                lower: BigInt::from_biguint(Sign::Minus, HALF_PRIME.clone()),
402                upper: BigInt::from_biguint(Sign::Plus, HALF_PRIME.clone()) + BigInt::one(),
403            };
404            let signed_dst_contains_src = signed_dst_range.lower <= src_range.lower
405                && signed_dst_range.upper >= src_range.upper;
406            dst_contains_src || signed_dst_contains_src
407        } else {
408            dst_contains_src
409        };
410
411        native_assert!(
412            dst_contains_src,
413            "cannot upcast `{:?}` into `{:?}`: target range doesn't contain source range",
414            info.signature.param_signatures[0].ty,
415            info.signature.branch_signatures[0].vars[0].ty
416        );
417    }
418
419    let src_width = if src_ty.is_bounded_int(registry)? {
420        src_range.offset_bit_width()
421    } else {
422        src_range.zero_based_bit_width()
423    };
424    let dst_width = if dst_ty.is_bounded_int(registry)? {
425        dst_range.offset_bit_width()
426    } else {
427        dst_range.zero_based_bit_width()
428    };
429
430    // Extend value to target bit width.
431    let dst_value = if dst_width > src_width {
432        if src_ty.is_bounded_int(registry)? {
433            // A bounded int is always represented as a positive integer,
434            // because we store the offset to the lower bound.
435            entry.extui(
436                src_value,
437                IntegerType::new(context, dst_width).into(),
438                location,
439            )?
440        } else if src_range.lower.sign() == Sign::Minus {
441            entry.extsi(
442                src_value,
443                IntegerType::new(context, dst_width).into(),
444                location,
445            )?
446        } else {
447            entry.extui(
448                src_value,
449                IntegerType::new(context, dst_width).into(),
450                location,
451            )?
452        }
453    } else {
454        src_value
455    };
456
457    // When converting to/from bounded ints, we need to take into account the offset.
458    let offset = if src_ty.is_bounded_int(registry)? && dst_ty.is_bounded_int(registry)? {
459        &src_range.lower - &dst_range.lower
460    } else if src_ty.is_bounded_int(registry)? {
461        src_range.lower.clone()
462    } else if dst_ty.is_bounded_int(registry)? {
463        -dst_range.lower
464    } else {
465        BigInt::ZERO
466    };
467    let offset_value = entry.const_int_from_type(context, location, offset, dst_value.r#type())?;
468    let dst_value = entry.addi(dst_value, offset_value, location)?;
469
470    // When converting to a felt from a signed integer, we need to convert
471    // the canonical signed integer representation, to the signed felt
472    // representation: `negative = P - absolute`.
473    let dst_value = if dst_ty.is_felt252(registry)? && src_range.lower.sign() == Sign::Minus {
474        let k0 = entry.const_int(context, location, 0, 252)?;
475        let is_negative = entry.cmpi(context, CmpiPredicate::Slt, dst_value, k0, location)?;
476
477        let k_prime = entry.const_int(context, location, PRIME.clone(), 252)?;
478        let adj_value = entry.addi(dst_value, k_prime, location)?;
479
480        entry.append_op_result(arith::select(is_negative, adj_value, dst_value, location))?
481    } else {
482        dst_value
483    };
484
485    helper.br(entry, 0, &[dst_value], location)
486}
