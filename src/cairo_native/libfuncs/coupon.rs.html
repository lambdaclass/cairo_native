27/// Generate MLIR operations for the `coupon` libfuncs.
28/// In native it's mostly a no-op operation.
29pub fn build<'ctx, 'this>(
30    context: &'ctx Context,
31    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
32    entry: &'this Block<'ctx>,
33    location: Location<'ctx>,
34    helper: &LibfuncHelper<'ctx, 'this>,
35    metadata: &mut MetadataStorage,
36    selector: &CouponConcreteLibfunc,
37) -> Result<()> {
38    match selector {
39        CouponConcreteLibfunc::Buy(info) => {
40            // Libfunc for buying a coupon for a function. The cost of the coupon is the cost of running the
41            // function (not including the `call` and `ret` instructions).
42            // The coupon can be used to pay in advance for running the function, and run it later for
43            // free (paying only for the `call` and `ret` instructions) using `coupon_call`.
44            build_buy(context, registry, entry, location, helper, metadata, info)
45        }
46        CouponConcreteLibfunc::Refund(info) => {
47            // Libfunc for getting a refund for an unused coupon. The refund is the cost of the function
48            // and it is added back to the gas wallet.
49            build_refund(context, registry, entry, location, helper, metadata, info)
50        }
51    }
52}

54/// Generate MLIR operations for the `coupon` libfunc.
55pub fn build_buy<'ctx, 'this>(
56    context: &'ctx Context,
57    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
58    entry: &'this Block<'ctx>,
59    location: Location<'ctx>,
60    helper: &LibfuncHelper<'ctx, 'this>,
61    metadata: &mut MetadataStorage,
62    info: &SignatureAndFunctionConcreteLibfunc,
63) -> Result<()> {
64    // In the future if the gas cost is required, this is how to get it.
65    // let gas = metadata.get::<GasMetadata>().ok_or(Error::MissingMetadata)?;
66    // let gas_cost = gas.initial_required_gas(&info.function.id);
67    let ty = registry.build_type(
68        context,
69        helper,
70        metadata,
71        &info.branch_signatures()[0].vars[0].ty,
72    )?;
73    let coupon = entry.append_op_result(llvm::undef(ty, location))?;
74
75    entry.append_operation(helper.br(0, &[coupon], location));
76
77    Ok(())
78}
