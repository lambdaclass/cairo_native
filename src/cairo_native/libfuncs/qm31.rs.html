28pub fn build<'ctx, 'this>(
29    context: &'ctx Context,
30    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
31    entry: &'this Block<'ctx>,
32    location: Location<'ctx>,
33    helper: &LibfuncHelper<'ctx, 'this>,
34    metadata: &mut MetadataStorage,
35    selector: &QM31Concrete,
36) -> Result<()> {
37    match selector {
38        QM31Concrete::BinaryOperation(info) => {
39            build_binary_op(context, registry, entry, location, helper, metadata, info)
40        }
41        QM31Concrete::Const(info) => {
42            build_const(context, registry, entry, location, helper, metadata, info)
43        }
44        QM31Concrete::IsZero(info) => {
45            build_is_zero(context, registry, entry, location, helper, metadata, info)
46        }
47        QM31Concrete::Pack(info) => {
48            build_pack(context, registry, entry, location, helper, metadata, info)
49        }
50        QM31Concrete::Unpack(info) => {
51            build_unpack(context, registry, entry, location, helper, metadata, info)
52        }
53        QM31Concrete::FromM31(info) => {
54            build_from_m31(context, registry, entry, location, helper, metadata, info)
55        }
56    }
57}

59/// Generate MLIR operations for the `qm31_const` libfunc.
60///
61/// Receives 4 const m31 and returns a qm31.
62///
63/// # Constraints
64///
65/// m31 are always between 0 and 2**31 - 2 (inclusive)
66///
67/// # Cairo Signature
68///
69/// ```cairo
70/// fn qm31_const<
71///     const W0: m31, const W1: m31, const W2: m31, const W3: m31,
72/// >() -> qm31 nopanic;
73/// ```
74pub fn build_const<'ctx, 'this>(
75    context: &'ctx Context,
76    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
77    entry: &'this Block<'ctx>,
78    location: Location<'ctx>,
79    helper: &LibfuncHelper<'ctx, 'this>,
80    metadata: &mut MetadataStorage,
81    info: &QM31ConstConcreteLibfunc,
82) -> Result<()> {
83    let m31_ty = IntegerType::new(context, 31).into();
84    let qm31_ty = registry.build_type(
85        context,
86        helper,
87        metadata,
88        &info.branch_signatures()[0].vars[0].ty,
89    )?;
90
91    let m31_0 = entry.const_int_from_type(context, location, info.w0, m31_ty)?;
92    let m31_1 = entry.const_int_from_type(context, location, info.w1, m31_ty)?;
93    let m31_2 = entry.const_int_from_type(context, location, info.w2, m31_ty)?;
94    let m31_3 = entry.const_int_from_type(context, location, info.w3, m31_ty)?;
95
96    let qm31 = entry.append_op_result(llvm::undef(qm31_ty, location))?;
97    let qm31 = entry.insert_values(context, location, qm31, &[m31_0, m31_1, m31_2, m31_3])?;
98
99    helper.br(entry, 0, &[qm31], location)
100}

102/// Generate MLIR operations for the `qm31_is_zero` libfunc.
103///
104/// Receives a qm31 and returns a Some(qm31) if the argument is not 0,
105/// otherwise a None.
106///
107/// # Cairo Signature
108///
109/// ```cairo
110/// fn qm31_is_zero(a: qm31) -> core::internal::OptionRev<NonZero<qm31>> nopanic;
111/// ```
112pub fn build_is_zero<'ctx, 'this>(
113    context: &'ctx Context,
114    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
115    entry: &'this Block<'ctx>,
116    location: Location<'ctx>,
117    helper: &LibfuncHelper<'ctx, 'this>,
118    _metadata: &mut MetadataStorage,
119    _info: &SignatureOnlyConcreteLibfunc,
120) -> Result<()> {
121    let qm31 = entry.arg(0)?;
122    let m31_ty = IntegerType::new(context, 31).into();
123
124    let m31_0 = entry.extract_value(context, location, qm31, m31_ty, 0)?;
125    let m31_1 = entry.extract_value(context, location, qm31, m31_ty, 1)?;
126    let m31_2 = entry.extract_value(context, location, qm31, m31_ty, 2)?;
127    let m31_3 = entry.extract_value(context, location, qm31, m31_ty, 3)?;
128
129    // Check that every limb is equal to 0:
130    //      (m31_0 | m31_1 | m31_2 | m31_3) == 0
131    let cond = entry.append_op_result(arith::ori(m31_0, m31_1, location))?;
132    let cond = entry.append_op_result(arith::ori(cond, m31_2, location))?;
133    let cond = entry.append_op_result(arith::ori(cond, m31_3, location))?;
134    let k0 = entry.const_int_from_type(context, location, 0, m31_ty)?;
135    let cond =
136        entry.append_op_result(arith::cmpi(context, CmpiPredicate::Eq, k0, cond, location))?;
137
138    helper.cond_br(
139        context,
140        entry,
141        cond,
142        [0, 1],
143        [&[], &[entry.arg(0)?]],
144        location,
145    )
146}

202pub fn build_unpack<'ctx, 'this>(
203    context: &'ctx Context,

214    let qm31 = entry.arg(1)?;
215
216    let m31_0 = entry.extract_value(context, location, qm31, m31_ty.into(), 0)?;

219    let m31_3 = entry.extract_value(context, location, qm31, m31_ty.into(), 3)?;
220
221    helper.br(
222        entry,
223        0,
224        &[range_check, m31_0, m31_1, m31_2, m31_3],

269fn m31_add<'ctx, 'this>(
270    context: &'ctx Context,
271    entry: &'this Block<'ctx>,
272    location: Location<'ctx>,
273    helper: &LibfuncHelper<'ctx, 'this>,
274) -> Result<()> {
275    let lhs_value = entry.arg(0)?;
276    let rhs_value = entry.arg(1)?;
277
278    let lhs_value = entry.extui(lhs_value, IntegerType::new(context, 32).into(), location)?;
279    let rhs_value = entry.extui(rhs_value, IntegerType::new(context, 32).into(), location)?;
280
281    let res = entry.append_op_result(arith::addi(lhs_value, rhs_value, location))?;
282    let prime = entry.const_int(context, location, M31_PRIME, 32)?;
283    let res_mod = entry.append_op_result(arith::subi(res, prime, location))?;
284    let is_out_of_range = entry.cmpi(context, CmpiPredicate::Uge, res, prime, location)?;
285
286    let res = entry.append_op_result(arith::select(is_out_of_range, res_mod, res, location))?;
287
288    let res = entry.trunci(res, IntegerType::new(context, 31).into(), location)?;
289
290    helper.br(entry, 0, &[res], location)
291}

309    helper.br(entry, 0, &[res], location)

312fn m31_mul<'ctx, 'this>(
313    context: &'ctx Context,
314    entry: &'this Block<'ctx>,

316    helper: &LibfuncHelper<'ctx, 'this>,

332    helper.br(entry, 0, &[res], location)

335fn m31_div<'ctx, 'this>(
336    context: &'ctx Context,
337    entry: &'this Block<'ctx>,
338    location: Location<'ctx>,
339    helper: &LibfuncHelper<'ctx, 'this>,
340    metadata: &mut MetadataStorage,
341) -> Result<()> {
342    let i31 = IntegerType::new(context, 31).into();
343    let i64 = IntegerType::new(context, 64).into();
344    let rhs_value = entry.arg(1)?;
345    let prime = entry.const_int_from_type(context, location, M31_PRIME, i31)?;
346
347    let runtime_bindings_meta = metadata
348        .get_mut::<RuntimeBindingsMeta>()
349        .to_native_assert_error("Unable to get the RuntimeBindingsMeta from MetadataStorage")?;
350    let euclidean_result = runtime_bindings_meta.u31_extended_euclidean_algorithm(
351        context,
352        helper.module,
353        entry,
354        location,
355        rhs_value,
356        prime,
357    )?;
358
359    // Here we omit checking if inverse is actually the inverse,
360    // satisfying gcd(a,b) == 1, because we are using a prime as the
361    // modulus. This ensures that for any value of a, included in the
362    // field, gcd(a,b) == 1.
363    let prime = entry.const_int_from_type(context, location, M31_PRIME, i64)?;
364    let inverse = {
365        let inverse = entry.extract_value(context, location, euclidean_result, i31, 1)?;
366        entry.extui(inverse, i64, location)?
367    };
368
369    // Perform lhs * (1 / rhs)
370    let lhs = entry.extui(entry.arg(0)?, i64, location)?;
371    let result = entry.muli(lhs, inverse, location)?;
372    // Apply modulo and convert result to m31
373    let result_mod = entry.append_op_result(arith::remui(result, prime, location))?;
374    let is_out_of_range = entry.cmpi(context, CmpiPredicate::Uge, result, prime, location)?;
375
376    let result =
377        entry.append_op_result(arith::select(is_out_of_range, result_mod, result, location))?;
378    let result = entry.trunci(result, i31, location)?;
379
380    helper.br(entry, 0, &[result], location)
381}

383/// Generate MLIR operations for the QM31 and M31 binary operations libfuncs.
384///
385/// Depending on the type of the parameters, it chooses which type of representation
386/// it will manage (QM31 or M31). It either receives 2 qm31 or 2 m31 (which are represented
387/// as bounded ints)
388///
389/// # Cairo Signature
390/// ```cairo
391/// // qm31
392/// fn qm31_add(a: qm31, b: qm31) -> qm31 nopanic;
393/// fn qm31_sub(a: qm31, b: qm31) -> qm31 nopanic;
394/// fn qm31_mul(a: qm31, b: qm31) -> qm31 nopanic;
395/// fn qm31_div(a: qm31, b: NonZero<qm31>) -> qm31 nopanic;
396///
397/// // m31
398/// extern fn m31_add(a: m31, b: m31) -> m31 nopanic;
399/// extern fn m31_sub(a: m31, b: m31) -> m31 nopanic;
400/// extern fn m31_mul(a: m31, b: m31) -> m31 nopanic;
401/// extern fn m31_div(a: m31, b: NonZero<m31>) -> m31 nopanic;
402/// ```
403pub fn build_binary_op<'ctx, 'this>(
404    context: &'ctx Context,
405    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
406    entry: &'this Block<'ctx>,
407    location: Location<'ctx>,
408    helper: &LibfuncHelper<'ctx, 'this>,
409    metadata: &mut MetadataStorage,
410    info: &QM31BinaryOpConcreteLibfunc,
411) -> Result<()> {
412    // If the parameter is a bounded int, then we need to generate the operations
413    // for the m31
414    let type_concrete = registry.get_type(&info.param_signatures()[0].ty)?;
415    if let CoreTypeConcrete::BoundedInt(_) = type_concrete {
416        match info.operator {
417            cairo_lang_sierra::extensions::qm31::QM31BinaryOperator::Add => {
418                return m31_add(context, entry, location, helper);
419            }
420            cairo_lang_sierra::extensions::qm31::QM31BinaryOperator::Sub => {
421                return m31_sub(context, entry, location, helper);
422            }
423            cairo_lang_sierra::extensions::qm31::QM31BinaryOperator::Mul => {
424                return m31_mul(context, entry, location, helper);
425            }
426            cairo_lang_sierra::extensions::qm31::QM31BinaryOperator::Div => {
427                return m31_div(context, entry, location, helper, metadata);
428            }
429        }
430    }
431    let qm31_ty = registry.build_type(context, helper, metadata, &info.param_signatures()[0].ty)?;
432
433    let lhs = entry.arg(0)?;
434    let rhs = entry.arg(1)?;
435
436    let lhs_ptr = entry.alloca1(context, location, qm31_ty, get_integer_layout(31).align())?;
437    let rhs_ptr = entry.alloca1(context, location, qm31_ty, get_integer_layout(31).align())?;
438
439    entry.store(context, location, lhs_ptr, lhs)?;
440    entry.store(context, location, rhs_ptr, rhs)?;
441
442    let result = metadata
443        .get_mut::<RuntimeBindingsMeta>()
444        .ok_or(Error::MissingMetadata)?
445        .libfunc_qm31_bin_op(
446            context,
447            helper,
448            entry,
449            lhs_ptr,
450            rhs_ptr,
451            info.operator,
452            location,
453        )?;
454
455    helper.br(entry, 0, &[result], location)
456}
