26/// Select and call the correct libfunc builder function from the selector.
27pub fn build<'ctx, 'this>(
28    context: &'ctx Context,
29    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
30    entry: &'this Block<'ctx>,
31    location: Location<'ctx>,
32    helper: &LibfuncHelper<'ctx, 'this>,
33    metadata: &mut MetadataStorage,
34    selector: &StructConcreteLibfunc,
35) -> Result<()> {
36    match selector {
37        StructConcreteLibfunc::Construct(info) => {
38            build_construct(context, registry, entry, location, helper, metadata, info)
39        }
40        StructConcreteLibfunc::Deconstruct(info)
41        | StructConcreteLibfunc::SnapshotDeconstruct(info) => {
42            build_deconstruct(context, registry, entry, location, helper, metadata, info)
43        }
44    }
45}

47/// Generate MLIR operations for the `struct_construct` libfunc.
48pub fn build_construct<'ctx, 'this>(
49    context: &'ctx Context,
50    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
51    entry: &'this Block<'ctx>,
52    location: Location<'ctx>,
53    helper: &LibfuncHelper<'ctx, 'this>,
54    metadata: &mut MetadataStorage,
55    info: &SignatureOnlyConcreteLibfunc,
56) -> Result<()> {
57    let mut fields = Vec::new();
58
59    for (i, _) in info.param_signatures().iter().enumerate() {
60        fields.push(entry.argument(i)?.into());
61    }
62
63    let value = build_struct_value(
64        context,
65        registry,
66        entry,
67        location,
68        helper,
69        metadata,
70        &info.branch_signatures()[0].vars[0].ty,
71        &fields,
72    )?;
73
74    helper.br(entry, 0, &[value], location)
75}

77/// Generate MLIR operations for the `struct_construct` libfunc.
78#[allow(clippy::too_many_arguments)]
79pub fn build_struct_value<'ctx, 'this>(
80    context: &'ctx Context,
81    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
82    entry: &'this Block<'ctx>,
83    location: Location<'ctx>,
84    helper: &LibfuncHelper<'ctx, 'this>,
85    metadata: &mut MetadataStorage,
86    struct_type: &ConcreteTypeId,
87    fields: &[Value<'ctx, 'this>],
88) -> Result<Value<'ctx, 'this>> {
89    let struct_ty = registry.build_type(context, helper, metadata, struct_type)?;
90
91    let acc = entry.append_operation(llvm::undef(struct_ty, location));
92
93    entry.insert_values(context, location, acc.result(0)?.into(), fields)
94}

96/// Generate MLIR operations for the `struct_deconstruct` libfunc.
97pub fn build_deconstruct<'ctx, 'this>(
98    context: &'ctx Context,
99    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
100    entry: &'this Block<'ctx>,
101    location: Location<'ctx>,
102    helper: &LibfuncHelper<'ctx, 'this>,
103    metadata: &mut MetadataStorage,
104    info: &SignatureOnlyConcreteLibfunc,
105) -> Result<()> {
106    let container = entry.arg(0)?;
107
108    let mut fields = Vec::<Value>::with_capacity(info.branch_signatures()[0].vars.len());
109    for (i, var_info) in info.branch_signatures()[0].vars.iter().enumerate() {
110        let type_info = registry.get_type(&var_info.ty)?;
111        let field_ty = type_info.build(context, helper, registry, metadata, &var_info.ty)?;
112
113        let value = entry.extract_value(context, location, container, field_ty, i)?;
114
115        fields.push(value);
116    }
117
118    helper.br(entry, 0, &fields, location)
119}
