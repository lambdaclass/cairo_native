28pub fn build<'ctx, 'this>(
29    context: &'ctx Context,
30    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
31    entry: &'this Block<'ctx>,
32    location: Location<'ctx>,
33    helper: &LibfuncHelper<'ctx, 'this>,
34    metadata: &mut MetadataStorage,
35    info: &CheatcodeConcreteLibfunc,
36) -> Result<()> {
37    // Calculate the result layout and type, based on the branch signature
38    let (result_type, result_layout) = registry.build_type_with_layout(
39        context,
40        helper,
41        metadata,
42        &info.branch_signatures()[0].vars[0].ty,
43    )?;
44
45    // Allocate the result pointer with calculated layout and type
46    let result_ptr = helper
47        .init_block()
48        .append_operation(alloca(
49            context,
50            helper.init_block().const_int(context, location, 1, 64)?,
51            llvm::r#type::pointer(context, 0),
52            location,
53            AllocaOptions::new()
54                .align(Some(IntegerAttribute::new(
55                    IntegerType::new(context, 64).into(),
56                    result_layout.align().try_into()?,
57                )))
58                .elem_type(Some(TypeAttribute::new(result_type))),
59        ))
60        .result(0)?
61        .into();
62
63    // Allocate and store selector. The type contains 256 bits as its interpreted as a [u8;32] from the runtime
64    let selector = helper
65        .init_block()
66        .const_int(context, location, info.selector.clone(), 256)?;
67    let selector_ptr =
68        helper
69            .init_block()
70            .alloca_int(context, location, 256, get_integer_layout(256).align())?;
71
72    helper
73        .init_block()
74        .store(context, location, selector_ptr, selector)?;
75
76    // Allocate and store arguments. The cairo type is a Span<Felt252> (the outer struct),
77    // which contains an Array<Felt252> (the inner struct)
78    let span_felt252_type = llvm::r#type::r#struct(
79        context,
80        &[llvm::r#type::r#struct(
81            context,
82            &[
83                llvm::r#type::pointer(context, 0),
84                IntegerType::new(context, 32).into(),
85                IntegerType::new(context, 32).into(),
86                IntegerType::new(context, 32).into(),
87            ],
88            false,
89        )],
90        false,
91    );
92    let args_ptr = helper.init_block().alloca1(
93        context,
94        location,
95        span_felt252_type,
96        get_integer_layout(64).align(),
97    )?;
98    entry.store(context, location, args_ptr, entry.argument(0)?.into())?;
99
100    // Call runtime cheatcode syscall wrapper
101    metadata
102        .get_mut::<RuntimeBindingsMeta>()
103        .expect("Runtime library not available.")
104        .vtable_cheatcode(
105            context,
106            helper,
107            entry,
108            location,
109            result_ptr,
110            selector_ptr,
111            args_ptr,
112        )?;
113
114    // Load result from result ptr and branch
115    let result = entry.append_op_result(llvm::load(
116        context,
117        result_ptr,
118        result_type,
119        location,
120        LoadStoreOptions::new(),
121    ))?;
122    helper.br(entry, 0, &[result], location)
123}
