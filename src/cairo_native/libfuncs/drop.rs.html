1//! # `AP` tracking libfuncs
2//!
3//! Most types are trivial and don't need dropping (or rather, they will be dropped automatically
4//! by MLIR). For those types, this libfunc is a no-op.
5//!
6//! However, types like an array need manual dropping.
7
8use super::LibfuncHelper;
9use crate::{
10    error::Result,
11    metadata::{drop_overrides::DropOverridesMeta, MetadataStorage},
12    utils::ProgramRegistryExt,
13};
14use cairo_lang_sierra::{
15    extensions::{
16        core::{CoreLibfunc, CoreType},
17        lib_func::SignatureOnlyConcreteLibfunc,
18    },
19    program_registry::ProgramRegistry,
20};
21use melior::{
22    helpers::BuiltinBlockExt,
23    ir::{Block, Location},
24    Context,
25};
26
27/// Generate MLIR operations for the `drop` libfunc.
28pub fn build<'ctx, 'this>(
29    context: &'ctx Context,
30    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
31    entry: &'this Block<'ctx>,
32    location: Location<'ctx>,
33    helper: &LibfuncHelper<'ctx, 'this>,
34    metadata: &mut MetadataStorage,
35    info: &SignatureOnlyConcreteLibfunc,
36) -> Result<()> {
37    registry.build_type(
38        context,
39        helper,
40        metadata,
41        &info.signature.param_signatures[0].ty,
42    )?;
43
44    if let Some(drop_overrides_meta) = metadata.get::<DropOverridesMeta>() {
45        drop_overrides_meta.invoke_override(
46            context,
47            entry,
48            location,
49            &info.signature.param_signatures[0].ty,
50            entry.arg(0)?,
51        )?;
52    }
53
54    helper.br(entry, 0, &[], location)
55}
