1//! # Dropping logic overrides
2//!
3//! By default, values are discarded, but some cases (like arrays, boxes, nullables, dictionaries
4//! and some structs and enums) need a drop implementation instad. This metadata is a register of
5//! types that require a drop implementation as well as the logic to register and invoke those
6//! implementations.
7//!
8//! ## Drop implementations
9//!
10//! The drop logic is implemented as a function for each type that requires it. It has to be a
11//! function to allow self-referencing types. If we inlined the drop implementations,
12//! self-referencing types would generate infinite code thus overflowing the stack when generating
13//! code.
14//!
15//! The generated functions are not public (they are internal) and follow this naming convention:
16//!
17//! ```text
18//! drop${type id}
19//! ```
20//!
21//! where `{type id}` is the numeric value of the `ConcreteTypeId`.
22

44use std::collections::HashSet;

46#[derive(Debug, Default)]
47pub struct DropOverridesMeta {
48    overriden_types: HashSet<ConcreteTypeId>,
49}
50
51impl DropOverridesMeta {
52    /// Register a drop override using a closure.
53    ///
54    /// This function does several things:
55    ///   - Registers `DropOverrideMeta` if it wasn't already present.
56    ///   - If the type id was already registered it returns and does nothing.
57    ///   - Registers the type (without it being actually registered yet).
58    ///   - Calls the closure, which returns an `Option<Region>`.
59    ///   - If the closure returns a region, generates the function implementation.
60    ///   - If the closure returns `None`, it removes the registry entry for the type.
61    ///
62    /// The type need to be registered before calling the closure, otherwise self-referencing types
63    /// would cause stack overflow when registering themselves.
64    ///
65    /// The callback serves two purposes:
66    ///   - To generate the drop implementation, if necessary.
67    ///   - To check if we need to generate the implementation (for example, in structs and enums).
68    pub(crate) fn register_with<'ctx>(
69        context: &'ctx Context,
70        module: &Module<'ctx>,
71        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
72        metadata: &mut MetadataStorage,
73        id: &ConcreteTypeId,
74        f: impl FnOnce(&mut MetadataStorage) -> Result<Option<Region<'ctx>>>,
75    ) -> Result<()> {
76        {
77            let drop_override_meta = metadata.get_or_insert_with(Self::default);
78            if drop_override_meta.overriden_types.contains(id) {
79                return Ok(());
80            }
81
82            drop_override_meta.overriden_types.insert(id.clone());
83        }
84
85        match f(metadata)? {
86            Some(region) => {
87                let location = Location::unknown(context);
88
89                let ty = registry.build_type(context, module, metadata, id)?;
90                let ptr_ty = llvm::r#type::pointer(context, 0);
91
92                let sierra_ty = registry.get_type(id)?;
93                let is_memory_allocated = sierra_ty.is_memory_allocated(registry)?;
94
95                let signature_ty = if is_memory_allocated { ptr_ty } else { ty };
96
97                // For memory allocated types, the generated function receives
98                // a pointer as argument. However, the user provided callback
99                // generates a region that receives a concrete value as
100                // argument. To workaround this, we insert a block at the start
101                // of the region that dereferences the pointer, and jumps to the
102                // user provided implementation.
103                if is_memory_allocated {
104                    let entry_block = region.first_block().unwrap();
105                    let pre_entry_block =
106                        region.insert_block_before(entry_block, Block::new(&[(ptr_ty, location)]));
107                    pre_entry_block.append_operation(cf::br(
108                        &entry_block,
109                        &[pre_entry_block.load(context, location, pre_entry_block.arg(0)?, ty)?],
110                        location,
111                    ));
112                }
113
114                module.body().append_operation(func::func(
115                    context,
116                    StringAttribute::new(context, &format!("drop${}", id.id)),
117                    TypeAttribute::new(FunctionType::new(context, &[signature_ty], &[]).into()),
118                    region,
119                    &[
120                        (
121                            Identifier::new(context, "sym_visibility"),
122                            StringAttribute::new(context, "public").into(),
123                        ),
124                        (
125                            Identifier::new(context, "llvm.CConv"),
126                            Attribute::parse(context, "#llvm.cconv<fastcc>")
127                                .ok_or(Error::ParseAttributeError)?,
128                        ),
129                        (
130                            Identifier::new(context, "llvm.linkage"),
131                            Attribute::parse(context, "#llvm.linkage<private>")
132                                .ok_or(Error::ParseAttributeError)?,
133                        ),
134                    ],
135                    Location::unknown(context),
136                ));
137            }
138            None => {
139                // The following getter should always return a value, but the if statement is kept
140                // just in case the meta has been removed (which it shouldn't).
141                if let Some(drop_override_meta) = metadata.get_mut::<Self>() {
142                    drop_override_meta.overriden_types.remove(id);
143                }
144            }
145        }
146
147        Ok(())
148    }
149
150    /// Returns whether a type has a registered drop implementation.
151    pub(crate) fn is_overriden(metadata: &mut MetadataStorage, id: &ConcreteTypeId) -> bool {
152        metadata
153            .get_or_insert_with(Self::default)
154            .overriden_types
155            .contains(id)
156    }
157
158    /// Generates code to invoke a drop implementation for a type, or does nothing if no
159    /// implementation was registered.
160    #[allow(clippy::too_many_arguments)]
161    pub(crate) fn invoke_override<'ctx, 'this>(
162        context: &'ctx Context,
163        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
164        module: &Module<'ctx>,
165        init_block: &'this Block<'ctx>,
166        block: &'this Block<'ctx>,
167        location: Location<'ctx>,
168        metadata: &mut MetadataStorage,
169        id: &ConcreteTypeId,
170        value: Value<'ctx, 'this>,
171    ) -> Result<()> {
172        if Self::is_overriden(metadata, id) {
173            let ty = registry.build_type(context, module, metadata, id)?;
174            let sierra_ty = registry.get_type(id)?;
175            let is_memory_allocated = sierra_ty.is_memory_allocated(registry)?;
176
177            // From memory allocated types, the drop function receives a pointer
178            // as argument, so we need to alloc the given value onto the stack
179            // and pass a pointer to it instead.
180            let value = if is_memory_allocated {
181                // The init_block is guaranteed to not be executed multiple
182                // times on tail-recursive functions.
183                let value_ptr = init_block.alloca1(
184                    context,
185                    location,
186                    ty,
187                    sierra_ty.layout(registry)?.align(),
188                )?;
189                block.store(context, location, value_ptr, value)?;
190                value_ptr
191            } else {
192                value
193            };
194
195            block.append_operation(func::call(
196                context,
197                FlatSymbolRefAttribute::new(context, &format!("drop${}", id.id)),
198                &[value],
199                &[],
200                location,
201            ));
202        }
203
204        Ok(())
205    }
