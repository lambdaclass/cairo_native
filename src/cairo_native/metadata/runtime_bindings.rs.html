97            }

100            }

103            }
104            RuntimeBinding::EcStateInit => {
105                crate::runtime::cairo_native__libfunc__ec__ec_state_init as *const ()
106            }
107            RuntimeBinding::EcStateAdd => {
108                crate::runtime::cairo_native__libfunc__ec__ec_state_add as *const ()
109            }
110            RuntimeBinding::EcPointTryNewNz => {
111                crate::runtime::cairo_native__libfunc__ec__ec_point_try_new_nz as *const ()

141    /// Register the global for the given binding, if not yet registered, and return
142    /// a pointer to the stored function.
143    ///
144    /// For the function to be available, `setup_runtime` must be called before running the module
145    fn build_function<'c, 'a>(
146        &mut self,
147        context: &'c Context,
148        module: &Module,
149        block: &'a Block<'c>,
150        location: Location<'c>,
151        binding: RuntimeBinding,
152    ) -> Result<Value<'c, 'a>> {
153        if self.active_map.insert(binding) {
154            module.body().append_operation(
155                ods::llvm::mlir_global(
156                    context,
157                    Region::new(),
158                    TypeAttribute::new(llvm::r#type::pointer(context, 0)),
159                    StringAttribute::new(context, binding.symbol()),
160                    Attribute::parse(context, "#llvm.linkage<weak>")
161                        .ok_or(Error::ParseAttributeError)?,
162                    location,
163                )
164                .into(),
165            );
166        }
167
168        let global_address = block.append_op_result(
169            ods::llvm::mlir_addressof(
170                context,
171                llvm::r#type::pointer(context, 0),
172                FlatSymbolRefAttribute::new(context, binding.symbol()),
173                location,
174            )
175            .into(),
176        )?;
177
178        Ok(block.load(
179            context,
180            location,
181            global_address,
182            llvm::r#type::pointer(context, 0),
183        )?)
184    }

186    /// Build if necessary the extended euclidean algorithm used in circuit inverse gates.
187    ///
188    /// After checking, calls the MLIR function with arguments `a` and `b` which are the initial remainders
189    /// used in the algorithm and returns a `Value` containing a struct where the first element is the
190    /// greatest common divisor of `a` and `b` and the second element is the bezout coefficient x.
191    pub fn extended_euclidean_algorithm<'c, 'a>(
192        &mut self,
193        context: &'c Context,
194        module: &Module,
195        block: &'a Block<'c>,
196        location: Location<'c>,
197        a: Value<'c, '_>,
198        b: Value<'c, '_>,
199    ) -> Result<Value<'c, 'a>>
200    where
201        'c: 'a,
202    {
203        let func_symbol = RuntimeBinding::ExtendedEuclideanAlgorithm.symbol();
204        if self
205            .active_map
206            .insert(RuntimeBinding::ExtendedEuclideanAlgorithm)
207        {
208            build_egcd_function(module, context, location, func_symbol)?;
209        }
210        let integer_type: Type = IntegerType::new(context, 384 * 2).into();
211        // The struct returned by the function that contains both of the results
212        let return_type = llvm::r#type::r#struct(context, &[integer_type, integer_type], false);
213        Ok(block
214            .append_operation(
215                OperationBuilder::new("llvm.call", location)
216                    .add_attributes(&[(
217                        Identifier::new(context, "callee"),
218                        FlatSymbolRefAttribute::new(context, func_symbol).into(),
219                    )])
220                    .add_operands(&[a, b])
221                    .add_results(&[return_type])
222                    .build()?,
223            )
224            .result(0)?
225            .into())
226    }

486    /// Register if necessary, then invoke the `dict_alloc_new()` function.
487    ///
488    /// Returns a opaque pointer as the result.
489    #[allow(clippy::too_many_arguments)]
490    pub fn dict_new<'c, 'a>(
491        &mut self,
492        context: &'c Context,
493        module: &Module,
494        block: &'a Block<'c>,
495        location: Location<'c>,
496        drop_fn: Option<Value<'c, 'a>>,
497        layout: Layout,
498    ) -> Result<Value<'c, 'a>>
499    where
500        'c: 'a,
501    {
502        let function =
503            self.build_function(context, module, block, location, RuntimeBinding::DictNew)?;
504
505        let i64_ty = IntegerType::new(context, 64).into();
506        let size = block.const_int_from_type(context, location, layout.size(), i64_ty)?;
507        let align = block.const_int_from_type(context, location, layout.align(), i64_ty)?;
508
509        let drop_fn = match drop_fn {
510            Some(x) => x,
511            None => {
512                block.append_op_result(llvm::zero(llvm::r#type::pointer(context, 0), location))?
513            }
514        };
515
516        Ok(block.append_op_result(
517            OperationBuilder::new("llvm.call", location)
518                .add_operands(&[function])
519                .add_operands(&[size, align, drop_fn])
520                .add_results(&[llvm::r#type::pointer(context, 0)])
521                .build()?,
522        )?)
523    }

578    /// Register if necessary, then invoke the `dict_get()` function.
579    ///
580    /// Gets the value for a given key, the returned pointer is null if not found.
581    ///
582    /// Returns a opaque pointer as the result.
583    #[allow(clippy::too_many_arguments)]
584    pub fn dict_get<'c, 'a>(
585        &mut self,
586        context: &'c Context,
587        helper: &LibfuncHelper<'c, 'a>,
588        block: &'a Block<'c>,
589        dict_ptr: Value<'c, 'a>, // ptr to the dict
590        key_ptr: Value<'c, 'a>,  // key must be a ptr to Felt
591        location: Location<'c>,
592    ) -> Result<(Value<'c, 'a>, Value<'c, 'a>)>
593    where
594        'c: 'a,
595    {
596        let function =
597            self.build_function(context, helper, block, location, RuntimeBinding::DictGet)?;
598
599        let value_ptr = helper.init_block().alloca1(
600            context,
601            location,
602            llvm::r#type::pointer(context, 0),
603            align_of::<*mut ()>(),
604        )?;
605
606        let is_present = block.append_op_result(
607            OperationBuilder::new("llvm.call", location)
608                .add_operands(&[function])
609                .add_operands(&[dict_ptr, key_ptr, value_ptr])
610                .add_results(&[IntegerType::new(context, c_int::BITS).into()])
611                .build()?,
612        )?;
613
614        let value_ptr = block.load(
615            context,
616            location,
617            value_ptr,
618            llvm::r#type::pointer(context, 0),
619        )?;
620
621        Ok((is_present, value_ptr))
622    }

685    /// Calls the cheatcode syscall with the given arguments.
686    ///
687    /// The result is stored in `result_ptr`.
688    #[allow(clippy::too_many_arguments)]
689    #[cfg(feature = "with-cheatcode")]
690    pub fn vtable_cheatcode<'c, 'a>(
691        &mut self,
692        context: &'c Context,
693        module: &Module,
694        block: &'a Block<'c>,
695        location: Location<'c>,
696        result_ptr: Value<'c, 'a>,
697        selector_ptr: Value<'c, 'a>,
698        args: Value<'c, 'a>,
699    ) -> Result<OperationRef<'c, 'a>>
700    where
701        'c: 'a,
702    {
703        let function = self.build_function(
704            context,
705            module,
706            block,
707            location,
708            RuntimeBinding::VtableCheatcode,
709        )?;
710
711        Ok(block.append_operation(
712            OperationBuilder::new("llvm.call", location)
713                .add_operands(&[function])
714                .add_operands(&[result_ptr, selector_ptr, args])
715                .build()?,
716        ))
717    }
718}

752/// The extended euclidean algorithm calculates the greatest common divisor (gcd) of two integers a and b,
753/// as well as the bezout coefficients x and y such that ax+by=gcd(a,b)
754/// if gcd(a,b) = 1, then x is the modular multiplicative inverse of a modulo b.
755/// See https://en.wikipedia.org/wiki/Extended_Euclidean_algorithm
756///
757/// This function declares a MLIR function that given two numbers a and b, returns a MLIR struct with gcd(a, b)
758/// and the bezout coefficient x. The declaration is done in the body of the module.
759fn build_egcd_function<'ctx>(
760    module: &Module,
761    context: &'ctx Context,
762    location: Location<'ctx>,
763    func_symbol: &str,
764) -> Result<()> {
765    let integer_type: Type = IntegerType::new(context, 384 * 2).into();
766    let region = Region::new();
767
768    let entry_block = region.append_block(Block::new(&[
769        (integer_type, location),
770        (integer_type, location),
771    ]));
772
773    let a = entry_block.arg(0)?;
774    let b = entry_block.arg(1)?;
775    // The egcd algorithm works by calculating a series of remainders `rem`, being each `rem_i` the remainder of dividing `rem_{i-1}` with `rem_{i-2}`
776    // For the initial setup, rem_0 = b, rem_1 = a.
777    // This order is chosen because if we reverse them, then the first iteration will just swap them
778    let remainder = a;
779    let prev_remainder = b;
780
781    // Similarly we'll calculate another series which starts 0,1,... and from which we
782    // will retrieve the modular inverse of a
783    let prev_inverse = entry_block.const_int_from_type(context, location, 0, integer_type)?;
784    let inverse = entry_block.const_int_from_type(context, location, 1, integer_type)?;
785
786    let loop_block = region.append_block(Block::new(&[
787        (integer_type, location),
788        (integer_type, location),
789        (integer_type, location),
790        (integer_type, location),
791    ]));
792    let end_block = region.append_block(Block::new(&[
793        (integer_type, location),
794        (integer_type, location),
795    ]));
796
797    entry_block.append_operation(cf::br(
798        &loop_block,
799        &[prev_remainder, remainder, prev_inverse, inverse],
800        location,
801    ));
802
803    // -- Loop body --
804    // Arguments are rem_(i-1), rem, inv_(i-1), inv
805    let prev_remainder = loop_block.arg(0)?;
806    let remainder = loop_block.arg(1)?;
807    let prev_inverse = loop_block.arg(2)?;
808    let inverse = loop_block.arg(3)?;
809
810    // First calculate q = rem_(i-1)/rem_i, rounded down
811    let quotient =
812        loop_block.append_op_result(arith::divui(prev_remainder, remainder, location))?;
813
814    // Then rem_(i+1) = rem_(i-1) - q * rem_i, and inv_(i+1) = inv_(i-1) - q * inv_i
815    let rem_times_quo = loop_block.muli(remainder, quotient, location)?;
816    let inv_times_quo = loop_block.muli(inverse, quotient, location)?;
817    let next_remainder =
818        loop_block.append_op_result(arith::subi(prev_remainder, rem_times_quo, location))?;
819    let next_inverse =
820        loop_block.append_op_result(arith::subi(prev_inverse, inv_times_quo, location))?;
821
822    // Check if rem_(i+1) is 0
823    // If true, then:
824    // - rem_i is the gcd of a and b
825    // - inv_i is the bezout coefficient x
826    let zero = loop_block.const_int_from_type(context, location, 0, integer_type)?;
827    let next_remainder_eq_zero =
828        loop_block.cmpi(context, CmpiPredicate::Eq, next_remainder, zero, location)?;
829    loop_block.append_operation(cf::cond_br(
830        context,
831        next_remainder_eq_zero,
832        &end_block,
833        &loop_block,
834        &[remainder, inverse],
835        &[remainder, next_remainder, inverse, next_inverse],
836        location,
837    ));
838
839    // Create the struct that will contain the results
840    let results = end_block.append_op_result(llvm::undef(
841        llvm::r#type::r#struct(context, &[integer_type, integer_type], false),
842        location,
843    ))?;
844    let results = end_block.insert_values(
845        context,
846        location,
847        results,
848        &[end_block.arg(0)?, end_block.arg(1)?],
849    )?;
850    end_block.append_operation(llvm::r#return(Some(results), location));
851
852    let func_name = StringAttribute::new(context, func_symbol);
853    module.body().append_operation(llvm::func(
854        context,
855        func_name,
856        TypeAttribute::new(llvm::r#type::function(
857            llvm::r#type::r#struct(context, &[integer_type, integer_type], false),
858            &[integer_type, integer_type],
859            false,
860        )),
861        region,
862        &[(
863            Identifier::new(context, "no_inline"), // Adding this attribute significantly improves compilation
864            Attribute::unit(context),
865        )],
866        location,
867    ));
868    Ok(())
869}
