1//! This file contains the gas calculation metadata.
2//!
3//! Each statement has an associated `GasCost`, which represents the cost of
4//! executing that statement, in terms of tokens.
5//!
6//! To calculate the actual cost, the amount of tokens is multiplied by the cost
7//! of the given token type. The cost of each token type is specified on runtime,
8//! with the `BuiltinCosts` structure.
9//!
10//! When implementing libfuncs, the `GasCost` metadata entry already contains
11//! the `GasCost` for the current sierra statement
12

30/// Holds global gas info.
31#[derive(Default)]
32pub struct GasMetadata(pub CairoGasMetadata);
33
34/// The gas cost associated to a determined sierra statement.
35///
36/// It contains the amount of tokens for each token type,
37/// that a given sierra statement costs.
38#[derive(Debug, Clone, PartialEq, Eq, Hash)]
39pub struct GasCost(pub Vec<(u64, CostTokenType)>);
40
41/// Error for metadata calculations.
42#[derive(Debug, thiserror::Error, Eq, PartialEq)]
43pub enum GasMetadataError {
44    #[error(transparent)]
45    ApChangeError(#[from] ApChangeError),
46    #[error(transparent)]
47    CostError(#[from] CostError),
48    #[error("Not enough gas to run the operation. Required: {:?}, Available: {:?}.", gas.0, gas.1)]
49    NotEnoughGas { gas: Box<(u64, u64)> },
50}
51
52impl GasMetadata {
53    pub fn new(
54        sierra_program: &Program,
55        config: Option<MetadataComputationConfig>,
56    ) -> Result<GasMetadata, GasMetadataError> {
57        let cairo_gas_metadata = if let Some(metadata_config) = config {
58            calc_metadata(sierra_program, metadata_config)?
59        } else {
60            calc_metadata_ap_change_only(sierra_program)?
61        };
62
63        Ok(GasMetadata::from(cairo_gas_metadata))
64    }
65
66    /// Returns the initial value for the gas counter.
67    /// If `available_gas` is None returns 0.
68    pub fn get_initial_available_gas(
69        &self,
70        func: &FunctionId,
71        available_gas: Option<u64>,
72    ) -> Result<u64, GasMetadataError> {
73        let Some(available_gas) = available_gas else {
74            return Ok(0);
75        };
76
77        // In case we don't have any costs - it means no gas equations were solved (and we are in
78        // the case of no gas checking enabled) - so the gas builtin is irrelevant, and we
79        // can return any value.
80        let Some(required_gas) = self.initial_required_gas(func) else {
81            return Ok(0);
82        };
83
84        available_gas
85            .checked_sub(required_gas)
86            .ok_or(GasMetadataError::NotEnoughGas {
87                gas: Box::new((required_gas, available_gas)),
88            })
89    }
90
91    pub fn initial_required_gas(&self, func: &FunctionId) -> Option<u64> {
92        if self.gas_info.function_costs.is_empty() {
93            return None;
94        }
95        Some(
96            self.gas_info.function_costs[func]
97                .iter()
98                .map(|(token_type, val)| {
99                    TryInto::<usize>::try_into(*val)
100                        .expect("could not cast gas cost from i64 to usize")
101                        * token_gas_cost(*token_type)
102                })
103                .sum::<usize>() as u64,
104        )
105    }
106
107    pub fn initial_required_gas_for_entry_points(
108        &self,
109    ) -> NativeResult<BTreeMap<u64, BTreeMap<u64, u64>>> {
110        self.gas_info
111            .function_costs
112            .iter()
113            .map(|func| {
114                Ok((func.0.id, {
115                    let mut costs = BTreeMap::new();
116
117                    for (token, val) in func.1.iter() {
118                        let offset: u64 = match token {
119                            CostTokenType::Const => 0,
120                            CostTokenType::Pedersen => 1,
121                            CostTokenType::Bitwise => 2,
122                            CostTokenType::EcOp => 3,
123                            CostTokenType::Poseidon => 4,
124                            CostTokenType::AddMod => 5,
125                            CostTokenType::MulMod => 6,
126                            _ => native_panic!("matched an unexpected CostTokenType"),
127                        };
128                        costs.insert(offset, *val as u64);
129                    }
130
131                    costs
132                }))
133            })
134            .collect()
135    }
136
137    pub fn get_gas_costs_for_statement(&self, idx: StatementIdx) -> Vec<(u64, CostTokenType)> {
138        let mut costs = Vec::new();
139        for cost_type in CostTokenType::iter_casm_tokens() {
140            if let Some(cost_count) =
141                self.get_gas_cost_for_statement_and_cost_token_type(idx, *cost_type)
142            {
143                if cost_count > 0 {
144                    costs.push((cost_count, *cost_type));
145                }
146            }
147        }
148        costs
149    }
150
151    pub fn get_gas_cost_for_statement_and_cost_token_type(
152        &self,
153        idx: StatementIdx,
154        cost_type: CostTokenType,
155    ) -> Option<u64> {
156        self.gas_info
157            .variable_values
158            .get(&(idx, cost_type))
159            .copied()
160            .map(|x| x.try_into().expect("gas cost couldn't be converted to u64"))
161    }
162}

164impl From<CairoGasMetadata> for GasMetadata {
165    fn from(value: CairoGasMetadata) -> Self {
166        Self(value)
167    }
168}
169
170impl Deref for GasMetadata {
171    type Target = CairoGasMetadata;
172
173    fn deref(&self) -> &Self::Target {
174        &self.0
175    }
176}
177
178impl fmt::Debug for GasMetadata {
179    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
180        f.debug_struct("GasMetadata")
181            .field("ap_change_info", &self.ap_change_info)
182            .field("gas_info", &self.gas_info)
183            .finish()
184    }
185}
186
187impl Clone for GasMetadata {
188    fn clone(&self) -> Self {
189        Self(CairoGasMetadata {
190            ap_change_info: ApChangeInfo {
191                variable_values: self.ap_change_info.variable_values.clone(),
192                function_ap_change: self.ap_change_info.function_ap_change.clone(),
193            },
194            gas_info: GasInfo {
195                variable_values: self.gas_info.variable_values.clone(),
196                function_costs: self.gas_info.function_costs.clone(),
197            },
198        })
199    }
200}

202impl From<CairoGasMetadataError> for GasMetadataError {
203    fn from(value: CairoGasMetadataError) -> Self {
204        match value {
205            CairoGasMetadataError::ApChangeError(x) => GasMetadataError::ApChangeError(x),
206            CairoGasMetadataError::CostError(x) => GasMetadataError::CostError(x),
207        }
208    }
