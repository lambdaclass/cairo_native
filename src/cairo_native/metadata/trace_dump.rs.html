23#[derive(Clone, Copy, Debug, Eq, Hash, PartialEq)]
24pub enum TraceBinding {
25    State,
26    Push,
27    TraceId,
28}
29
30impl TraceBinding {
31    pub const fn symbol(self) -> &'static str {
32        match self {
33            TraceBinding::State => "cairo_native__trace_dump__add_variable_to_state",
34            TraceBinding::Push => "cairo_native__trace_dump__push_state_to_trace_dump",
35            TraceBinding::TraceId => "cairo_native__trace_dump__trace_id",
36        }
37    }
38
39    const fn function_ptr(self) -> *const () {
40        match self {
41            TraceBinding::State => trace_dump_runtime::add_variable_to_state as *const (),
42            TraceBinding::Push => trace_dump_runtime::push_state_to_trace_dump as *const (),
43            // it has no function pointer, as its a global constant
44            TraceBinding::TraceId => ptr::null(),
45        }
46    }
47}

55    /// Register the global for the given binding, if not yet registered, and return
56    /// a pointer to the stored value.
57    ///
58    /// For the function to be available, `setup_runtime` must be called before running the module
59    fn build_function<'c, 'a>(
60        &mut self,
61        context: &'c Context,
62        module: &Module,
63        block: &'a Block<'c>,
64        location: Location<'c>,
65        binding: TraceBinding,
66    ) -> Result<Value<'c, 'a>> {
67        if self.active_map.insert(binding) {
68            module.body().append_operation(
69                ods::llvm::mlir_global(
70                    context,
71                    Region::new(),
72                    TypeAttribute::new(llvm::r#type::pointer(context, 0)),
73                    StringAttribute::new(context, binding.symbol()),
74                    Attribute::parse(context, "#llvm.linkage<weak>")
75                        .ok_or(Error::ParseAttributeError)?,
76                    location,
77                )
78                .into(),
79            );
80        }
81
82        let global_address = block.append_op_result(
83            ods::llvm::mlir_addressof(
84                context,
85                llvm::r#type::pointer(context, 0),
86                FlatSymbolRefAttribute::new(context, binding.symbol()),
87                location,
88            )
89            .into(),
90        )?;
91
92        block.load(
93            context,
94            location,
95            global_address,
96            llvm::r#type::pointer(context, 0),
97        )
98    }

100    #[allow(clippy::too_many_arguments)]
101    pub fn build_state(
102        &mut self,
103        context: &Context,
104        module: &Module,
105        block: &Block,
106        var_id: &VarId,
107        value_ty: &ConcreteTypeId,
108        value_ptr: Value,
109        location: Location,
110    ) -> Result<()> {
111        let trace_id = self.build_trace_id(context, module, block, location)?;
112
113        let var_id = block.const_int(context, location, var_id.id, 64)?;
114        let value_ty = block.const_int(context, location, value_ty.id, 64)?;
115
116        let function =
117            self.build_function(context, module, block, location, TraceBinding::State)?;
118        block.append_operation(
119            OperationBuilder::new("llvm.call", location)
120                .add_operands(&[function])
121                .add_operands(&[trace_id, var_id, value_ty, value_ptr])
122                .build()?,
123        );
124
125        Ok(())
126    }

128    pub fn build_push(
129        &mut self,
130        context: &Context,
131        module: &Module,
132        block: &Block,
133        statement_idx: StatementIdx,
134        location: Location,
135    ) -> Result<()> {
136        let trace_id = self.build_trace_id(context, module, block, location)?;
137        let statement_idx = block.const_int(context, location, statement_idx.0, 64)?;
138
139        let function = self.build_function(context, module, block, location, TraceBinding::Push)?;
140
141        block.append_operation(
142            OperationBuilder::new("llvm.call", location)
143                .add_operands(&[function])
144                .add_operands(&[trace_id, statement_idx])
145                .build()?,
146        );
147
148        Ok(())
149    }

151    pub fn build_trace_id<'c, 'a>(
152        &mut self,
153        context: &'c Context,
154        module: &Module,
155        block: &'a Block<'c>,
156        location: Location<'c>,
157    ) -> Result<Value<'c, 'a>> {
158        if self.active_map.insert(TraceBinding::TraceId) {
159            module.body().append_operation(memref::global(
160                context,
161                TraceBinding::TraceId.symbol(),
162                None,
163                MemRefType::new(IntegerType::new(context, 64).into(), &[], None, None),
164                None,
165                false,
166                None,
167                location,
168            ));
169        }
170
171        let trace_id_ptr = block
172            .append_op_result(memref::get_global(
173                context,
174                TraceBinding::TraceId.symbol(),
175                MemRefType::new(IntegerType::new(context, 64).into(), &[], None, None),
176                location,
177            ))
178            .unwrap();
179
180        block.append_op_result(memref::load(trace_id_ptr, &[], location))
181    }

184pub fn setup_runtime(find_symbol_ptr: impl Fn(&str) -> Option<*mut c_void>) {
185    let bindings = &[TraceBinding::State, TraceBinding::Push];
186
187    for binding in bindings {
188        if let Some(global) = find_symbol_ptr(binding.symbol()) {
189            let global = global.cast::<*const ()>();
190            unsafe { *global = binding.function_ptr() };
191        }
192    }

256    /// Adds a new variable to the current state of the trace dump with the
257    /// given identifier.
258    ///
259    /// Receives a pointer to the value, even if the value is a pointer itself.
260    pub unsafe extern "C" fn add_variable_to_state(
261        trace_id: u64,
262        var_id: u64,
263        type_id: u64,
264        value_ptr: NonNull<()>,
265    ) {
266        let mut trace_dump = TRACE_DUMP.lock().unwrap();
267        let Some(trace_dump) = trace_dump.get_mut(&trace_id) else {
268            eprintln!("Could not find trace dump!");
269            return;
270        };
271
272        let type_id = ConcreteTypeId::new(type_id);
273        let value = value_from_ptr(&trace_dump.registry, &type_id, value_ptr);
274
275        trace_dump.state.insert(VarId::new(var_id), value);
276    }
277
278    /// Pushes the latest state to the trace dump with the given identifier.
279    ///
280    /// It is called after all variables have been added with `add_variable_to_state`.
281    pub unsafe extern "C" fn push_state_to_trace_dump(trace_id: u64, statement_idx: u64) {
282        let mut trace_dump = TRACE_DUMP.lock().unwrap();
283        let Some(trace_dump) = trace_dump.get_mut(&trace_id) else {
284            eprintln!("Could not find trace dump!");
285            return;
286        };
287
288        let mut items = OrderedHashMap::default();
289        swap(&mut items, &mut trace_dump.state);
290
291        trace_dump
292            .trace
293            .push(StateDump::new(StatementIdx(statement_idx as usize), items));
294    }

296    /// TODO: Can we reuse `cairo_native::Value::from_ptr`?
297    unsafe fn value_from_ptr(
298        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
299        type_id: &ConcreteTypeId,
300        value_ptr: NonNull<()>,
301    ) -> Value {
302        let type_info = registry.get_type(type_id).unwrap();
303        match type_info {
304            CoreTypeConcrete::Felt252(_)
305            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::ContractAddress(_))
306            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::ClassHash(_))
307            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::StorageAddress(_))
308            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::StorageBaseAddress(_)) => {
309                Value::Felt(Felt::from_bytes_le(value_ptr.cast().as_ref()))
310            }
311            CoreTypeConcrete::Uint8(_) => Value::U8(value_ptr.cast().read()),
312            CoreTypeConcrete::Uint16(_) => Value::U16(value_ptr.cast().read()),
313            CoreTypeConcrete::Uint32(_) => Value::U32(value_ptr.cast().read()),
314            CoreTypeConcrete::Uint64(_) | CoreTypeConcrete::GasBuiltin(_) => {
315                Value::U64(value_ptr.cast().read())
316            }
317            CoreTypeConcrete::Uint128(_) => Value::U128(value_ptr.cast().read()),
318
319            CoreTypeConcrete::BoundedInt(BoundedIntConcreteType { range, .. }) => {
320                let n_bits = ((range.size() - BigInt::one()).bits() as u32).max(1);
321                let n_bytes = n_bits.next_multiple_of(8) >> 3;
322
323                let data = NonNull::slice_from_raw_parts(value_ptr.cast::<u8>(), n_bytes as usize);
324
325                let value = BigInt::from_bytes_le(num_bigint::Sign::Plus, data.as_ref());
326
327                Value::BoundedInt {
328                    range: Range {
329                        start: range.lower.clone(),
330                        end: range.upper.clone(),
331                    },
332                    value: value + &range.lower,
333                }
334            }
335
336            CoreTypeConcrete::EcPoint(_) => {
337                let layout = Layout::new::<()>();
338                let (x, layout) = {
339                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
340                    (
341                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
342                        layout,
343                    )
344                };
345                let (y, _) = {
346                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
347                    (
348                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
349                        layout,
350                    )
351                };
352
353                Value::EcPoint { x, y }
354            }
355            CoreTypeConcrete::EcState(_) => {
356                let layout = Layout::new::<()>();
357                let (x0, layout) = {
358                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
359                    (
360                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
361                        layout,
362                    )
363                };
364                let (y0, layout) = {
365                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
366                    (
367                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
368                        layout,
369                    )
370                };
371                let (x1, layout) = {
372                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
373                    (
374                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
375                        layout,
376                    )
377                };
378                let (y1, _) = {
379                    let (layout, offset) = layout.extend(Layout::new::<[u128; 2]>()).unwrap();
380                    (
381                        Felt::from_bytes_le(value_ptr.byte_add(offset).cast().as_ref()),
382                        layout,
383                    )
384                };
385
386                Value::EcState { x0, y0, x1, y1 }
387            }
388
389            CoreTypeConcrete::Uninitialized(info) => Value::Uninitialized {
390                ty: info.ty.clone(),
391            },
392            CoreTypeConcrete::Box(info) => {
393                value_from_ptr(registry, &info.ty, value_ptr.cast::<NonNull<()>>().read())
394            }
395            CoreTypeConcrete::Array(info) => {
396                let array = value_ptr.cast::<ArrayAbi<()>>().read();
397
398                let layout = registry
399                    .get_type(&info.ty)
400                    .unwrap()
401                    .layout(registry)
402                    .unwrap()
403                    .pad_to_align();
404
405                let mut data = Vec::with_capacity((array.until - array.since) as usize);
406
407                if !array.ptr.is_null() {
408                    let data_ptr = array.ptr.read();
409                    for index in (array.since)..array.until {
410                        let index = index as usize;
411
412                        data.push(value_from_ptr(
413                            registry,
414                            &info.ty,
415                            NonNull::new(data_ptr.byte_add(layout.size() * index)).unwrap(),
416                        ));
417                    }
418                }
419
420                Value::Array {
421                    ty: info.ty.clone(),
422                    data,
423                }
424            }
425
426            CoreTypeConcrete::Struct(info) => {
427                let mut layout = Layout::new::<()>();
428                let mut members = Vec::with_capacity(info.members.len());
429                for member_ty in &info.members {
430                    let type_info = registry.get_type(member_ty).unwrap();
431                    let member_layout = type_info.layout(registry).unwrap();
432
433                    let offset;
434                    (layout, offset) = layout.extend(member_layout).unwrap();
435
436                    let current_ptr = value_ptr.byte_add(offset);
437                    members.push(value_from_ptr(registry, member_ty, current_ptr));
438                }
439
440                Value::Struct(members)
441            }
442            CoreTypeConcrete::Enum(info) => {
443                let tag_bits = info.variants.len().next_power_of_two().trailing_zeros();
444                let (tag_value, layout) = match tag_bits {
445                    0 => (0, Layout::new::<()>()),
446                    width if width <= 8 => {
447                        (value_ptr.cast::<u8>().read() as usize, Layout::new::<u8>())
448                    }
449                    width if width <= 16 => (
450                        value_ptr.cast::<u16>().read() as usize,
451                        Layout::new::<u16>(),
452                    ),
453                    width if width <= 32 => (
454                        value_ptr.cast::<u32>().read() as usize,
455                        Layout::new::<u32>(),
456                    ),
457                    width if width <= 64 => (
458                        value_ptr.cast::<u64>().read() as usize,
459                        Layout::new::<u64>(),
460                    ),
461                    width if width <= 128 => (
462                        value_ptr.cast::<u128>().read() as usize,
463                        Layout::new::<u128>(),
464                    ),
465                    _ => todo!(),
466                };
467
468                let payload = {
469                    let (_, offset) = layout
470                        .extend(
471                            registry
472                                .get_type(&info.variants[tag_value])
473                                .unwrap()
474                                .layout(registry)
475                                .unwrap(),
476                        )
477                        .unwrap();
478
479                    value_from_ptr(
480                        registry,
481                        &info.variants[tag_value],
482                        value_ptr.byte_add(offset),
483                    )
484                };
485
486                Value::Enum {
487                    self_ty: type_id.clone(),
488                    index: tag_value,
489                    payload: Box::new(payload),
490                }
491            }
492
493            CoreTypeConcrete::NonZero(info) | CoreTypeConcrete::Snapshot(info) => {
494                value_from_ptr(registry, &info.ty, value_ptr)
495            }
496
497            // Builtins and other unit types:
498            CoreTypeConcrete::Bitwise(_)
499            | CoreTypeConcrete::EcOp(_)
500            | CoreTypeConcrete::Pedersen(_)
501            | CoreTypeConcrete::Poseidon(_)
502            | CoreTypeConcrete::RangeCheck96(_)
503            | CoreTypeConcrete::RangeCheck(_)
504            | CoreTypeConcrete::SegmentArena(_)
505            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::System(_))
506            | CoreTypeConcrete::Uint128MulGuarantee(_) => Value::Unit,
507
508            CoreTypeConcrete::BuiltinCosts(_) => {
509                let builtin_costs = value_ptr.cast::<&[u64; 7]>().read();
510                Value::BuiltinCosts(sierra_emu::BuiltinCosts {
511                    r#const: builtin_costs[0],
512                    pedersen: builtin_costs[1],
513                    bitwise: builtin_costs[2],
514                    ecop: builtin_costs[3],
515                    poseidon: builtin_costs[4],
516                    add_mod: builtin_costs[5],
517                    mul_mod: builtin_costs[6],
518                })
519            }
520
521            // TODO:
522            CoreTypeConcrete::Coupon(_) => todo!("CoreTypeConcrete::Coupon"),
523            CoreTypeConcrete::Circuit(circuit) => match circuit {
524                CircuitTypeConcrete::AddMod(_) => Value::Unit,
525                CircuitTypeConcrete::MulMod(_) => Value::Unit,
526                CircuitTypeConcrete::AddModGate(_) => Value::Unit,
527                CircuitTypeConcrete::Circuit(_) => Value::Unit,
528                CircuitTypeConcrete::CircuitData(info) => {
529                    let Some(GenericArg::Type(circuit_type_id)) =
530                        info.info.long_id.generic_args.first()
531                    else {
532                        panic!("generic arg should be a type");
533                    };
534                    let CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(circuit)) =
535                        registry.get_type(circuit_type_id).unwrap()
536                    else {
537                        panic!("generic arg should be a Circuit");
538                    };
539
540                    let u384_layout = Layout::from_size_align(48, 16).unwrap();
541
542                    let n_inputs = circuit.circuit_info.n_inputs;
543                    let mut values = Vec::with_capacity(n_inputs);
544
545                    let value_ptr = value_ptr.cast::<[u8; 48]>();
546
547                    for i in 0..n_inputs {
548                        let size = u384_layout.pad_to_align().size();
549                        let current_ptr = value_ptr.byte_add(size * i);
550                        let current_value = current_ptr.as_ref();
551                        values.push(BigUint::from_bytes_le(current_value));
552                    }
553
554                    Value::Circuit(values)
555                }
556                CircuitTypeConcrete::CircuitOutputs(info) => {
557                    let Some(GenericArg::Type(circuit_type_id)) =
558                        info.info.long_id.generic_args.first()
559                    else {
560                        panic!("generic arg should be a type");
561                    };
562                    let CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(circuit)) =
563                        registry.get_type(circuit_type_id).unwrap()
564                    else {
565                        panic!("generic arg should be a Circuit");
566                    };
567
568                    let u384_layout = Layout::from_size_align(48, 16).unwrap();
569
570                    let n_outputs = circuit.circuit_info.values.len();
571                    let mut values = Vec::with_capacity(n_outputs);
572
573                    let value_ptr = value_ptr.cast::<[u8; 48]>();
574
575                    for i in 0..n_outputs {
576                        let size = u384_layout.pad_to_align().size();
577                        let current_ptr = value_ptr.byte_add(size * i);
578                        let current_value = current_ptr.as_ref();
579                        values.push(BigUint::from_bytes_le(current_value));
580                    }
581
582                    Value::CircuitOutputs(values)
583                }
584                CircuitTypeConcrete::CircuitPartialOutputs(_) => {
585                    todo!("CircuitTypeConcrete::CircuitPartialOutputs")
586                }
587                CircuitTypeConcrete::CircuitDescriptor(_) => Value::Unit,
588                CircuitTypeConcrete::CircuitFailureGuarantee(_) => {
589                    todo!("CircuitTypeConcrete::CircuitFailureGuarantee")
590                }
591                CircuitTypeConcrete::CircuitInput(_) => {
592                    todo!("CircuitTypeConcrete::CircuitInput")
593                }
594                CircuitTypeConcrete::CircuitInputAccumulator(info) => {
595                    let Some(GenericArg::Type(circuit_type_id)) =
596                        info.info.long_id.generic_args.first()
597                    else {
598                        panic!("generic arg should be a type");
599                    };
600                    let CoreTypeConcrete::Circuit(CircuitTypeConcrete::Circuit(_)) =
601                        registry.get_type(circuit_type_id).unwrap()
602                    else {
603                        panic!("generic arg should be a Circuit");
604                    };
605
606                    let u64_layout = Layout::new::<u64>();
607                    let u384_layout = Layout::from_size_align(48, 16).unwrap();
608
609                    let length = unsafe { *value_ptr.cast::<u64>().as_ptr() };
610
611                    let (_, input_start_offset) = u64_layout.extend(u384_layout).unwrap();
612                    let start_ptr = value_ptr.byte_add(input_start_offset).cast::<[u8; 48]>();
613
614                    let mut values = Vec::with_capacity(length as usize);
615
616                    for i in 0..length {
617                        let size = u384_layout.pad_to_align().size();
618                        let current_ptr = start_ptr.byte_add(size * i as usize);
619                        let current_value = current_ptr.as_ref();
620                        values.push(BigUint::from_bytes_le(current_value));
621                    }
622
623                    Value::Circuit(values)
624                }
625                CircuitTypeConcrete::CircuitModulus(_) => {
626                    let value_ptr = value_ptr.cast::<[u8; 48]>();
627                    let value = unsafe { value_ptr.as_ref() };
628                    Value::CircuitModulus(BigUint::from_bytes_le(value))
629                }
630
631                CircuitTypeConcrete::InverseGate(_) => Value::Unit,
632                CircuitTypeConcrete::MulModGate(_) => Value::Unit,
633                CircuitTypeConcrete::SubModGate(_) => Value::Unit,
634                CircuitTypeConcrete::U96Guarantee(_) => {
635                    let value_ptr = value_ptr.cast::<[u8; 12]>();
636                    let value = unsafe { value_ptr.as_ref() };
637
638                    let mut array_value = [0u8; 16];
639                    array_value[..12].clone_from_slice(value);
640
641                    Value::U128(u128::from_le_bytes(array_value))
642                }
643                CircuitTypeConcrete::U96LimbsLessThanGuarantee(_) => Value::Unit,
644            },
645            CoreTypeConcrete::Const(_) => todo!("CoreTypeConcrete::Const"),
646            CoreTypeConcrete::Sint8(_) => Value::I8(value_ptr.cast().read()),
647            CoreTypeConcrete::Sint16(_) => todo!("CoreTypeConcrete::Sint16"),
648            CoreTypeConcrete::Sint32(_) => Value::I32(value_ptr.cast().read()),
649            CoreTypeConcrete::Sint64(_) => todo!("CoreTypeConcrete::Sint64"),
650            CoreTypeConcrete::Sint128(_) => Value::I128(value_ptr.cast().read()),
651            CoreTypeConcrete::Nullable(info) => {
652                let inner_ptr = value_ptr.cast::<*mut ()>().read();
653                match NonNull::new(inner_ptr) {
654                    Some(inner_ptr) => value_from_ptr(registry, &info.ty, inner_ptr),
655                    None => Value::Uninitialized {
656                        ty: info.ty.clone(),
657                    },
658                }
659            }
660
661            CoreTypeConcrete::SquashedFelt252Dict(info) | CoreTypeConcrete::Felt252Dict(info) => {
662                let value = value_ptr.cast::<&FeltDict>().read();
663
664                let data = value
665                    .mappings
666                    .iter()
667                    .map(|(k, &i)| {
668                        let p = value
669                            .elements
670                            .byte_offset((value.layout.size() * i) as isize);
671                        let v = match NonNull::new(p) {
672                            Some(value_ptr) => value_from_ptr(registry, &info.ty, value_ptr.cast()),
673                            None => Value::Uninitialized {
674                                ty: info.ty.clone(),
675                            },
676                        };
677                        let k = Felt::from_bytes_le(k);
678                        (k, v)
679                    })
680                    .collect::<HashMap<Felt, Value>>();
681
682                Value::FeltDict {
683                    ty: info.ty.clone(),
684                    count: value.count,
685                    data,
686                }
687            }
688            CoreTypeConcrete::Felt252DictEntry(info) => {
689                let value = value_ptr.cast::<FeltDictEntry>().read();
690
691                let data = value
692                    .dict
693                    .mappings
694                    .iter()
695                    .map(|(k, &i)| {
696                        let p = value
697                            .dict
698                            .elements
699                            .byte_offset((value.dict.layout.size() * i) as isize);
700                        let v = match NonNull::new(p) {
701                            Some(value_ptr) => value_from_ptr(registry, &info.ty, value_ptr.cast()),
702                            None => Value::Uninitialized {
703                                ty: info.ty.clone(),
704                            },
705                        };
706                        let k = Felt::from_bytes_le(k);
707                        (k, v)
708                    })
709                    .collect::<HashMap<Felt, Value>>();
710                let key = Felt::from_bytes_le(value.key);
711
712                Value::FeltDictEntry {
713                    ty: info.ty.clone(),
714                    data,
715                    count: value.dict.count,
716                    key,
717                }
718            }
719            CoreTypeConcrete::Span(_) => todo!("CoreTypeConcrete::Span"),
720            CoreTypeConcrete::Starknet(selector) => match selector {
721                StarknetTypeConcrete::Secp256Point(selector) => match selector {
722                    Secp256PointTypeConcrete::K1(_) => {
723                        let point: Secp256Point = value_ptr.cast().read();
724                        let emu_point = EmuSecp256k1Point {
725                            x: EmuU256 {
726                                lo: point.x.lo,
727                                hi: point.x.hi,
728                            },
729                            y: EmuU256 {
730                                lo: point.y.lo,
731                                hi: point.y.hi,
732                            },
733                        };
734                        emu_point.into_value()
735                    }
736                    Secp256PointTypeConcrete::R1(_) => {
737                        let point: Secp256Point = value_ptr.cast().read();
738                        let emu_point = EmuSecp256r1Point {
739                            x: EmuU256 {
740                                lo: point.x.lo,
741                                hi: point.x.hi,
742                            },
743                            y: EmuU256 {
744                                lo: point.y.lo,
745                                hi: point.y.hi,
746                            },
747                        };
748                        emu_point.into_value()
749                    }
750                },
751                StarknetTypeConcrete::Sha256StateHandle(_) => {
752                    let raw_data = value_ptr.cast::<NonNull<[u32; 8]>>().read().read();
753                    let data = raw_data.into_iter().map(Value::U32).collect_vec();
754                    Value::Struct(data)
755                }
756                _ => unreachable!(),
757            },
758            CoreTypeConcrete::Bytes31(_) => {
759                let original_data: [u8; 31] = value_ptr.cast().read();
760                let mut data = [0u8; 32];
761                for (i, v) in original_data.into_iter().enumerate() {
762                    data[i] = v
763                }
764
765                Value::Bytes31(Felt::from_bytes_le(&data))
766            }
767            CoreTypeConcrete::IntRange(_)
768            | CoreTypeConcrete::Blake(_)
769            | CoreTypeConcrete::QM31(_) => {
770                todo!()
771            }
772        }
773    }

779    }
