68/// Generation of MLIR types from their Sierra counterparts.
69///
70/// All possible Sierra types must implement it. It is already implemented for all the core Sierra
71/// types, contained in [CoreTypeConcrete].
72pub trait TypeBuilder {
73    /// Error type returned by this trait's methods.
74    type Error: Error;
75
76    /// Build the MLIR type.
77    fn build<'ctx>(
78        &self,
79        context: &'ctx Context,
80        module: &Module<'ctx>,
81        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
82        metadata: &mut MetadataStorage,
83        self_ty: &ConcreteTypeId,
84    ) -> Result<Type<'ctx>, Self::Error>;
85
86    /// Return whether the type is a builtin.
87    fn is_builtin(&self) -> bool;
88    /// Return whether the type requires a return pointer when returning,
89    /// instead of using the CPU registers.
90    ///
91    /// This attribute does not modify the compilation, and it only reflects
92    /// what the ABI of the target architecture already specifies.
93    /// - For x86-64: https://gitlab.com/x86-psABIs/x86-64-ABI.
94    /// - For AArch64: https://github.com/ARM-software/abi-aa.
95    ///
96    /// We can validate this empirically, by building a Cairo program that
97    /// returns a particular type, and seeing how it is lowered to machine code.
98    ///
99    /// ```bash
100    /// llc a.llvmir -o - --mtriple "aarch64"
101    /// llc a.llvmir -o - --mtriple "x86_64"
102    /// ```
103    fn is_complex(
104        &self,
105        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
106    ) -> Result<bool, Self::Error>;
107
108    /// Return whether the Sierra type resolves to a zero-sized type.
109    fn is_zst(
110        &self,
111        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
112    ) -> Result<bool, Self::Error>;
113
114    /// Generate the layout of the MLIR type.
115    ///
116    /// Used in both the compiler and the interface when calling the compiled code.
117    fn layout(
118        &self,
119        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
120    ) -> Result<Layout, Self::Error>;
121
122    /// Whether the layout should be allocated in memory (either the stack or
123    /// the heap) when used as a function invocation argument or return value.
124    ///
125    /// Unlike `is_complex`, this attribute alters the compilation:
126    ///
127    /// - When passing a memory allocated value to a function, we allocate that
128    ///   value on the stack, and pass a pointer to it.
129    ///
130    /// - If a function returns a memory allocated value, we receive a return
131    ///   pointer as its first argument, and write the return value there
132    ///   instead.
133    ///
134    /// The rationale behind allocating a value in memory, rather than
135    /// registers, is to avoid putting too much pressure on the register
136    /// allocation pass for really complex types, like enums.
137    fn is_memory_allocated(
138        &self,
139        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
140    ) -> Result<bool, Self::Error>;
141
142    /// If the type is an integer, return its value range.
143    fn integer_range(
144        &self,
145        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
146    ) -> Result<Range, Self::Error>;
147
148    /// Return whether the type is a `BoundedInt<>`, either directly or indirectly (ex. through
149    /// `NonZero<BoundedInt<>>`).
150    fn is_bounded_int(
151        &self,
152        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
153    ) -> Result<bool, Self::Error>;
154
155    /// Return whether the type is a `felt252`, either directly or indirectly (ex. through
156    /// `NonZero<BoundedInt<>>`).
157    fn is_felt252(
158        &self,
159        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
160    ) -> Result<bool, Self::Error>;
161
162    /// If the type is a enum type, return all possible variants.
163    ///
164    /// TODO: How is it used?
165    fn variants(&self) -> Option<&[ConcreteTypeId]>;
166
167    #[allow(clippy::too_many_arguments)]
168    fn build_default<'ctx, 'this>(
169        &self,
170        context: &'ctx Context,
171        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
172        entry: &'this Block<'ctx>,
173        location: Location<'ctx>,
174        helper: &LibfuncHelper<'ctx, 'this>,
175        metadata: &mut MetadataStorage,
176        self_ty: &ConcreteTypeId,
177    ) -> Result<Value<'ctx, 'this>, Self::Error>;
178}

493    fn is_complex(
494        &self,
495        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
496    ) -> Result<bool, Self::Error> {
497        Ok(match self {
498            // Builtins.
499            CoreTypeConcrete::Bitwise(_)
500            | CoreTypeConcrete::EcOp(_)
501            | CoreTypeConcrete::GasBuiltin(_)
502            | CoreTypeConcrete::BuiltinCosts(_)
503            | CoreTypeConcrete::RangeCheck(_)
504            | CoreTypeConcrete::Pedersen(_)
505            | CoreTypeConcrete::Poseidon(_)
506            | CoreTypeConcrete::RangeCheck96(_)
507            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::System(_)) // u64 is not complex
508            | CoreTypeConcrete::SegmentArena(_) => false,
509
510            CoreTypeConcrete::Box(_)
511            | CoreTypeConcrete::Uint8(_)
512            | CoreTypeConcrete::Uint16(_)
513            | CoreTypeConcrete::Uint32(_)
514            | CoreTypeConcrete::Uint64(_)
515            | CoreTypeConcrete::Uint128(_)
516            | CoreTypeConcrete::Uint128MulGuarantee(_)
517            | CoreTypeConcrete::Sint8(_)
518            | CoreTypeConcrete::Sint16(_)
519            | CoreTypeConcrete::Sint32(_)
520            | CoreTypeConcrete::Sint64(_)
521            | CoreTypeConcrete::Sint128(_)
522            | CoreTypeConcrete::Nullable(_)
523            | CoreTypeConcrete::Felt252Dict(_)
524            | CoreTypeConcrete::SquashedFelt252Dict(_) => false,
525
526            CoreTypeConcrete::Array(_) => true,
527            CoreTypeConcrete::EcPoint(_) => true,
528            CoreTypeConcrete::EcState(_) => true,
529            CoreTypeConcrete::Felt252DictEntry(_) => true,
530
531            CoreTypeConcrete::Felt252(_)
532            | CoreTypeConcrete::Bytes31(_)
533            | CoreTypeConcrete::Starknet(
534                StarknetTypeConcrete::ClassHash(_)
535                | StarknetTypeConcrete::ContractAddress(_)
536                | StarknetTypeConcrete::StorageAddress(_)
537                | StarknetTypeConcrete::StorageBaseAddress(_)
538            ) => {
539                #[cfg(target_arch = "x86_64")]
540                let value = true;
541
542                #[cfg(target_arch = "aarch64")]
543                let value = false;
544
545                value
546            },
547
548            CoreTypeConcrete::NonZero(info)
549            | CoreTypeConcrete::Uninitialized(info)
550            | CoreTypeConcrete::Snapshot(info) => registry.get_type(&info.ty)?.is_complex(registry)?,
551
552            CoreTypeConcrete::Enum(info) => match info.variants.len() {
553                0 => false,
554                1 => registry.get_type(&info.variants[0])?.is_complex(registry)?,
555                _ => !self.is_zst(registry)?,
556            },
557            CoreTypeConcrete::Struct(_) => true,
558
559            CoreTypeConcrete::BoundedInt(_info) => {
560                #[cfg(target_arch = "x86_64")]
561                let value = _info.range.offset_bit_width() > 128;
562
563                #[cfg(target_arch = "aarch64")]
564                let value = false;
565
566                value
567            },
568            CoreTypeConcrete::Const(_) => native_panic!("todo: check Const is complex"),
569            CoreTypeConcrete::Span(_) => native_panic!("todo: check Span is complex"),
570            CoreTypeConcrete::Starknet(StarknetTypeConcrete::Secp256Point(_))
571            | CoreTypeConcrete::Starknet(StarknetTypeConcrete::Sha256StateHandle(_)) => native_panic!("todo: check Sha256StateHandle is complex"),
572            CoreTypeConcrete::Coupon(_) => false,
573
574            CoreTypeConcrete::Circuit(info) => circuit::is_complex(info),
575
576            CoreTypeConcrete::IntRange(_info) => false,
577            CoreTypeConcrete::Blake(_info) => native_panic!("Implement is_complex for Blake type"),
578            CoreTypeConcrete::QM31(_info) => native_panic!("Implement is_complex for QM31 type"),
579            CoreTypeConcrete::GasReserve(_info) => native_panic!("Implement is_complex for GasReserve type"),
580        })
581    }

583    fn is_zst(
584        &self,
585        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
586    ) -> Result<bool, Self::Error> {
587        Ok(match self {
588            // Builtin counters:
589            CoreTypeConcrete::Bitwise(_)
590            | CoreTypeConcrete::EcOp(_)
591            | CoreTypeConcrete::RangeCheck(_)
592            | CoreTypeConcrete::Pedersen(_)
593            | CoreTypeConcrete::Poseidon(_)
594            | CoreTypeConcrete::RangeCheck96(_)
595            | CoreTypeConcrete::SegmentArena(_) => false,
596
597            // A ptr to a list of costs.
598            CoreTypeConcrete::BuiltinCosts(_) => false,
599
600            // Other builtins:
601            CoreTypeConcrete::Uint128MulGuarantee(_) | CoreTypeConcrete::Coupon(_) => true,
602
603            // Normal types:
604            CoreTypeConcrete::Array(_)
605            | CoreTypeConcrete::Box(_)
606            | CoreTypeConcrete::Bytes31(_)
607            | CoreTypeConcrete::EcPoint(_)
608            | CoreTypeConcrete::EcState(_)
609            | CoreTypeConcrete::Felt252(_)
610            | CoreTypeConcrete::GasBuiltin(_)
611            | CoreTypeConcrete::Uint8(_)
612            | CoreTypeConcrete::Uint16(_)
613            | CoreTypeConcrete::Uint32(_)
614            | CoreTypeConcrete::Uint64(_)
615            | CoreTypeConcrete::Uint128(_)
616            | CoreTypeConcrete::Sint8(_)
617            | CoreTypeConcrete::Sint16(_)
618            | CoreTypeConcrete::Sint32(_)
619            | CoreTypeConcrete::Sint64(_)
620            | CoreTypeConcrete::Sint128(_)
621            | CoreTypeConcrete::Felt252Dict(_)
622            | CoreTypeConcrete::Felt252DictEntry(_)
623            | CoreTypeConcrete::SquashedFelt252Dict(_)
624            | CoreTypeConcrete::Starknet(_)
625            | CoreTypeConcrete::Nullable(_) => false,
626
627            // Containers:
628            CoreTypeConcrete::NonZero(info)
629            | CoreTypeConcrete::Uninitialized(info)
630            | CoreTypeConcrete::Snapshot(info) => {
631                let type_info = registry.get_type(&info.ty)?;
632                type_info.is_zst(registry)?
633            }
634
635            // Enums and structs:
636            CoreTypeConcrete::Enum(info) => {
637                info.variants.is_empty()
638                    || (info.variants.len() == 1
639                        && registry.get_type(&info.variants[0])?.is_zst(registry)?)
640            }
641            CoreTypeConcrete::Struct(info) => {
642                let mut is_zst = true;
643                for member in &info.members {
644                    if !registry.get_type(member)?.is_zst(registry)? {
645                        is_zst = false;
646                        break;
647                    }
648                }
649                is_zst
650            }
651
652            CoreTypeConcrete::BoundedInt(_) => false,
653            CoreTypeConcrete::Const(info) => {
654                let type_info = registry.get_type(&info.inner_ty)?;
655                type_info.is_zst(registry)?
656            }
657            CoreTypeConcrete::Span(_) => native_panic!("todo: check Span is zero sized"),
658            CoreTypeConcrete::Circuit(info) => circuit::is_zst(info),
659
660            CoreTypeConcrete::IntRange(info) => {
661                let type_info = registry.get_type(&info.ty)?;
662                type_info.is_zst(registry)?
663            }
664            CoreTypeConcrete::Blake(_info) => native_panic!("Implement is_zst for Blake type"),
665            CoreTypeConcrete::QM31(_info) => native_panic!("Implement is_zst for QM31 type"),
666            CoreTypeConcrete::GasReserve(_info) => {
667                native_panic!("Implement is_zst for GasReserve type")
668            }
669        })
670    }

672    fn layout(
673        &self,
674        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
675    ) -> Result<Layout, Self::Error> {
676        Ok(match self {
677            CoreTypeConcrete::Array(_) => {
678                Layout::new::<*mut ()>()
679                    .extend(get_integer_layout(32))?
680                    .0
681                    .extend(get_integer_layout(32))?
682                    .0
683                    .extend(get_integer_layout(32))?
684                    .0
685            }
686            CoreTypeConcrete::Bitwise(_) => Layout::new::<u64>(),
687            CoreTypeConcrete::Box(_) => Layout::new::<*mut ()>(),
688            CoreTypeConcrete::EcOp(_) => Layout::new::<u64>(),
689            CoreTypeConcrete::EcPoint(_) => layout_repeat(&get_integer_layout(252), 2)?.0,
690            CoreTypeConcrete::EcState(_) => layout_repeat(&get_integer_layout(252), 4)?.0,
691            CoreTypeConcrete::Felt252(_) => get_integer_layout(252),
692            CoreTypeConcrete::GasBuiltin(_) => get_integer_layout(64),
693            CoreTypeConcrete::BuiltinCosts(_) => Layout::new::<*const ()>(),
694            CoreTypeConcrete::Uint8(_) => get_integer_layout(8),
695            CoreTypeConcrete::Uint16(_) => get_integer_layout(16),
696            CoreTypeConcrete::Uint32(_) => get_integer_layout(32),
697            CoreTypeConcrete::Uint64(_) => get_integer_layout(64),
698            CoreTypeConcrete::Uint128(_) => get_integer_layout(128),
699            CoreTypeConcrete::Uint128MulGuarantee(_) => Layout::new::<()>(),
700            CoreTypeConcrete::NonZero(info) => registry.get_type(&info.ty)?.layout(registry)?,
701            CoreTypeConcrete::Nullable(_) => Layout::new::<*mut ()>(),
702            CoreTypeConcrete::RangeCheck(_) => Layout::new::<u64>(),
703            CoreTypeConcrete::Uninitialized(info) => {
704                registry.get_type(&info.ty)?.layout(registry)?
705            }
706            CoreTypeConcrete::Enum(info) => {
707                let tag_layout =
708                    get_integer_layout(info.variants.len().next_power_of_two().trailing_zeros());
709
710                info.variants.iter().try_fold(tag_layout, |acc, id| {
711                    let layout = tag_layout
712                        .extend(registry.get_type(id)?.layout(registry)?)?
713                        .0;
714
715                    Result::<_, Self::Error>::Ok(Layout::from_size_align(
716                        acc.size().max(layout.size()),
717                        acc.align().max(layout.align()),
718                    )?)
719                })?
720            }
721            CoreTypeConcrete::Struct(info) => info
722                .members
723                .iter()
724                .try_fold(Option::<Layout>::None, |acc, id| {
725                    Result::<_, Self::Error>::Ok(Some(match acc {
726                        Some(layout) => layout.extend(registry.get_type(id)?.layout(registry)?)?.0,
727                        None => registry.get_type(id)?.layout(registry)?,
728                    }))
729                })?
730                .unwrap_or(Layout::from_size_align(0, 1)?),
731            CoreTypeConcrete::Felt252Dict(_) => Layout::new::<*mut std::ffi::c_void>(), // ptr
732            CoreTypeConcrete::Felt252DictEntry(_) => {
733                get_integer_layout(252)
734                    .extend(Layout::new::<*mut std::ffi::c_void>())?
735                    .0
736                    .extend(Layout::new::<*mut std::ffi::c_void>())?
737                    .0
738            }
739            CoreTypeConcrete::SquashedFelt252Dict(_) => Layout::new::<*mut std::ffi::c_void>(), // ptr
740            CoreTypeConcrete::Pedersen(_) => Layout::new::<u64>(),
741            CoreTypeConcrete::Poseidon(_) => Layout::new::<u64>(),
742            CoreTypeConcrete::Span(_) => native_panic!("todo: create layout for Span"),
743            CoreTypeConcrete::Starknet(info) => match info {
744                StarknetTypeConcrete::ClassHash(_) => get_integer_layout(252),
745                StarknetTypeConcrete::ContractAddress(_) => get_integer_layout(252),
746                StarknetTypeConcrete::StorageBaseAddress(_) => get_integer_layout(252),
747                StarknetTypeConcrete::StorageAddress(_) => get_integer_layout(252),
748                StarknetTypeConcrete::System(_) => Layout::new::<*mut ()>(),
749                StarknetTypeConcrete::Secp256Point(_) => {
750                    get_integer_layout(256)
751                        .extend(get_integer_layout(256))?
752                        .0
753                        .extend(get_integer_layout(1))?
754                        .0
755                }
756                StarknetTypeConcrete::Sha256StateHandle(_) => Layout::new::<*mut ()>(),
757            },
758            CoreTypeConcrete::SegmentArena(_) => Layout::new::<u64>(),
759            CoreTypeConcrete::Snapshot(info) => registry.get_type(&info.ty)?.layout(registry)?,
760            CoreTypeConcrete::Sint8(_) => get_integer_layout(8),
761            CoreTypeConcrete::Sint16(_) => get_integer_layout(16),
762            CoreTypeConcrete::Sint32(_) => get_integer_layout(32),
763            CoreTypeConcrete::Sint64(_) => get_integer_layout(64),
764            CoreTypeConcrete::Sint128(_) => get_integer_layout(128),
765            CoreTypeConcrete::Bytes31(_) => get_integer_layout(248),
766            CoreTypeConcrete::BoundedInt(info) => get_integer_layout(info.range.offset_bit_width()),
767
768            CoreTypeConcrete::Const(const_type) => {
769                registry.get_type(&const_type.inner_ty)?.layout(registry)?
770            }
771            CoreTypeConcrete::Coupon(_) => Layout::new::<()>(),
772            CoreTypeConcrete::RangeCheck96(_) => get_integer_layout(64),
773            CoreTypeConcrete::Circuit(info) => circuit::layout(registry, info)?,
774
775            CoreTypeConcrete::IntRange(info) => {
776                let inner = registry.get_type(&info.ty)?.layout(registry)?;
777                inner.extend(inner)?.0
778            }
779            CoreTypeConcrete::Blake(_info) => native_panic!("Implement layout for Blake type"),
780            CoreTypeConcrete::QM31(_info) => native_panic!("Implement layout for QM31 type"),
781            CoreTypeConcrete::GasReserve(_info) => {
782                native_panic!("Implement layout for GasReserve type")
783            }
784        }
785        .pad_to_align())
786    }

788    fn is_memory_allocated(
789        &self,
790        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
791    ) -> Result<bool, Self::Error> {
792        // Right now, only enums and other structures which may end up passing a flattened enum as
793        // arguments.
794        Ok(match self {
795            CoreTypeConcrete::IntRange(_) => false,
796            CoreTypeConcrete::Blake(_info) => {
797                native_panic!("Implement is_memory_allocated for Blake type")
798            }
799            CoreTypeConcrete::Array(_) => false,
800            CoreTypeConcrete::Bitwise(_) => false,
801            CoreTypeConcrete::Box(_) => false,
802            CoreTypeConcrete::EcOp(_) => false,
803            CoreTypeConcrete::EcPoint(_) => false,
804            CoreTypeConcrete::EcState(_) => false,
805            CoreTypeConcrete::Felt252(_) => false,
806            CoreTypeConcrete::GasBuiltin(_) => false,
807            CoreTypeConcrete::BuiltinCosts(_) => false,
808            CoreTypeConcrete::Uint8(_) => false,
809            CoreTypeConcrete::Uint16(_) => false,
810            CoreTypeConcrete::Uint32(_) => false,
811            CoreTypeConcrete::Uint64(_) => false,
812            CoreTypeConcrete::Uint128(_) => false,
813            CoreTypeConcrete::Uint128MulGuarantee(_) => false,
814            CoreTypeConcrete::Sint8(_) => false,
815            CoreTypeConcrete::Sint16(_) => false,
816            CoreTypeConcrete::Sint32(_) => false,
817            CoreTypeConcrete::Sint64(_) => false,
818            CoreTypeConcrete::Sint128(_) => false,
819            CoreTypeConcrete::NonZero(_) => false,
820            CoreTypeConcrete::Nullable(_) => false,
821            CoreTypeConcrete::RangeCheck(_) => false,
822            CoreTypeConcrete::RangeCheck96(_) => false,
823            CoreTypeConcrete::Uninitialized(_) => false,
824            CoreTypeConcrete::Enum(info) => match info.variants.len() {
825                0 => false,
826                1 => registry
827                    .get_type(&info.variants[0])?
828                    .is_memory_allocated(registry)?,
829                _ => true,
830            },
831            CoreTypeConcrete::Struct(info) => {
832                let mut is_memory_allocated = false;
833                for member in &info.members {
834                    if registry.get_type(member)?.is_memory_allocated(registry)? {
835                        is_memory_allocated = true;
836                        break;
837                    }
838                }
839                is_memory_allocated
840            }
841            CoreTypeConcrete::Felt252Dict(_) => false,
842            CoreTypeConcrete::Felt252DictEntry(_) => false,
843            CoreTypeConcrete::SquashedFelt252Dict(_) => false,
844            CoreTypeConcrete::Pedersen(_) => false,
845            CoreTypeConcrete::Poseidon(_) => false,
846            CoreTypeConcrete::Span(_) => false,
847            CoreTypeConcrete::Starknet(_) => false,
848            CoreTypeConcrete::SegmentArena(_) => false,
849            CoreTypeConcrete::Snapshot(info) => {
850                registry.get_type(&info.ty)?.is_memory_allocated(registry)?
851            }
852            CoreTypeConcrete::Bytes31(_) => false,
853
854            CoreTypeConcrete::BoundedInt(_) => false,
855            CoreTypeConcrete::Const(info) => registry
856                .get_type(&info.inner_ty)?
857                .is_memory_allocated(registry)?,
858            CoreTypeConcrete::Coupon(_) => false,
859            CoreTypeConcrete::Circuit(_) => false,
860            CoreTypeConcrete::QM31(_) => native_panic!("Implement is_memory_allocated for QM31"),
861            CoreTypeConcrete::GasReserve(_) => {
862                native_panic!("Implement is_memory_allocated for GasReserve")
863            }
864        })
865    }

867    fn integer_range(
868        &self,
869        registry: &ProgramRegistry<CoreType, CoreLibfunc>,
870    ) -> Result<Range, Self::Error> {
871        fn range_of<T>() -> Range
872        where
873            T: Bounded + Into<BigInt>,
874        {
875            Range {
876                lower: T::min_value().into(),
877                upper: T::max_value().into() + BigInt::one(),
878            }
879        }
880
881        Ok(match self {
882            Self::Uint8(_) => range_of::<u8>(),
883            Self::Uint16(_) => range_of::<u16>(),
884            Self::Uint32(_) => range_of::<u32>(),
885            Self::Uint64(_) => range_of::<u64>(),
886            Self::Uint128(_) => range_of::<u128>(),
887            Self::Felt252(_) => Range {
888                lower: BigInt::ZERO,
889                upper: BigInt::from_biguint(Sign::Plus, PRIME.clone()),
890            },
891            Self::Sint8(_) => range_of::<i8>(),
892            Self::Sint16(_) => range_of::<i16>(),
893            Self::Sint32(_) => range_of::<i32>(),
894            Self::Sint64(_) => range_of::<i64>(),
895            Self::Sint128(_) => range_of::<i128>(),
896
897            Self::BoundedInt(info) => info.range.clone(),
898            Self::Bytes31(_) => Range {
899                lower: BigInt::ZERO,
900                upper: BigInt::one() << 248,
901            },
902            Self::Const(info) => registry.get_type(&info.inner_ty)?.integer_range(registry)?,
903            Self::NonZero(info) => registry.get_type(&info.ty)?.integer_range(registry)?,
904
905            _ => return Err(crate::error::Error::IntegerLikeTypeExpected),
906        })
907    }

942    fn build_default<'ctx, 'this>(
943        &self,
944        context: &'ctx Context,
945        _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
946        entry: &'this Block<'ctx>,
947        location: Location<'ctx>,
948        _helper: &LibfuncHelper<'ctx, 'this>,
949        _metadata: &mut MetadataStorage,
950        _self_ty: &ConcreteTypeId,
951    ) -> Result<Value<'ctx, 'this>, Self::Error> {
952        static BOOL_USER_TYPE_ID: OnceLock<UserTypeId> = OnceLock::new();
953        let bool_user_type_id =
954            BOOL_USER_TYPE_ID.get_or_init(|| UserTypeId::from_string("core::bool"));
955
956        Ok(match self {
957            Self::Enum(info) => match &info.info.long_id.generic_args[0] {
958                GenericArg::UserType(id) if id == bool_user_type_id => {
959                    let tag = entry.const_int(context, location, 0, 1)?;
960
961                    let value = entry.append_op_result(llvm::undef(
962                        llvm::r#type::r#struct(
963                            context,
964                            &[
965                                IntegerType::new(context, 1).into(),
966                                llvm::r#type::array(IntegerType::new(context, 8).into(), 0),
967                            ],
968                            false,
969                        ),
970                        location,
971                    ))?;
972
973                    entry.insert_value(context, location, value, tag, 0)?
974                }
975                _ => native_panic!("unsupported dict value type"),
976            },
977            Self::Felt252(_) => entry.const_int(context, location, 0, 252)?,
978            Self::Nullable(_) => {
979                entry.append_op_result(llvm::zero(llvm::r#type::pointer(context, 0), location))?
980            }
981            Self::Uint8(_) => entry.const_int(context, location, 0, 8)?,
982            Self::Uint16(_) => entry.const_int(context, location, 0, 16)?,
983            Self::Uint32(_) => entry.const_int(context, location, 0, 32)?,
984            Self::Uint64(_) => entry.const_int(context, location, 0, 64)?,
985            Self::Uint128(_) => entry.const_int(context, location, 0, 128)?,
986            _ => native_panic!("unsupported dict value type"),
987        })
988    }
