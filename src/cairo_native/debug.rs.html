435pub fn generic_type_to_name(
436    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
437    name: &str,
438    args: &[ConcreteTypeId],
439) -> String {
440    format!(
441        "{}<{}>",
442        name,
443        args.iter()
444            .map(|field_type| {
445                registry
446                    .get_type(field_type)
447                    .expect("failed to find type in registry")
448            })
449            .map(|field_type| type_to_name(registry, field_type))
450            .join(",")
451    )
452}
