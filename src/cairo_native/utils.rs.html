76impl BuiltinCosts {
77    pub fn index_for_token_type(token_type: &CostTokenType) -> NativeResult<usize> {
78        let index = match token_type {
79            CostTokenType::Const => 0,
80            CostTokenType::Pedersen => 1,
81            CostTokenType::Bitwise => 2,
82            CostTokenType::EcOp => 3,
83            CostTokenType::Poseidon => 4,
84            CostTokenType::AddMod => 5,
85            CostTokenType::MulMod => 6,
86            _ => native_panic!("matched an unexpected CostTokenType which is not being used"),
87        };
88
89        Ok(index)
90    }

137/// Generate a function name.
138///
139/// If the program includes function identifiers, return those. Otherwise return `f` followed by the
140/// identifier number.
141pub fn generate_function_name(
142    function_id: &FunctionId,
143    is_for_contract_executor: bool,
144) -> Cow<str> {
145    // Generic functions can omit their type in the debug_name, leading to multiple functions
146    // having the same name, we solve this by adding the id number even if the function has a debug_name
147
148    if is_for_contract_executor {
149        Cow::Owned(format!("f{}", function_id.id))
150    } else if let Some(name) = function_id.debug_name.as_deref() {
151        Cow::Owned(format!("{}(f{})", name, function_id.id))
152    } else {
153        Cow::Owned(format!("f{}", function_id.id))
154    }
155}

157/// Decode an UTF-8 error message replacing invalid bytes with their hexadecimal representation, as
158/// done by Python's `x.decode('utf-8', errors='backslashreplace')`.
159pub fn decode_error_message(data: &[u8]) -> String {
160    let mut pos = 0;
161    utf8_iter::ErrorReportingUtf8Chars::new(data).fold(String::new(), |mut acc, ch| {
162        match ch {
163            Ok(ch) => {
164                acc.push(ch);
165                pos += ch.len_utf8();
166            }
167            Err(_) => {
168                acc.push_str(&format!("\\x{:02x}", data[pos]));
169                pos += 1;
170            }
171        };
172
173        acc
174    })
175}

177/// Return the layout for an integer of arbitrary width.
178///
179/// This assumes the platform's maximum (effective) alignment is 16 bytes, and that every integer
180/// with a size in bytes of a power of two has the same alignment as its size.
181pub fn get_integer_layout(width: u32) -> Layout {
182    if width == 0 {
183        Layout::new::<()>()
184    } else if width <= 8 {
185        Layout::new::<u8>()
186    } else if width <= 16 {
187        Layout::new::<u16>()
188    } else if width <= 32 {
189        Layout::new::<u32>()
190    } else if width <= 64 {
191        Layout::new::<u64>()
192    } else if width <= 128 {
193        Layout::new::<u128>()
194    } else {
195        // According to the docs this should never return an error.
196        Layout::from_size_align((width as usize).next_multiple_of(8) >> 3, 16)
197            .expect("layout size rounded up to the next multiple of 16 should never be greater than ISIZE::MAX")
198    }
199}

201/// Compile a cairo program found at the given path to sierra.
202pub fn cairo_to_sierra(program: &Path) -> crate::error::Result<Arc<Program>> {
203    if program
204        .extension()
205        .map(|x| {
206            x.to_ascii_lowercase()
207                .to_string_lossy()
208                .eq_ignore_ascii_case("cairo")
209        })
210        .unwrap_or(false)
211    {
212        cairo_lang_compiler::compile_cairo_project_at_path(
213            program,
214            CompilerConfig {
215                replace_ids: true,
216                ..Default::default()
217            },
218        )
219        .map_err(|err| crate::error::Error::ProgramParser(err.to_string()))
220    } else {
221        let source = std::fs::read_to_string(program)?;
222        cairo_lang_sierra::ProgramParser::new()
223            .parse(&source)
224            .map_err(|err| crate::error::Error::ProgramParser(err.to_string()))
225    }
226    .map(Arc::new)
227}

262/// Parse a numeric string into felt, wrapping negatives around the prime modulo.
263pub fn felt252_str(value: &str) -> Felt {
264    let value = value
265        .parse::<BigInt>()
266        .expect("value must be a digit number");
267
268    let value = match value.sign() {
269        Sign::Minus => &*PRIME - value.magnitude(),
270        _ => value.magnitude().clone(),
271    };
272
273    value.into()
274}

276/// Parse any type that can be a bigint to a felt that can be used in the cairo-native input.
277pub fn felt252_bigint(value: impl Into<BigInt>) -> Felt {
278    let value: BigInt = value.into();
279    let value = match value.sign() {
280        Sign::Minus => Cow::Owned(&*PRIME - value.magnitude()),
281        _ => Cow::Borrowed(value.magnitude()),
282    };
283
284    value.as_ref().into()
285}

343/// Edit: Copied from the std lib.
344///
345/// Returns the amount of padding we must insert after `layout`
346/// to ensure that the following address will satisfy `align`
347/// (measured in bytes).
348///
349/// e.g., if `layout.size()` is 9, then `layout.padding_needed_for(4)`
350/// returns 3, because that is the minimum number of bytes of
351/// padding required to get a 4-aligned address (assuming that the
352/// corresponding memory block starts at a 4-aligned address).
353///
354/// The return value of this function has no meaning if `align` is
355/// not a power-of-two.
356///
357/// Note that the utility of the returned value requires `align`
358/// to be less than or equal to the alignment of the starting
359/// address for the whole allocated block of memory. One way to
360/// satisfy this constraint is to ensure `align <= layout.align()`.
361#[inline]
362pub const fn padding_needed_for(layout: &Layout, align: usize) -> usize {
363    let len = layout.size();
364
365    // Rounded up value is:
366    //   len_rounded_up = (len + align - 1) & !(align - 1);
367    // and then we return the padding difference: `len_rounded_up - len`.
368    //
369    // We use modular arithmetic throughout:
370    //
371    // 1. align is guaranteed to be > 0, so align - 1 is always
372    //    valid.
373    //
374    // 2. `len + align - 1` can overflow by at most `align - 1`,
375    //    so the &-mask with `!(align - 1)` will ensure that in the
376    //    case of overflow, `len_rounded_up` will itself be 0.
377    //    Thus the returned padding, when added to `len`, yields 0,
378    //    which trivially satisfies the alignment `align`.
379    //
380    // (Of course, attempts to allocate blocks of memory whose
381    // size and padding overflow in the above manner should cause
382    // the allocator to yield an error anyway.)
383
384    let len_rounded_up = len.wrapping_add(align).wrapping_sub(1) & !align.wrapping_sub(1);
385    len_rounded_up.wrapping_sub(len)
386}

397/// Copied from std.
398///
399/// Creates a layout describing the record for `n` instances of
400/// `self`, with a suitable amount of padding between each to
401/// ensure that each instance is given its requested size and
402/// alignment. On success, returns `(k, offs)` where `k` is the
403/// layout of the array and `offs` is the distance between the start
404/// of each element in the array.
405///
406/// On arithmetic overflow, returns `LayoutError`.
407//#[unstable(feature = "alloc_layout_extra", issue = "55724")]
408#[inline]
409pub fn layout_repeat(layout: &Layout, n: usize) -> Result<(Layout, usize), LayoutError> {
410    // This cannot overflow. Quoting from the invariant of Layout:
411    // > `size`, when rounded up to the nearest multiple of `align`,
412    // > must not overflow isize (i.e., the rounded value must be
413    // > less than or equal to `isize::MAX`)
414    let padded_size = layout.size() + padding_needed_for(layout, layout.align());
415    let alloc_size = padded_size.checked_mul(n).ok_or(LayoutError)?;
416
417    // The safe constructor is called here to enforce the isize size limit.
418    let layout = Layout::from_size_align(alloc_size, layout.align()).map_err(|_| LayoutError)?;
419    Ok((layout, padded_size))
420}

592    pub fn run_program(
593        program: &(String, Program),
594        entry_point: &str,
595        args: &[Value],
596    ) -> ExecutionResult {
597        let entry_point = format!("{0}::{0}::{1}", program.0, entry_point);
598        let program = &program.1;
599
600        let entry_point_id = &program
601            .funcs
602            .iter()
603            .find(|x| x.id.debug_name.as_deref() == Some(&entry_point))
604            .expect("Test program entry point not found.")
605            .id;
606
607        let context = NativeContext::new();
608
609        let module = context
610            .compile(program, false, Some(Default::default()))
611            .expect("Could not compile test program to MLIR.");
612
613        let executor = JitNativeExecutor::from_native_module(module, OptLevel::Less).unwrap();
614        executor
615            .invoke_dynamic_with_syscall_handler(
616                entry_point_id,
617                args,
618                Some(u64::MAX),
619                &mut StubSyscallHandler::default(),
620            )
621            .unwrap()
622    }
