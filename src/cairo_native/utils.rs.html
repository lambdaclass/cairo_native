79impl BuiltinCosts {
80    pub fn index_for_token_type(token_type: &CostTokenType) -> NativeResult<usize> {
81        let index = match token_type {
82            CostTokenType::Const => 0,
83            CostTokenType::Pedersen => 1,
84            CostTokenType::Bitwise => 2,
85            CostTokenType::EcOp => 3,
86            CostTokenType::Poseidon => 4,
87            CostTokenType::AddMod => 5,
88            CostTokenType::MulMod => 6,
89            _ => native_panic!("matched an unexpected CostTokenType which is not being used"),
90        };
91
92        Ok(index)
93    }

140/// Generate a function name.
141///
142/// If the program includes function identifiers, return those. Otherwise return `f` followed by the
143/// identifier number.
144pub fn generate_function_name(
145    function_id: &'_ FunctionId,
146    is_for_contract_executor: bool,
147) -> Cow<'_, str> {
148    // Generic functions can omit their type in the debug_name, leading to multiple functions
149    // having the same name, we solve this by adding the id number even if the function has a debug_name
150
151    if is_for_contract_executor {
152        Cow::Owned(format!("f{}", function_id.id))
153    } else if let Some(name) = function_id.debug_name.as_deref() {
154        Cow::Owned(format!("{}(f{})", name, function_id.id))
155    } else {
156        Cow::Owned(format!("f{}", function_id.id))
157    }
158}

160/// Decode an UTF-8 error message replacing invalid bytes with their hexadecimal representation, as
161/// done by Python's `x.decode('utf-8', errors='backslashreplace')`.
162pub fn decode_error_message(data: &[u8]) -> String {
163    let mut pos = 0;
164    utf8_iter::ErrorReportingUtf8Chars::new(data).fold(String::new(), |mut acc, ch| {
165        match ch {
166            Ok(ch) => {
167                acc.push(ch);
168                pos += ch.len_utf8();
169            }
170            Err(_) => {
171                acc.push_str(&format!("\\x{:02x}", data[pos]));
172                pos += 1;
173            }
174        };
175
176        acc
177    })
178}

180/// Return the layout for an integer of arbitrary width.
181///
182/// This assumes the platform's maximum (effective) alignment is 16 bytes, and that every integer
183/// with a size in bytes of a power of two has the same alignment as its size.
184pub fn get_integer_layout(width: u32) -> Layout {
185    if width == 0 {
186        Layout::new::<()>()
187    } else if width <= 8 {
188        Layout::new::<u8>()
189    } else if width <= 16 {
190        Layout::new::<u16>()
191    } else if width <= 32 {
192        Layout::new::<u32>()
193    } else if width <= 64 {
194        Layout::new::<u64>()
195    } else if width <= 128 {
196        Layout::new::<u128>()
197    } else {
198        // According to the docs this should never return an error.
199        Layout::from_size_align((width as usize).next_multiple_of(8) >> 3, 16)
200            .expect("layout size rounded up to the next multiple of 16 should never be greater than ISIZE::MAX")
201    }
202}

237/// Parse a numeric string into felt, wrapping negatives around the prime modulo.
238pub fn felt252_str(value: &str) -> Felt {
239    let value = value
240        .parse::<BigInt>()
241        .expect("value must be a digit number");
242
243    let value = match value.sign() {
244        Sign::Minus => &*PRIME - value.magnitude(),
245        _ => value.magnitude().clone(),
246    };
247
248    value.into()
249}
250
251/// Parse any type that can be a bigint to a felt that can be used in the cairo-native input.
252pub fn felt252_bigint(value: impl Into<BigInt>) -> Felt {
253    let value: BigInt = value.into();
254    let value = match value.sign() {
255        Sign::Minus => Cow::Owned(&*PRIME - value.magnitude()),
256        _ => Cow::Borrowed(value.magnitude()),
257    };
258
259    value.as_ref().into()
260}

262/// Parse a short string into a felt that can be used in the cairo-native input.
263pub fn felt252_short_str(value: &str) -> Felt {
264    let values: Vec<_> = value
265        .chars()
266        .filter_map(|c| c.is_ascii().then_some(c as u8))
267        .collect();
268
269    assert!(values.len() < 32, "A felt can't longer than 32 bytes");
270    Felt::from_bytes_be_slice(&values)
271}

318/// Edit: Copied from the std lib.
319///
320/// Returns the amount of padding we must insert after `layout`
321/// to ensure that the following address will satisfy `align`
322/// (measured in bytes).
323///
324/// e.g., if `layout.size()` is 9, then `layout.padding_needed_for(4)`
325/// returns 3, because that is the minimum number of bytes of
326/// padding required to get a 4-aligned address (assuming that the
327/// corresponding memory block starts at a 4-aligned address).
328///
329/// The return value of this function has no meaning if `align` is
330/// not a power-of-two.
331///
332/// Note that the utility of the returned value requires `align`
333/// to be less than or equal to the alignment of the starting
334/// address for the whole allocated block of memory. One way to
335/// satisfy this constraint is to ensure `align <= layout.align()`.
336#[inline]
337pub const fn padding_needed_for(layout: &Layout, align: usize) -> usize {
338    let len = layout.size();
339
340    // Rounded up value is:
341    //   len_rounded_up = (len + align - 1) & !(align - 1);
342    // and then we return the padding difference: `len_rounded_up - len`.
343    //
344    // We use modular arithmetic throughout:
345    //
346    // 1. align is guaranteed to be > 0, so align - 1 is always
347    //    valid.
348    //
349    // 2. `len + align - 1` can overflow by at most `align - 1`,
350    //    so the &-mask with `!(align - 1)` will ensure that in the
351    //    case of overflow, `len_rounded_up` will itself be 0.
352    //    Thus the returned padding, when added to `len`, yields 0,
353    //    which trivially satisfies the alignment `align`.
354    //
355    // (Of course, attempts to allocate blocks of memory whose
356    // size and padding overflow in the above manner should cause
357    // the allocator to yield an error anyway.)
358
359    let len_rounded_up = len.wrapping_add(align).wrapping_sub(1) & !align.wrapping_sub(1);
360    len_rounded_up.wrapping_sub(len)
361}

372/// Copied from std.
373///
374/// Creates a layout describing the record for `n` instances of
375/// `self`, with a suitable amount of padding between each to
376/// ensure that each instance is given its requested size and
377/// alignment. On success, returns `(k, offs)` where `k` is the
378/// layout of the array and `offs` is the distance between the start
379/// of each element in the array.
380///
381/// On arithmetic overflow, returns `LayoutError`.
382//#[unstable(feature = "alloc_layout_extra", issue = "55724")]
383#[inline]
384pub fn layout_repeat(layout: &Layout, n: usize) -> Result<(Layout, usize), LayoutError> {
385    // This cannot overflow. Quoting from the invariant of Layout:
386    // > `size`, when rounded up to the nearest multiple of `align`,
387    // > must not overflow isize (i.e., the rounded value must be
388    // > less than or equal to `isize::MAX`)
389    let padded_size = layout.size() + padding_needed_for(layout, layout.align());
390    let alloc_size = padded_size.checked_mul(n).ok_or(LayoutError)?;
391
392    // The safe constructor is called here to enforce the isize size limit.
393    let layout = Layout::from_size_align(alloc_size, layout.align()).map_err(|_| LayoutError)?;
394    Ok((layout, padded_size))
395}

397/// Returns the total layout size for the given types.
398pub fn get_types_total_size(
399    types_ids: &[ConcreteTypeId],
400    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
401) -> crate::error::Result<usize> {
402    let mut total_size = 0;
403    for type_id in types_ids {
404        let type_concrete = registry.get_type(type_id)?;
405        let layout = type_concrete.layout(registry)?;
406        total_size += layout.size();
407    }
408    Ok(total_size)
409}
