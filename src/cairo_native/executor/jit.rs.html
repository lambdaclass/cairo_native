24/// A MLIR JIT execution engine in the context of Cairo Native.
25pub struct JitNativeExecutor<'m> {
26    engine: ExecutionEngine,
27
28    module: Module<'m>,
29    registry: ProgramRegistry<CoreType, CoreLibfunc>,
30
31    gas_metadata: GasMetadata,
32    dict_overrides: Felt252DictOverrides,
33}
34
35unsafe impl Send for JitNativeExecutor<'_> {}
36unsafe impl Sync for JitNativeExecutor<'_> {}
37
38impl std::fmt::Debug for JitNativeExecutor<'_> {
39    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
40        f.debug_struct("JitNativeExecutor")
41            .field("module", &self.module)
42            .field("gas_metadata", &self.gas_metadata)
43            .finish()
44    }
45}

47impl<'m> JitNativeExecutor<'m> {
48    pub fn from_native_module(
49        native_module: NativeModule<'m>,
50        opt_level: OptLevel,
51    ) -> Result<Self, Error> {
52        let NativeModule {
53            module,
54            registry,
55            mut metadata,
56        } = native_module;
57
58        let executor = Self {
59            engine: create_engine(&module, &metadata, opt_level),
60            module,
61            registry,
62            gas_metadata: metadata.remove().ok_or(Error::MissingMetadata)?,
63            dict_overrides: metadata.remove().unwrap_or_default(),
64        };
65
66        setup_runtime(|name| executor.find_symbol_ptr(name));
67
68        #[cfg(feature = "with-debug-utils")]
69        crate::metadata::debug_utils::setup_runtime(|name| executor.find_symbol_ptr(name));
70
71        Ok(executor)
72    }

82    /// Execute a program with the given params.
83    pub fn invoke_dynamic(
84        &self,
85        function_id: &FunctionId,
86        args: &[Value],
87        gas: Option<u64>,
88    ) -> Result<ExecutionResult, Error> {
89        let available_gas = self
90            .gas_metadata
91            .get_initial_available_gas(function_id, gas)
92            .map_err(crate::error::Error::GasMetadataError)?;
93
94        super::invoke_dynamic(
95            &self.registry,
96            self.find_function_ptr(function_id),
97            self.extract_signature(function_id)?,
98            args,
99            available_gas,
100            Option::<DummySyscallHandler>::None,
101            self.build_find_dict_drop_override(),
102        )
103    }

129    pub fn invoke_contract_dynamic(
130        &self,
131        function_id: &FunctionId,
132        args: &[Felt],
133        gas: Option<u64>,
134        syscall_handler: impl StarknetSyscallHandler,
135    ) -> Result<ContractExecutionResult, Error> {
136        let available_gas = self
137            .gas_metadata
138            .get_initial_available_gas(function_id, gas)
139            .map_err(crate::error::Error::GasMetadataError)?;
140
141        ContractExecutionResult::from_execution_result(super::invoke_dynamic(
142            &self.registry,
143            self.find_function_ptr(function_id),
144            self.extract_signature(function_id)?,
145            &[Value::Struct {
146                fields: vec![Value::Array(
147                    args.iter().cloned().map(Value::Felt252).collect(),
148                )],
149                debug_name: None,
150            }],
151            available_gas,
152            Some(syscall_handler),
153            self.build_find_dict_drop_override(),
154        )?)
155    }

157    pub fn find_function_ptr(&self, function_id: &FunctionId) -> *mut c_void {
158        let function_name = generate_function_name(function_id, false);
159        let function_name = format!("_mlir_ciface_{function_name}");
160
161        // Arguments and return values are hardcoded since they'll be handled by the trampoline.
162        self.engine.lookup(&function_name) as *mut c_void
163    }
164
165    pub fn find_symbol_ptr(&self, name: &str) -> Option<*mut c_void> {
166        let ptr = self.engine.lookup(name) as *mut c_void;
167
168        if ptr.is_null() {
169            None
170        } else {
171            Some(ptr)
172        }
173    }
174
175    fn extract_signature(&self, function_id: &FunctionId) -> Result<&FunctionSignature, Error> {
176        Ok(self
177            .program_registry()
178            .get_function(function_id)
179            .map(|func| &func.signature)?)
180    }
181
182    fn build_find_dict_drop_override(
183        &self,
184    ) -> impl '_ + Copy + Fn(&ConcreteTypeId) -> Option<extern "C" fn(*mut c_void)> {
185        |type_id| {
186            self.dict_overrides
187                .get_drop_fn(type_id)
188                .and_then(|symbol| self.find_symbol_ptr(symbol))
189                .map(|ptr| unsafe { transmute(ptr as *const ()) })
190        }
191    }
